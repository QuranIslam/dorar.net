الفَصْلُ التَّاسِعُ: حَرْفُ السِّينِ

محتويات الصفحة
انظر أيضا

سِجِّيلٌ:
قالَ تعالى: فَلَمَّا جَاءَ أَمْرُنَا جَعَلْنَا عَالِيَهَا سَافِلَهَا وَأَمْطَرْنَا عَلَيْهَا حِجَارَةً مِنْ سِجِّيلٍ مَنْضُودٍ [هود: 82] .
مَعْناها واللُّغةُ الَّتي نُقِلَتْ مِنها:
السِّجِّيلُ: حِجارةٌ مِن طينٍ [520] ينظر: ((تفسير مجاهد)) (ص: 390)، ((تفسير الطبري = جامع البيان)) (14/ 94)، ((معاني القرآن وإعرابه)) للزجاج (3/ 184). .
واللَّفْظُ مَنْقولٌ مِن الفارِسيَّةِ؛ مَأخوذٌ مِن "سنگ" أي: حَجَرٍ، و"گل" أي: طينٍ، وتَعْني مُجْتمِعةً: حِجارةٌ كالطِّينِ اليابِسِ [521] يُنظر: ((تفسير غريب القرآن)) لابن قتيبة (ص: 207)، ((المهذب فيما وقع في القرآن من المعرَّب)) للسيوطي (ص: 97)، ((الأصل والبيان في معرب القرآن)) لحمزة فتح الله (ص: 13). .
سِجِّينٌ:
وَرَدَتْ هذه المُفرَدةُ في قَوْلِه تعالى: كَلَّا إِنَّ كِتَابَ الفُجَّارِ لَفِي سِجِّينٍ * وَمَا أَدْرَاكَ مَا سِجِّينٌ [المُطَفِّفينَ: 7، 8].
مَعْناها واللُّغةُ الَّتي نُقِلَتْ مِنها:
اخْتَلَفَ المُفَسِّرونَ فيها؛ فقيلَ: الأرْضُ السَّابِعةُ، وقيلَ: بِئْرٌ في جَهَنَّمَ، وقيلَ: هو فِعِّيلٌ مُشتَقٌّ مِن السَّجْنِ [522] ينظر: ((تفسير الطبري = جامع البيان)) (24/ 193)، ((معاني القرآن وإعرابه)) للزجاج (5/ 298). .
والكَلِمةُ مُعرَّبةٌ مِن اللُّغةِ الفارِسيَّةِ، وتَأتي فيها بمَعْنى الدَّائِمِ والثَّابِتِ والشَّديدِ، واسمِ مَكانٍ في جَهَنَّمَ [523] يُنظر: ((قاموس الفارِسيَّة)) لعبد النعيم محمد حسنين (ص: 356)، ((المعرب في القرآن الكريم)) لمحمد السيد (ص: 225). .
السِّجِلُّ:
قالَ اللهُ تعالى: يَوْمَ نَطْوِي السَّمَاءَ كَطَيِّ السِّجِلِّ لِلْكُتُبِ الآية [الأنبياء: 104].
مَعْناها واللُّغةُ الَّتي نُقِلَتْ مِنها:
يُطلَقُ على عِدَّةِ مَعانٍ؛ منها: السِّجِلُّ: الكِتابُ، وقيلَ: العَهْدُ، وقيلَ: الصَّحيفةُ الَّتي فيها الكِتابُ، وقيلَ: مَلِكٌ، وقيلَ: السِّجِلُّ: الرَّجُلُ [524] يُنظر: ((تفسير غريب القرآن)) لابن قتيبة (ص: 288)، ((البرهان)) للزركشي (1/ 288)، ((لسان العَرب)) لابن منظور (11/ 326)، ((مُعجَم غريب القرآن)) لمحمد فؤاد عبد الباقي (ص: 86). .
وفي أصْلِ الكَلِمةِ: قيلَ: حَبَشيَّةٌ، ومَعْناها: الرَّجُلُ [525] يُنظر: ((لسان العَرب)) لابن منظور (11/ 326)، ((المهذب فيما وقع في القرآن من المعرَّب)) للسيوطي (ص: 96). . وقيلَ: فارِسيَّةٌ، ومَعْناها: السِّجِلُّ الَّذي يَكتُبُ القاضي فيه صُوَرَ الدَّعاوى والمُسْتَنداتِ والأحْكامِ [526] يُنظر: ((المهذب فيما وقع في القرآن من المعرَّب)) للسيوطي (ص: 96)، ((قاموس الفارِسيَّة)) لعبد النعيم محمد حسنين (ص: 356). .
وقيلَ: لاتينيَّةٌ، أصْلُها: sigillum، بمَعْنى المَرْسومِ المَلَكيِّ، ثُمَّ انْتَقَلَ إلى اليونانيَّةِ فصارَ سِكِلُّن، ثُمَّ انْتَقَلَ إلى السُّرْيانيَّةِ: "سيكليون" بمَعْنى الوَثيقةِ، ومِنها إلى العَربيَّةِ [527] ينظر: ((المعرَّب والدَّخيل في اللُّغة العَربيَّة)) للسبحان (ص: 279)، ((المعرَّب في القرآن الكريم)) لمحمد السيد (ص: 226). .
سِرَاجٌ:
قالَ اللهُ تعالى: تَبَارَكَ الَّذِي جَعَلَ فِي السَّمَاءِ بُرُوجًا وَجَعَلَ فِيهَا سِرَاجًا وَقَمَرًا مُنِيرًا [الفُرْقان: 61] .
مَعْناها واللُّغةُ الَّتي نُقِلَتْ مِنها:
المِصْباحُ [528] ينظر: ((تفسير الطبري = جامع البيان)) (17/ 304)، ((معاني القرآن وإعرابه)) للزجاج (4/ 43). .
وفي أصْلِها: قيلَ: فارِسيَّةٌ، وأصْلُها: "چِراغ" بالغَيْنِ، ولمَّا تَعرَّبَتْ تَحوَّلَتِ الجيمُ القَريبةُ مِن الشِّينِ إلى سينٍ، فصارَتْ سِراجًا، ومَعْناها: المِصْباحُ [529] يُنظر: ((التَّطوُّر النحوي للغة العَربيَّة)) للمستشرق الألماني برجشتراسر (ص: 216)، ((قاموس الفارِسيَّة)) لعبد النعيم محمد حسنين (ص: 359). .
وقيلَ: بل هي آراميَّةٌ في الأصْلِ، أصْلُها: سراجَا، وعُرِّبَتْ مُباشَرةً مِن الآراميَّةِ؛ نَظَرًا لقُرْبِها في اللَّفْظِ [530] يُنظر: ((المعرَّب في القرآن الكريم)) لمحمد السيد (ص: 228). .
سُرادِقٌ:
قالَ اللهُ تعالى: إِنَّا أَعْتَدْنَا لِلظَّالِمِينَ نَارًا أَحَاطَ بِهِمْ سُرَادِقُهَا [الكهف: 29] .
مَعْناها واللُّغةُ الَّتي نُقِلَتْ مِنها:
لها عِدَّةُ مَعانٍ، هي: السُّورُ والحائِطُ الَّذي يُحيطُ بالبِناءِ، أو الغُبارُ والدُّخَانُ واللَّهيبُ المُتَصاعِدُ، أو البَحْرُ المُحيطُ بالدُّنْيا [531] يُنظر: ((تفسير الطبري = جامع البيان)) (15/ 245)، ((تفسير الماوردي = النكت والعيون)) (3/ 303)، ((لسان العَرب)) لابن منظور (10/ 158). .
والكَلِمةُ مُعرَّبةٌ مِن الفارِسيَّةِ، لكنَّهم اخْتَلَفوا في أصْلِ الكَلِمةِ الَّتي عُرِّبَتْ مِنها؛ فقيلَ: مِن "سرادار" بمَعْنى الدِّهْليزِ [532] يُنظر: ((المعرَّب)) للجواليقي (ص: 200). ، وقيلَ: بلْ مِن "سرابرده" بمَعْنى سِتْرِ الدَّارِ [533] يُنظر: ((المهذب فيما وقع في القرآن من المعرَّب)) للسيوطي (ص: 98). ، وقيلَ: مِن "سرا وسراي" بمَعْنى البَيْتِ والقَصْرِ والبِناءِ العالي [534] يُنظر: ((المعرَّب والدَّخيل في اللُّغة العَربيَّة)) للسبحان (ص: 292). . وفي قاموسِ الفارِسيَّةِ: السُّرادِقُ: الخَيمةُ الكَبيرةُ، والسِّتارةُ الكَبيرةُ الَّتي يَجْعَلونَها فوقَ صَحْنِ المَنزِلِ، والدُّخانُ أو الغُبارُ الَّذي يَثورُ حَوْلَ أطْرافِ الشَّيءِ [535] يُنظر: ((قاموس الفارِسيَّة)) لعبد النعيم محمد حسن (ص: 359)، ((المعرَّب في القرآن الكريم)) لمحمد السيد (ص: 229). .
سِرْبالٌ:
قالَ اللهُ تعالى: وَاللَّهُ جَعَلَ لَكُمْ مِمَّا خَلَقَ ظِلَالًا وَجَعَلَ لَكُمْ مِنَ الْجِبَالِ أَكْنَانًا وَجَعَلَ لَكُمْ سَرَابِيلَ تَقِيكُمُ الْحَرَّ وَسَرَابِيلَ تَقِيكُمْ بَأْسَكُمْ كَذَلِكَ يُتِمُّ نِعْمَتَهُ عَلَيْكُمْ لَعَلَّكُمْ تُسْلِمُونَ [النَّحْلُ: 81] .
مَعْناها واللُّغةُ الَّتي نُقِلَتْ مِنها:
قيلَ: هي القَميصُ، وقيلَ: الدِّرْعُ، وقيلَ: كلُّ ما لُبِسَ فهو سِرْبالٌ [536] يُنظر: ((تفسير الطبري = جامع البيان)) (14/ 321)، ((معاني القرآن وإعرابه)) للزجاج (3/ 215)، ((لسان العَرب)) لابن منظور (11/ 335). .
وهي مُعرَّبةٌ مِن الفارِسيَّةِ، مُركَّبةٌ مِن "سر" بمَعْنى فوق، و"بال" أي: القامةِ، وهي بمَعْنى القَميصِ والرِّداءِ [537] يُنظر: ((قاموس الفارِسيَّة)) لعبد النعيم محمد حسنين (ص: 361)، ((الألْفاظ الفارِسيَّة المعرَّبة)) لأدى شير (ص: 88)، ((تفسير الألْفاظ الدَّخيلة في اللُّغة العَربيَّة)) لطوبيا العنيسي (ص: 35). .
سَرِيًّا:
وَرَدَتْ هذه المُفرَدةُ في قَوْلِه تعالى: فَنَادَاهَا مِنْ تَحْتِهَا أَلَّا تَحْزَنِي قَدْ جَعَلَ رَبُّكِ تَحْتَكِ سَرِيًّا [مريم: 24] .
مَعْناها واللُّغةُ الَّتي نُقِلَتْ مِنها:
السَّريُّ: النَّهْرُ الصَّغيرُ، جَمْعُه: أسْرِيَةٌ وسُرْيانٌ [538] يُنظر: ((تفسير غريب القرآن)) لابن قتيبة (ص: 274)، ((لسان العَرب)) لابن منظور (14/ 380). .
اخْتُلِفَ في أصْلِها؛ فقيلَ: نَبَطيَّةٌ، والسَّرِيُّ بمَعْنى النَّهْرِ في النَّبَطيَّةِ [539] يُنظر: ((ظاهرة الغريب في اللُّغة العَربيَّة)) لحسن محمد تقي (2/ 141). ، وقيلَ: يونانيَّةٌ [540] يُنظر: ((البرهان)) للزركشي (1/ 288). . وقيلَ: سُرْيانيَّةٌ [541] يُنظر: ((المهذب فيما وقع في القرآن من المعرَّب)) للسيوطي (ص: 99)، ((مُعجَم غريب القرآن)) لمحمد فؤاد عبد الباقي (ص: 288). .
سَفَرةٌ:
قالَ اللهُ تعالى: بِأَيْدِي سَفَرَةٍ [عبس: 15] .
مَعْناها واللُّغةُ الَّتي نُقِلَتْ مِنها:
السَّفَرةُ: كَتَبةُ المَلائِكةِ الَّذين يُحْصونَ الأعْمالَ، والسَّفَرةُ: الكَتَبةُ، واحِدُهم سافِرٌ، وقيلَ للكاتِبِ سافِرٌ، وللكِتابِ سِفْرٌ؛ لأنَّ مَعْناه أنَّه يُبيِّنُ الشَّيءَ ويُوَضِّحُه [542] يُنظر: ((تفسير غريب القرآن)) لابن قتيبة (ص: 514)، ((لسان العَرب)) لابن منظور (4/ 370). .
وفي أصْلِها: قيلَ: نَبَطيَّةٌ، أصْلُها: سافرا، ومَعْناها الكَتَبةُ [543] يُنظر: ((لسان العَرب)) لابن منظور (4/ 370)، ((المهذب فيما وقع في القرآن من المعرَّب)) للسيوطي (ص: 100)، ((الأصل والبيان في معرب القرآن)) لحمزة فتح الله (ص: 13). . وقيلَ: آراميَّةٌ، أصْلُها: سافر، بمَعْنى كاتِبٍ [544] يُنظر: ((غرائب اللُّغة العَربيَّة)) لرفائيل نخلة اليسوعي (ص: 187)، نقلًا عن ((المعرب في القرآن الكريم)) لمحمد السيد (ص: 231). .
وقيلَ: بلْ سُرْيانيَّةٌ، وقيلَ: حَبَشيَّةٌ. ولِهذا يُرجِّحُ الدُّكْتورُ مُحمَّد السَّيِّد أنَّها لَفْظةٌ تَوارَدَ فيها كَثيرٌ مِن اللُّغاتِ السَّاميَّةِ، واشْتَرَكَتْ في لَفْظِها ومَعْناها [545] يُنظر: ((المعرَّب في القرآن الكريم)) لمحمد السيد (ص: 232). .
سَقَرُ:
قالَ اللهُ تعالى: يَوْمَ يُسْحَبُونَ فِي النَّارِ عَلَى وُجُوهِهِمْ ذُوقُوا مَسَّ سَقَرَ [القمر: 48].
مَعْناها واللُّغةُ الَّتي نُقِلَتْ مِنها:
سَقَرُ: اسمُ عَلَمٍ مِن أسْماءِ جَهَنَّمَ.
اخْتُلِفَ في أصْلِها؛ فقيلَ: عَربيَّةٌ، سُمِّيَتِ النَّارُ بذلك؛ لأنَّها تُذيبُ الأجْسامَ والأرْواحَ، مِن قَوْلِهم: سَقَرَتْه الشَّمْسُ، أي: أذابَتْه. والسَّاقورُ أيضًا: حَديدةٌ تُحْمى ويُكْوى بها الحِمارُ، وإنَّما امْتَنَعَ مِن الصَّرْفِ؛ لأنَّه عَلَمٌ مُؤَنَّثٌ؛ قالَ اللهُ تعالى: لَا تُبْقِي وَلَا تَذَرُ [546] يُنظر: ((المعرَّب)) للجواليقي (ص: 198)، ((لسان العَرب)) لابن منظور (4/ 372). .
وقيلَ: آراميَّةُ الأصْلِ، مَعْناها في تلك اللُّغةِ: الإحْراقُ بالنَّارِ [547] يُنظر: ((غرائب اللُّغة العَربيَّة)) لرفائيل نخلة اليسوعي (ص: 187)، نقلًا عن ((المعرب في القرآن الكريم)) لمحمد السيد (ص: 233)، ((المعرَّب والدَّخيل في اللُّغة العَربيَّة)) للسبحان (ص: 289). .
سَكَرًا:
قالَ اللهُ تعالى: وَمِنْ ثَمَرَاتِ النَّخِيلِ وَالْأَعْنَابِ تَتَّخِذُونَ مِنْهُ سَكَرًا وَرِزْقًا حَسَنًا إِنَّ فِي ذَلِكَ لَآيَةً لِقَوْمٍ يَعْقِلُونَ [النحل: 67] .
مَعْناها واللُّغةُ الَّتي نُقِلَتْ مِنها:
السَّكَرُ: الخَمْرُ [548] يُنظر: ((تفسير غريب القرآن)) لابن قتيبة (ص: 245)، ((لسان العَرب)) لابن منظور (4/ 374). .
وقدِ اخْتُلِفَ فيها؛ فقيلَ: حَبَشيَّةٌ، ومَعْناها بالحَبَشيَّةِ: الخَلُّ [549] ينظر: ((المهذب فيما وقع في القرآن من المعرَّب)) للسيوطي (ص: 101)، ((الأصل والبيان في معرب القرآن)) لحمزة فتح الله (ص: 14). .
وقيلَ: آراميَّةٌ، ومَعْناها: الخَمْرُ وكُلُّ مُسكِرٍ [550] يُنظر: ((غرائب اللُّغة العَربيَّة)) لرفائيل نخلة اليسوعي (ص: 188)، نقلًا عن ((المعرب في القرآن الكريم)) لمحمد السيد (ص: 234). .
والأرْجَحُ أنَّها مِن الألْفاظِ المُشتَرَكةِ بيْنَ اللُّغاتِ السَّاميَّةِ، تَوارَدَتِ اللُّغاتُ عليها بنفْسِ اللَّفْظِ والمَعْنى؛ فهي في العَربيَّةِ والحَبَشيَّةِ والآراميَّةِ وغيْرِها، لا يُدْرى أيُّهم أسْبَقُ [551] يُنظر: ((التَّطوُّر النحوي للغة العَربيَّة)) برجشتراسر (ص: 208، 209). .
سَلْسَبيلُ:
قالَ اللهُ تعالى: عَيْنًا فِيهَا تُسَمَّى سَلْسَبِيلًا [الإنسان: 18] .
مَعْناها واللُّغةُ الَّتي نُقِلَتْ مِنها:
السَّلْسَبيلُ: اسمٌ لعَيْنٍ في الجَنَّةِ؛ قيلَ: مَعْناها: السَّلاسةُ والجَرَيانُ [552] ينظر: ((تفسير الطبري = جامع البيان)) (23/ 561)، ((الهداية إلى بلوغ النهاية)) لمكي بن أبي طالب (12/ 7929). .
والكَلِمةُ مُعرَّبةٌ مِن الفارِسيَّةِ؛ ففي قاموسِ الفارِسيَّةِ: سَلْسَبيلُ: الماءُ العَذْبُ، والخَمْرُ الحَسَنةُ المَذاقِ، واسمُ عَيْنٍ في الجَنَّةِ [553] يُنظر: ((قاموس الفارِسيَّة)) لعبد النعيم محمد حسنين (ص: 308)، ((المعرب في القرآن الكريم)) لمحمد السيد (ص: 235). .
سُلَيْمانُ:
قالَ اللهُ تعالى: وَاتَّبَعُوا مَا تَتْلُو الشَّيَاطِينُ عَلَى مُلْكِ سُلَيْمَانَ وَمَا كَفَرَ سُلَيْمَانُ وَلَكِنَّ الشَّيَاطِينَ كَفَرُوا [البقرة: 102] .
مَعْناها واللُّغةُ الَّتي نُقِلَتْ مِنها:
هي اسمُ عَلَمٍ لنَبيِّ اللهِ سُلَيمانَ بنِ داودَ عليهما السَّلامُ.
والكَلِمةُ مَنْقولةٌ مِن العِبْريَّةِ: "شلومو"، ثُمَّ انْتَقَلَتْ إلى السُّرْيانيَّةِ فصارَتْ: "شليمون"، وفي اليونانيَّةِ: "سلومون"، ويَبْدو أنَّها دَخَلَتْ إلى العَربيَّةِ مِن العِبْريَّةِ عن طَريقِ السُّرْيانيَّةِ [554] ينظر: ((المعرب)) للجواليقي (ص: 191)، ((المعرب والدَّخيل)) للسبحان (ص: 275). .
سُنْدُسٌ:
قالَ اللهُ تعالى: وَيَلْبَسُونَ ثِيَابًا خُضْرًا مِنْ سُنْدُسٍ وَإِسْتَبْرَقٍ [الكهف: 31] .
مَعْناها واللُّغةُ الَّتي نُقِلَتْ مِنها:
السُّنْدُسُ: رَقيقُ الدِّيباجِ ورَفيعُه، وقيلَ: بلْ هو ألْطَفُ مِن الدِّيباجِ [555] ينظر: ((تفسير غريب القرآن)) لابن قتيبة (ص: 267)، ((الهداية إلى بلوغ النهاية)) لمكي بن أبي طالب (6/ 4377)، ((لسان العَرب)) لابن منظور (6/ 107). .
اخْتُلِفَ فيها؛ فقيلَ: هِنْديَّةٌ، ومَعْناها: الرَّقيقُ مِن السِّتْرِ [556] يُنظر: ((البرهان)) للزركشي (1/ 228)، ((المهذب فيما وقع في القرآن من المعرَّب)) للسيوطي (ص: 102). . وقيلَ: فارِسيَّةٌ [557] ينظر: ((فقه اللُّغة وسر العَربيَّة)) للثعالبي (ص: 208)، ((الأصل والبيان في معرب القرآن)) لحمز فتح الله (ص: 13). ، وقيلَ: يونانيَّةٌ، أصْلُها: (سندكس)، مِن مَعانيها: الدِّيباجُ الرَّقيقُ، النَّسيجُ مِن القُطْنِ، الكِتابُ الرَّقيقُ جدًّا، وتُطلَقُ على مَلابِسِ النِّساءِ المُفصَّلةِ مِن الكَتَّانِ الرَّقيقِ الشَّفافِ [558] يُنظر: ((غرائب اللُّغة العَربيَّة)) لرفائيل نخلة اليسوعي (ص: 260)، نقلًا عن ((المعرب في القرآن الكريم)) لمحمد السيد (ص: 237)، ((المعرَّب والدَّخيل في اللُّغة العَربيَّة للسبحان (ص: 256). .
سِينينُ:
قالَ اللهُ تعالى: وَطُورِ سِينِينَ [التين: 2] . كما وَرَدَتْ بلَفْظِ "سَيناءَ" في قَوْلِه تعالى: وَشَجَرَةً تَخْرُجُ مِنْ طُورِ سَيْنَاءَ تَنْبُتُ بِالدُّهْنِ وَصِبْغٍ لِلْآكِلِينَ [المؤمنون: 20] .
مَعْناها واللُّغةُ الَّتي نُقِلَتْ مِنها:
سينينُ، أو طورُ سِينينَ، أو سينا، أو سيناءُ: اسمُ المَكانِ المَشْهورِ [559] يُنظر: ((الكشاف)) للزمخشري (3/ 29)، ((لسان العَرب)) لابن منظور (13/ 230). .
اخْتُلِفَ في أصْلِها؛ فقيلَ: سينينُ: نَبَطيَّةٌ، ومَعْناها: الحَسَنُ [560] يُنظر: ((البرهان)) للزركشي (1/288)، ((المهذب فيما وقع في القرآن من المعرَّب)) للسيوطي (ص: 103). ، وقيلَ: حَبَشيَّةٌ، ومَعْناها الحَسَنُ كذلك [561] يُنظر: ((الأصل والبيان في معرب القرآن)) لحمزة فتح الله (ص: 14). .
وقيلَ هي بالعِبْريَّةِ: "سيناي"، ومِنها أُخِذَتْ: "سَيناءُ" بالمَدِّ، وكذلك "سِينين" بزِيادةِ النُّونِ وحَذْفِ الألِفِ وكَسْرِ النُّونِ قبْلَها، وباليونانيَّةِ: "سينا"، وقدِ انْتَقَلَتْ إلى العَربيَّةِ كما هي، فقيلَ: "سِينا" بالقَصْرِ [562] يُنظر: ((المعرَّب والدَّخيل في اللُّغة العَربيَّة)) للسبحان (ص: 286، 287). .

انظر أيضا:

  • تَمْهيدٌ: معجم الألْفاظُ المُعرَّبةُ في القُرآنِ الكَريمِ.
  • الفَصْلُ الأوَّلُ: حَرْفُ الهَمْزةِ.
  • الفَصْلُ الثَّاني: حَرْفُ الباءِ.
  • الفَصْلُ الثَّالِثُ: حَرْفُ التَّاءِ.
