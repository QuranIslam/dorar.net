الفَصْلُ الثَّامِنُ: حَرْفُ الدَّالِ

محتويات الصفحة
انظر أيضا

دادة عامِّيٌّ:
الدَّادةُ: المُرَبِّيةُ والحاضِنةُ، والمُسنَّةُ الَّتي تُرَبِّي الوَلَدَ وتُلَاعبُه وتَخدُمُه مُنْذُ الطُّفولةِ.
واللَّفْظُ مُعرَّبٌ مِن التُّرْكيَّةِ: (dada) [1775] يُنظر: ((مُعجَم الدَّخيل في اللُّغة العَربيَّة الحديثة)) للدكتور/ ف. عبد الرحيم (ص: 103)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 202). .
دارِش:
الدَّرشُ: السَّوادُ، والدَّارِشُ يُطلَقُ على الجِلْدِ الأسْوَدِ.
واللَّفْظُ مُعرَّبٌ مِن الفارِسيَّةِ [1776] يُنظر: ((غريب الحديث)) لإبراهيم الحربي (2/ 532)، ((جمهرة اللُّغة)) لابن دُرَيْدٍ (2/ 628)، ((لسان العَرب)) لابن منظور (6/ 301)، ((قصد السبيل)) لمحمد الأمين المحبي (2/ 8). .
داشِن:
الدَّاشِنُ: هو الثَّوْبُ الَّذي لم يُلبَسْ، أو الدَّارُ الجَديدةُ الَّتي لم تُسكَنْ.
واللَّفْظُ مُعرَّبٌ مِن الفارِسيَّةِ: (دُشْن)، ومَعْناه: الأجْرةُ والعَطاءُ، وما يَتَصدَّقُ به المَجوسُ في أعْيادِهم، ومِنه أُخِذَ (دَشَنَ)، بمَعْنى: أعْطى [1777] يُنظر: ((العين)) للخليل (6/ 243)، ((المعرَّب)) للجواليقي (ص: 301)، ((لسان العَرب)) لابن منظور (13/ 155)، ((القاموس المحيط)) للفيروزابادي (ص: 1196)، ((قصد السبيل فيما في اللُّغة العَربيَّة من الدَّخيل)) لمحمد الأمين المحبي (2/ 10)، ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 64، 65). .
داموق:
الدَّاموقُ: يَوْمُ الحَرِّ الشَّديدِ معَ سُكونِ الرِّيحِ، يقالُ: يَوْمٌ داموقٌ.
وهو مُعرَّبٌ مِن الفارِسيَّةِ، وذَكَرَ أبو حاتِمٍ أنَّ أصْلَه: (دَمَهْ كِر)؛ مُركَّبٌ مِن (دمه) أي: النَّفْسِ، و(كر)، بمَعْنى: يَأخُذُ، فيكونُ مَعْناها: يَأخُذُ بالنَّفْسِ [1778] يُنظر: ((الجمهرة)) لابن دُرَيْدٍ (2/ 1207). ، وذَكَرَ أَدِّي شِير أنَّ أصْلَه: (دمكاه)، بمَعْنى: الأَتُونِ [1779] يُنظر: ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 66). ، ورَجَّحَ الدُّكْتورُ ف. عبْدُ الرَّحيمِ أنَّ أصْلَه (دمكه)، مَحْذوفَ ألِفِ (دمكاه)، وعُرِّبَ بصورةِ: (دَمَقة) [1780] يُنظر: ((المعرَّب)) للجواليقي (ص: 308)، ((قصد السبيل فيما في اللُّغة العَربيَّة من الدَّخيل)) لمحمد الأمين المحبي (2/ 11). .
داهِل:
الدَّاهِلُ: المُتَحيِّرُ، وقَوْلُهم: لا دَهْلَ، أي: لا تَخَفْ.
وهو نَبَطيٌّ مُعرَّبٌ: (دالَه) [1781] يُنظر: ((التهذيب)) للأزهري (6/ 113)، ((لسان العَرب)) لابن منظور (11/ 251)، ((قصد السبيل فيما في اللُّغة العَربيَّة من الدَّخيل)) لمحمد الأمين المحبي (2/ 13). .
دِبْس:
الدِّبْسُ والدِّبِسُ: عُصارةُ التَّمْرِ، وما يَسيلُ مِن الرُّطَبِ، وقدْ يُطلَقُ على عَسَلِ النَّحْلِ، ويقالُ مِن المَأثورِ: "ائْتَدِموا بالدِّبْسِ".
وهو مُعرَّبٌ مِن السُّرْيانيَّةِ: (delicho) [1782] يُنظر: ((الصحاح)) للجوهري (3/ 926)، ((تاج العروس)) للزبيدي (16/ 48)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 205). .
دَبُّوس:
الدَّبُّوسُ: المَقمَعُ مِن حَديدٍ وغيْرِه، وهي -في الاسْتِعمالِ الحَديثِ- هِراوةٌ قَصيرةٌ برأسٍ رَفيعٍ مُسنَّنٍ، والجَمْعُ: دَبابيسُ.
واللَّفْظُ مُعرَّبٌ [1783] يُنظر: ((الصحاح تاج اللُّغة وصحاح العَربيَّة)) للجوهري (3/ 926)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 205). مِن الفارِسيَّةِ، وذَكَرَ الزَّبيديُّ أنَّ أصْلَه (دَبُّوز) [1784] يُنظر: ((تاج العروس)) للزبيدي (16/ 49). ، وذَكَرَ أَدِّي شِير أنَّ أصْلَه (دَبُّوس) [1785] يُنظر: ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 60). .
دِبْنٌ:
الدِّبْنُ: حَظيرةُ الغَنَمِ الَّتي تُصنَعُ مِن القَصَبِ.
وهو مُعرَّبٌ مِن الفارِسيَّةِ [1786] يُنظر: ((المحكم والمحيط الأعظم)) لابن سيده (9/ 353)، ((لسان العَرب)) لابن منظور (13/ 146)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 205) .
دِجْرٌ:
الدِّجْرُ: اللُّوبياءُ، وقدْ يُقالُ بفَتْحِ الدَّالِ وضَمِّها، وأعْلى اللُّغاتِ فيه الكَسْرُ.
لم تُشِرِ المَعاجِمُ إلى تَعْريبِه، وهو مُعرَّبٌ مِن الفارِسيَّةِ: (دَجِر) [1787] يُنظر: ((تهذيب اللُّغة)) للأزهري (10/ 334)، ((لسان العَرب)) لابن منظور (4/ 277)، ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 61). .
دَخْدار:
الدَّخْدارُ: نَوْعٌ مِن الثِّيابِ النَّفيسةِ [1788] يُنظر: ((لسان العَرب)) لابن منظور (4/ 279). ، والذَّهَبُ، واشْتَقُّوا مِن المَعْنى الثَّاني فِعْلًا، وهو: دَخْدَرَ القُرْطَ، أي: طَلاه بالذَّهَبِ [1789] يُنظر: ((القاموس المحيط)) للفيروزابادي (ص: 391). .
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (تخت دار)، مُكوَّنٌ مِن كَلِمتَينِ: (تخت) أي: سَريرٌ أو خِزانةٌ، و(دار) أي: ذو وصاحِبٌ، ويُقصَدُ به المَصونُ في الخَزائِنِ لنَفاستِه [1790] يُنظر: ((تهذيب اللُّغة)) للأزهري (7/ 276)، ((المعرَّب)) للجواليقي (ص: 292، 293). .
دَرابْزِين:
الدَّرابْزِينُ: قَوائِمُ مَصْفوفةٌ مِن خَشَبٍ أو حَديدٍ، به تُحاطُ السَّلالِمُ فتَمنَعُ مِن السُّقوطِ، وتُشكِّلُ سِياجًا حاجِزًا.
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (دَرابْزين)، وهي مُكوَّنةٌ مِن كَلِمتَينِ: (در)، أي: بابٌ، و(بزين) أي: تَخْتٌ [1791] يُنظر: ((القاموس المحيط)) للفيروزابادي (ص: 872)، ((قصد السبيل)) لمحمد الأمين المحبي (2/ 18)، ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 61)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 207). .
دِرَفْس:
الدِّرَفْسُ: الرَّايةُ والعَلَمُ الكَبيرُ.
واللَّفْظُ مُعرَّبٌ مِن الفارِسيَّةِ: (درفش)، ومَعْناه: الرَّايةُ، أمَّا الدِّرَفْسُ بمَعْنى البَعيرِ الضَّخْمِ فهو عَربيٌّ [1792] يُنظر: ((المعرَّب)) للجواليقي (ص: 309، 310)، ((القاموس المحيط)) للفيروزابادي (ص: 544)، ((قصد السبيل فيما في اللُّغة العَربيَّة من الدَّخيل)) لمحمد الأمين المحبي (2/ 18)، ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 62). .
دُرَّاقِن:
الدُّرَّاقِنُ والدُّرَاقِنُ: الخَوخُ.
وقدْ ذَكَرَ ابنُ دُرَيْدٍ أنَّ اللَّفْظَ مُعرَّبٌ مِن السُّرْيانيَّةِ أو الرُّوميَّةِ [1793] يُنظر: ((الجمهرة)) لابن دُرَيْدٍ (3/ 1326). ، وذَكَرَ الدُّكْتورُ/ ف. عبْدُ الرَّحيمِ أنَّه مُعرَّبٌ مِن السُّرْيانيَّةِ: (داورقينا)، وأصْلُ اللَّفْظِ السُّرْيانيِّ مَأخوذٌ مِن اليونانيَّةِ [1794] يُنظر: ((المعرَّب)) للجواليقي (ص: 296)، ((القاموس المحيط)) للفيروزابادي (ص: 1196)، ((تفسير الألْفاظ الدَّخيلة في اللُّغة العَربيَّة)) لطوبيا العنيسي (ص: 26). .
دَرْبان:
الدَّرْبانُ والدِّرِبانُ والدُّرْبانُ: البَوَّابُ والحاجِبُ، وجَمْعُها: درابِنةٌ؛ قالَ المُثقَّبُ العَبْديُّ:
كدُكَّانِ الدَّرابِنةِ المَطينِ
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (دَرْبان)، وهو لَفْظٌ مُركَّبٌ مِن كَلِمتَينِ؛ (در)، أي: البابُ، و(بان)، أي: حافِظٌ [1795] يُنظر: ((المعرَّب)) للجواليقي (ص: 291، 292)، ((لسان العَرب)) لابن منظور (13/ 154)، ((القاموس المحيط)) للفيروزابادي (ص: 1196). .
دِرْباس تِرْباس: عامِّيٌّ:
الدِّرْباسُ: هو حَديدةٌ تكونُ خَلْفَ البابِ، يُغلَقُ بها البابُ، ولا يُفتَحُ بمِفْتاحٍ، وعَربيَّتُه: المِزْلاجُ.
لم تُشِرِ المَعاجِمُ القَديمةُ لكَوْنِه مُعرَّبًا، وهو مُعرَّبٌ مِن الفارِسيَّةِ: (در) أي: بابٌ، و(بَسْتن)، أي: الإغْلاقُ [1796] يُنظر: ((تاج العروس)) للزبيدي (25/ 413)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 208). .
دَرْزٌ:
الدَّرْزُ: الشَّقُّ في القُماشِ، ومَوضِعُ الخِياطةِ في الثَّوْبِ، ويُقالُ مَجازًا للخَيَّاطينَ والحاكةِ: "بَنو درز"، ويُقالُ: "أوْلادُ دَرْزةَ"، ويُقصَدُ بهم الغَوْغاءُ والسَّفِلةُ مِن الخَلْقِ، وتُكَنَّى الدُّنْيا بـ "أم دَرْزٍ"، ويقالُ: "ابنُ دَرزةَ" لِمَن لا يُعرَفُ له نَسَبٌ، ويُجمَعُ على دُروزٍ.
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (دَرْز)، وهو ارْتِفاعُ الثَّوْبِ إذا جُمِعَ طَرَفاه [1797] يُنظر: ((العين)) للخليل (7/ 356)، ((جمهرة اللُّغة)) لابن دُرَيْدٍ (2/ 627)، ((قصد السبيل فيما في اللُّغة العَربيَّة من الدَّخيل)) لمحمد الأمين المحبي (2/ 20)، ((تاج العروس)) للزبيدي (15/ 145)، ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 62). ، وفي العامِّيَّةِ "ترزي"، وهو لَفْظٌ مُعرَّبٌ مِن التُّرْكيَّةِ: (terzi)، واللَّفْظُ التُّرْكيُّ تَحْريفٌ لـ "دَرْزي" مِن الفارِسيَّةِ لمَن يَخيطُ الثِّيابَ [1798] يُنظر: ((مُعجَم الدَّخيل في اللُّغة العَربيَّة الحديثة)) للدكتور/ ف. عبد الرحيم (ص: 79)، (المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 211). .
دَرَقة:
الدَّرَقةُ: نَوْعٌ مِن التُّروسِ الَّتي يَتَترَّسُ بها المُحارِبونَ وتُصنَعُ مِن جُلودِ الحَيَوانِ، وبابٌ صَغيرٌ (خوخة) يكونُ في النَّهْرِ، وتُجمَعُ على: دَرَقٍ، ودِراقٍ، وأدْراقٍ؛ قالَ رُؤْبةُ:
لو صَفَّ أدْراقًا مَضى مِن الدَّرَقْ
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (دَرِيچه) [1799] يُنظر: ((جمهرة اللُّغة)) لابن دُرَيْدٍ (2/ 635)، ((مختار الصحاح)) للرازي (ص: 104)، ((القاموس المحيط)) للفيروزابادي (ص: 882)، ((تاج العروس)) للزبيدي (25/ 280، 281). أو مِن اليونانيَّةِ: (thorake) [1800] يُنظر: ((جمهرة اللُّغة)) لابن دُرَيْدٍ (2/ 635)، ((مختار الصحاح)) للرازي (ص: 104)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 211). .
دَرْمَق:
الدَّرْمَقُ والدَّرْمَكُ: هو الدَّقيقُ النَّقِيُّ المَصْنوعُ مِن لُبابِ القَمْحِ، ويُطلَقُ على التُّرابِ الدَّقيقِ، جاءَ في الحَديثِ عن تُرْبةِ الجنَّةِ أنَّها: ((دَرْمَكةٌ بَيْضاءُ مِسْكٌ خالِصٌ )) [1801] أخرجه مسلم (2928) من حَديثِ أبي سعيدٍ الخُدْريِّ رَضِيَ اللهُ عنه. .
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (دَرْمقة) [1802] يُنظر: ((تهذيب اللُّغة)) للأزهري (9/ 307)، ((لسان العَرب)) لابن منظور (10/ 423)، ((قصد السبيل)) لمحمد الأمين المحبي (2/ 22)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 212). .
دُرْنُوك:
الدُّرْنُوكُ والدَّرْنيكُ والدِّرْنِكُ والدُّرْموكُ: نَوْعٌ مِن البُسُطِ -أو الثِّيابِ- له أطْرافٌ مُتَدَلِّيةٌ قَصيرةٌ، ويُجمَعُ على دَرانِكَ ودَرانيكَ؛ قالَ رُؤْبةُ:
جَعْدِ الدَّرانيكِ رِفَلِّ الأجْلادِ
كأنَّه مُخْتَضِبٌ في أجْسادِ
وقالَ الرَّاجِزُ:
أرْسَلْتُ فيها قَطِمًا لُكالِكا
كأنَّ فوقَ ظَهْرِهِ دَرانِكا
واللَّفْظُ مُعرَّبٌ مِن الحَبَشيَّةِ [1803] يُنظر: ((الصحاح)) للجوهري (4/ 1583، 1584)، ((المحكم والمحيط الأعظم)) لابن سيده (7/ 165)، ((المعرَّب)) للجواليقي (ص: 314)، ((المجموع المغيث في غريبي القرآن والحديث)) لمحمد بن عمر الأصبهاني (1/ 653)، ((لسان العَرب)) لابن منظور (10/ 423)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 212). .
دَرْويش:
الدَّرْويشُ: الشَّحَّاذُ، الفَقيرُ المِسْكينُ، ويُطلَقُ على الزَّاهِدِ في الدُّنْيا.
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (دَرْويش)، وأصْلُ مَعْناه: (دَر) أي: بابٌ، و(پيش) أي: أمامَ، فهو الواقِفُ بالبابِ، ويُكَنَّى بذلك عن الفَقيرِ [1804] يُنظر: ((تاج العروس)) للزبيدي (17/ 202)، ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 63)، ((مُعجَم متن اللُّغة)) لأحمد رضا (2/ 401)، ((مُعجَم الدَّخيل في اللُّغة العَربيَّة الحديثة)) للدكتور/ ف. عبد الرحيم (ص: 104)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 214). .
دَست:
الدَّسْتُ: الصَّحْراءُ، وهو الثِّيابُ الَّتي تَكْفي الإنْسانَ في قَضاءِ حَوائِجِه، والوَرَقُ، والرِّياسةُ، والحِيلةُ، والوِسادةُ، واسْتُعمِلَ أيضًا بمَعْنى الدِّيوانِ، وفي القِمارِ يَقولونَ: تَمَّ له الدَّسْتُ، أي: غَلَبَ في القِمارِ، وتَمَّ عليه الدَّسْتُ، أي: غُلِبَ وخَسِرَ.
واللَّفْظُ مُعرَّبٌ مِن الفارِسيَّةِ، وذَكَرَتِ المَعاجِمُ القَديمةُ أنَّ أصْلَه في كلِّ المَعاني: (دشت)، في حينِ فَصَلَ أَدِّي شِير -ووافَقَه الدُّكْتورُ/ ف. عبْدُ الرَّحيمِ- بيْنَ اللَّفْظِ بمَعْنى الصَّحْراءِ، فذَكَروا أنَّ أصْلَه: (دشت)، وأصْلُه بالمَعاني الأُخرى: (دست)، بمَعْنى: اليَدِ، وصَدْرِ البَيْتِ ومَجلِسِ المُلوكِ، وغيْرِ ذلك مِن المَعاني [1805] يُنظر: ((مجمل اللُّغة)) لابن فارس (ص: 326)، ((المعرَّب)) للجواليقي (ص: 288، 289)، ((شفاء الغليل)) للشهاب الخفاجي (ص: 122 – 124)، ((تاج العروس)) للزبيدي (4/ 518، 519)، ((مُعجَم متن اللُّغة)) لأحمد رضا (2/ 408)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 215). .
دَسْتَبان:
الدَّسْتَبانُ: رِباطٌ يكونُ على يَدِ مُربِّي جَوارِحِ الطُّيورِ كالصَّقْرِ أو الشَّاهينِ.
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (دست) أي: يَدٌ، و(بان) أي: حارِسٌ وحافِظٌ، فهو حامي اليَدِ [1806] يُنظر: ((المخصص)) لابن سيده (2/ 331)، ((مُعجَم متن اللُّغة)) لأحمد رضا (2/ 408)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 216). .
دُسْتورٌ:
الدُّسْتورُ: الدَّفْتَرُ الَّذي تُكتَبُ فيه أسْماءُ المَجْموعاتِ مُطلَقًا، كأسْماءِ الجُنْدِ، والدَّفْتَرُ الَّذي تُجمَعُ فيه قَوانينُ المَلِكِ، وأُطلِقَ على الوَزيرِ الأعْظَمِ، وفي العَصرِ الحَديثِ يُطلَقُ على أعلى وَثيقةٍ قانونيَّةٍ في الدَّوْلةِ، ويُجمَعُ على: دَساتيرَ.
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (دَستور)، بمَعْنى: يَدِ الشَّاةِ القَويَّةِ؛ مُركَّبةٌ مِن (دست) أي: القاعِدةِ، و(وَر) أي: صاحِبٍ، فهو صاحِبُ السُّلْطةِ [1807] يُنظر: ((تقويم اللسان)) لابن الجوزي (ص: 105)، ((تاج العروس)) للزبيدي (11/ 292، 293)، ((مُعجَم متن اللُّغة)) لأحمد رضا (2/ 408، 409)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 217). .
دَسْتِيج:
الدَّسْتِيجُ: إناءٌ يُحوَّلُ ويُنقَلُ عن طَريقِ اليَدِ.
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (دَسْتي) أي: يَدٌ، والياءُ للنِّسْبةِ في الفارِسيَّةِ، فهو الإناءُ اليَدَويُّ [1808] يُنظر: ((التكملة والذيل والصلة)) للصغاني (1/ 431)، ((تاج العروس)) للزبيدي (5/ 566)، ((مُعجَم متن اللُّغة)) لأحمد رضا (2/ 408)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 217). .
دَفْتَر:
الدَّفْتَرُ والدِّفْتَرُ: الصُّحُفُ المَضْمومُ بعضُها إلى بعضٍ، وفيه لُغةٌ لبَني أسَدٍ أو قَيْسٍ: التَّفْتَرُ.
ذَكَرَتِ المَعاجِمُ أنَّه عَربيٌّ صَحيحٌ وليس مُعرَّبًا، لكنَّه لم يُعرَفْ له اشْتِقاقٌ [1809] يُنظر: ((لسان العَرب)) لابن منظور (4/ 289)، ((تاج العروس)) للزبيدي (11/ 304، 305). ، وذَكَرَ ابنُ سِيدَهْ أنَّ (التَّفْتَرَ) -وهو لُغةٌ في الدَّفْتَرِ- أعْجَميٌّ [1810] يُنظر: ((المحكم)) لابن سيده (9/ 552). ، وهو بالفارِسيَّةِ: (دَفْتَر)، ورَجَّحَ أَدِّي شِير أنَّ أصْلَه مِن اليونانيَّةِ [1811] يُنظر: ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 65). ، وذَكَرَ ذلك الدُّكْتورُ/ ف. عبْدُ الرَّحيمِ [1812] يُنظر: ((المعرَّب)) للجواليقي (ص: 304). ، أمَّا طوبيا العُنَيْسيُّ فذَكَرَ أنَّ الأصْلَ اليونانيَّ يَعودُ لأصْلٍ فينيقيٍّ، وهو (ديفتارا)، أي: كِتابٌ صَغيرٌ [1813] يُنظر: ((تفسير الألْفاظ الدَّخيلة)) لطوبيا العنيسي (ص: 28). .
دَكَّةٌ:
الدَّكَّةُ: المِسْطَبةُ، وهي بِناءٌ يُسوَّى سَطْحُه ليَكونَ مُهيَّأً للقُعودِ، والعامَّةُ تَكسِرُ الدَّالَ يقولونَ: الدِّكَّةُ، وجَمْعُها: دِكَكٌ.
ذَكَرَ الفَيُّوميُّ أنَّه مُعرَّبٌ [1814] يُنظر: ((المصباح المنير)) للفيومي (1/ 198). ، وكذا ذَكَرَ مُحمَّد الأمين المُحبِّي [1815] يُنظر: ((قصد السبيل فيما في اللُّغة العَربيَّة من الدَّخيل)) لمحمد الأمين المحبي (2/ 31). .
دُكَّانٌ:
الدُّكَّانُ: الحانوتُ، يُذكَّرُ ويُؤنَّثُ، ويُجمَعُ على: دَكاكينَ، وهو الدَّكَّةُ أيضًا؛ ففي حَديثِ أبي هُرَيْرةَ: ((فبَنَيْنا له دُكَّانًا مِن طينٍ يَجلِسُ عليه)) [1816] أخرجه أبو داود (4698)، والنَّسائيُّ (4991) باختلافٍ يسيرٍ من حَديثِ أبي هُرَيرةَ وأبي ذَرٍّ رَضِيَ اللهُ عنهما. ؛ وقالَ المُثقَّبُ العَبْديُّ:
فأبْقى باطلي والجِدُّ مِنها
كدُكَّانِ الدَّرابِنةِ المَطينِ [1817] يُنظر: ((تاج العروس)) للزبيدي (27/ 152)، ((مُعجَم متن اللُّغة)) لأحمد رضا (2/ 437).
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (دُكَّان) [1818] يُنظر: ((الصحاح)) للجوهري (5/ 2114)، ((تفسير الألْفاظ الدَّخيلة)) لطوبيا العنيسي (ص: 28). .
دُلْبٌ:
الدُّلْبُ: نَوْعٌ مِن شَجَرِ الزِّينةِ، الواحِدةُ دُلْبةٌ، وأرْضٌ مَدْلَبةٌ: ذاتُ دُلْبٍ كَثيرٍ.
وهو مُعرَّبٌ مِن السُّرْيانيَّةِ: (doulbo) [1819] يُنظر: ((الصحاح تاج اللُّغة وصحاح العَربيَّة)) للجوهري (1/ 125)، ((لسان العَرب)) لابن منظور (1/ 377)، ((قصد السبيل)) لمحمد الأمين المحبي (2/ 32)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 220). .
دُلْفِينٌ:
الدُّلْفِينُ: مِن الدَّوابِّ الَّتي تَعيشُ في البَحْرِ، وتُنْجي الغَريقَ وتَحمِلُه على ظَهْرِها، وعَربيَّتُه: دُخَسٌ [1820] يُنظر: ((تاج العروس)) للزبيدي (15/ 476). .
واللَّفْظُ مُعرَّبٌ مِن اليونانيَّةِ (delfin) [1821] يُنظر: ((مُعجَم متن اللُّغة)) لأحمد رضا (2/ 441)، ((تفسير الألْفاظ الدَّخيلة)) لطوبيا العنيسي (ص: 28). .
دَلَق:
الدَّلَقُ: حَيَوانٌ يُشبِهُ الهِرَّةَ، ظَهْرُه طَويلٌ، يُصنَعُ مِن جِلْدِه الفِراءُ.

الدُّمُسْتُقُ: أميرُ الأمَراءِ.
واللَّفْظُ مُعرَّبٌ مِن الرُّوميَّةِ [1823] يُنظر: ((مفاتيح العلوم)) للخوارزمي (ص: 148)، ((قصد السبيل فيما في اللُّغة العَربيَّة من الدَّخيل)) لمحمد الأمين المحبي (2/ 33). .
دَمْغة:
الدَّمْغةُ: هي العَلامةُ والخَتْمُ، وهي خاتَمُ الأميرِ أو السُّلْطانِ، أو شَهادةُ بَراءةٍ تَصدُرُ عن السُّلْطانِ أو الحاكِمِ بالعَفْوِ عن مُجرِمٍ أو تَأمينِ خائِفٍ، أو رَسْمٌ تَتَقاضاه الدَّوْلةُ أَو أحَدُ الأشْخاصِ العَامَّةِ على المُحرَّراتِ.
ولم تُشِرِ المَعاجِمُ القَديمةُ لكَوْنِه مُعرَّبًا، وهو مُعرَّبٌ مِن التُّرْكيَّةِ: (damga)، وفي التُّرْكيَّةِ القَديمةِ: (تمغا)، بمَعْنى: الطَّابَعِ والخَتْمِ [1824] يُنظر: ((مُعجَم الألْفاظ التَّاريخية في العصر المملوكي)) لمحمد أحمد دهمان (ص: 109)، ((مُعجَم الدَّخيل في اللُّغة العَربيَّة الحديثة)) للدكتور/ ف. عبد الرحيم (ص: 107)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 221)، ((المُعجَم الوسيط)) (1/297). .
دَمَق:
الدَّمَقُ: الرِّيحُ المُصاحِبُ لها ثَلْجٌ.
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (دَمَه) [1825] يُنظر: ((القاموس المحيط)) للفقيروزابادي (ص: 884)، ((تاج العروس)) للزبيدي (25/ 307). .
دِمَقْس:
الدِّمَقْسُ: الحَريرُ الأبْيَضُ، وفيه لُغاتٌ: المِدَقْسُ، والدِّقَمْسُ، والدِّمَقْصُ، والدِّمْقاسُ؛ قالَ امْرُؤُ القَيْسِ:
فظَلَّ العَذارى يَرْتَمينَ بلَحْمِها
وشَحْمٍ كهُدَّابِ الدِّمَقْسِ المُفَتَّلِ [1826] يُنظر: ((التهذيب)) للأزهري (9/ 292)، ((القاموس المحيط)) للفيروزابادي (ص: 546)، ((تاج العروس)) للزبيدي (16/ 91، 92).
واللَّفْظُ مُعرَّبٌ مِن الفارِسيَّةِ: (دِمْسَه)، بمَعْنى: الحَريرِ الأبْيَضِ، ذَكَرَ أَدِّي شِير أنَّه قد يكونُ مِن اليونانيَّةِ مَنْسوبًا إلى دِمَشْقَ [1827] يُنظر: ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 66). ، وجَزَمَ بذلك طوبيا العُنَيْسيُّ: (damaskos) [1828] يُنظر: ((تفسير الألْفاظ الدَّخيلة)) لطوبيا العنيسي (ص: 28، 29). ، ورأى الدُّكْتورُ ف/ عبْدُ الرَّحيمِ أنَّه مِن اليونانيَّةِ، لكنَّه يَخْتلِفُ معَ طوبيا العُنَيْسيِّ في أصْلِه فيَرى أنَّ أصْلَه: (مِتَكْسا)، ومَعْناه: الحَريرُ، ثُمَّ قُلِبَتْ في التَّعْريبِ، واللُّغاتُ الأخْرى قيلَت على القَلْبِ المَكانيِّ [1829] يُنظر: ((المعرَّب)) للجواليقي (ص: 311، 312). .
دَنْحٌ:
الدَّنْحُ: عيدٌ للنَّصارى، وأقْباطُ مِصْرَ يُسمُّونَه الغُطَاسُ أو العِمادَ.
وهو مُعرَّبٌ مِن السُّرْيانيَّةِ: (denho)، بمَعْنى: الظُّهورِ، مُشتَقٌّ مِن الفِعلِ (dnah)، ومَعْناه ظَهَرَ ولاحَ [1830] يُنظر: ((الجمهرة)) لابن دُرَيْدٍ (1/ 506)، ((تهذيب اللُّغة)) للأزهري (4/ 246)، ((المعرَّب)) للجواليقي (ص: 299)، ((قصد السبيل فيما في اللُّغة العَربيَّة من الدَّخيل)) لمحمد الأمين المحبي (2/ 34). .
دُوَّاجٌ:
الدُّوَّاجُ: اللِّحافُ المَلْبوسُ.
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (دَواج) [1831] يُنظر: ((جمهرة اللُّغة)) لابن دُرَيْدٍ (2/ 1038)، ((المعرَّب)) للجواليقي (ص: 305، 306)، ((القاموس المحيط)) للفيروزابادي (ص: 190). .
دُوَّامة:
الدُّوَّامةُ: لُعبةٌ يَلعَبُ بها الأطْفالُ، تُلَفُّ بسيرٍ أو خَيْطٍ ثُمَّ تُرْمى على الأرْضِ فتَدورُ.
وهو فارِسيٌّ مُعرَّبٌ: (دوابه) [1832] يُنظر: ((التهذيب)) للأزهري (12/ 216)، ((المحكم والمحيط الأعظم)) لابن سيده (9/ 446)، ((قصد السبيل فيما في اللُّغة العَربيَّة من الدَّخيل)) لمحمد الأمين المحبي (2/ 38). .
دَوْرَقٌ:
الدَّوْرَقُ: إبْريقٌ كَبيرٌ له عُرْوتانِ، وأيضًا مِكْيالٌ لِما يُشرَبُ.
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (دُوْرَه)، وهو بالضَّمِّ مَعْناه: جَرَّةٌ صَغيرةٌ، و(دَوْرَه) بفَتْحِ الدَّالِ مَعْناه: مِكْيالٌ للشَّرابِ، واللَّفْظُ بالفَهْلَويَّةِ: (دَوْرك)، وهذا هو أصْلُ اللَّفْظِ المُعرَّبِ [1833] يُنظر: ((جمهرة اللُّغة)) لابن دُرَيْدٍ (2/ 635)، ((التهذيب)) للأزهري (9/ 45)، ((الصحاح)) للجوهري (4/ 1474)، ((المعرَّب)) للجواليقي (ص: 301، 302)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 226). .
دوشابٌ:
الدُّوشابُ: نَبيذُ التَّمْرِ الأسْوَدِ.
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (دُوشَاب) أي: عَصيرُ العِنَبِ أو التَّمْرِ [1834] يُنظر: ((شفاء الغليل)) للشهاب الخفاجي (ص: 125)، ((مُعجَم متن اللُّغة)) لأحمد رضا (2/ 412)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 226). .
دُوغٌ:
الدُّوغُ: اللَّبَنُ المَنْزوعُ مِنه زُبدتُه، وعَربيَّتُه: المَخيضُ.
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (دوغ) [1835] يُنظر: ((المعرَّب)) للجواليقي (ص: 319)، ((النظم المستعذب في تفسير غريب ألْفاظ المهذب)) لابن بطال (2/ 204)، ((المصباح المنير)) للفيومي (1/ 203)، ((القاموس المحيط)) للفيروزابادي (ص: 781)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 227). .
دُولابٌ:
الدُّولابُ: السَّاقِيةُ الَّتي تُديرُها الدَّابَّةُ كي يُسْتَقى الماءُ.
واللَّفْظُ مُعرَّبٌ مِن الفارِسيَّةِ: (دولا) أي: إناءٌ، و(آب) أي: ماءٌ [1836] يُنظر: ((مُعجَم ديوان الأدب)) للفارابي (2/ 59)، ((الصحاح تاج اللُّغة وصحاح العَربيَّة)) للجوهري (1/ 125)، ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 65). .
دُهانِج:
الدُّهانِجُ: البَعيرُ يكونُ له سَنامانِ، أو هو البَعيرُ السَّريعُ، وهو العَظيمُ مِن كلِّ شيءٍ [1837] يُنظر: ((الجمهرة)) لابن دُرَيْدٍ (2/ 1136). ، وفيه لُغاتٌ: الدُّهامِجُ، والدَّهْمَجُ، والدَّهْنَجُ، والدَّمْهَجُ، والدُّناهِجُ.
نَصَّ الجَوْهَريُّ على تَعْريبِه [1838] يُنظر: ((الصحاح)) للجوهري (1/ 316). ، وكذا جاءَ في لِسانِ العَربِ [1839] يُنظر: ((لسان العَرب)) لابن منظور (2/ 277). ، وقدْ رأى الدُّكْتورُ ف. عبْدُ الرَّحيمِ أنَّ اللَّفْظَ عَربيٌّ صَحيحٌ، وأنَّه وَقَعَ خَلْطٌ بيْنَ هذا اللَّفْظِ ولَفْظِ (الدَّهْنَجِ)، وهو نَوْعٌ مِن الجَواهِرِ، وهو مُعرَّبٌ مِن الفارِسيَّةِ، وأصْلُه: (دَهْنَهْ) [1840] يُنظر: ((التهذيب)) للأزهري (6/ 271)، ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 68). ، وفيه لُغةٌ أُخرى: (دَهانَهْ)، وذَكَرَ أنَّه عُرِّبَ مِن هذه اللُّغةِ (الدُّهانِج)، بدَليلِ أنَّ ثَمَّةَ لُغةً فيه، وهي (دَهَنج) [1841] يُنظر: ((القاموس المحيط)) للفيروزابادي (ص: 190). ، فكأنَّ الألِفَ سَقَطَتْ مِنه [1842] يُنظر: ((المعرَّب)) للجواليقي (ص: 318، 319)، ((التكملة والذيل والصلة)) للصغاني (1/ 434). .
دَهْدَارٌ:
الدَّهْدَارُ: الكِبْرُ والباطِلُ، وهو لُغةٌ في التَّهْتارِ والتَّهَتُّرِ.
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (داهدار) مُركَّبٌ مِن (داه) أي: دَناءةٍ، و(دار) أي: صاحِبٍ [1843] يُنظر: ((جمهرة اللُّغة)) لابن دُرَيْدٍ (2/ 683)، ((تهذيب اللُّغة)) للأزهري (6/ 129)، ((الألْفاظ الفارِسيَّة المعربة)) لأدى شير (ص: 68). .
دِهْليزٌ:
الدِّهْليزُ والدَّهْليزُ: هو المَمَرُّ الَّذي يكونُ بيْنَ الدَّارِ وبابِها، ويُجمَعُ على: دَهاليزَ، ويُقالُ: القَبْرُ دِهْليزُ الآخِرةِ.
وهو مُعرَّبٌ مِن الفارِسيَّةِ، وأصْلُه ذَكَرَه الجَوْهَريُّ: (داليز) [1844] يُنظر: ((الصحاح تاج اللُّغة وصحاح العَربيَّة)) للجوهري (3/ 878)، ((قصد السبيل فيما في اللُّغة العَربيَّة من الدَّخيل)) لمحمد الأمين المحبي (2/ 42). ، وذَكَرَ أَدِّي شِير أنَّ أصْلَه: (دهله)، بمَعْنى: القَنْطرةِ [1845] يُنظر: ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 68). .
دَيَابُوذٌ:
الدَّيَابُوذُ والدَّيابودُ: ثَوْبٌ يُنسَجُ على خَيْطَينِ، قالَ الأعْشى:
عليه دَيَابُوذٌ تَسَرْبَلَ تحتَهُ
أرَنْدَجَ إسْكافٍ يُخَالِطُ عِظْلِمَا
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (دوپوذ)، وهو مُركَّبٌ مِن (دو) أي: اثْنَينِ، و(پوذ) أي: لُحْمةٍ [1846] يُنظر: ((جمهرة اللُّغة)) لابن دُرَيْدٍ (3/ 1322)، ((الصحاح)) للجوهري (2/ 564)، ((قصد السبيل فيما في اللُّغة العَربيَّة من الدَّخيل)) لمحمد الأمين المحبي (2/ 43). .
دَيْدَبان:
الدَّيْدَبانُ: الطَّليعةُ، وعَيْنُ القَوْمِ الَّذي يَأتي لهم بالأخْبارِ، وفي العَربيَّةِ: الرَّبِيئةُ.
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (دِيدَه) أي: عَيْنٌ، و(بان) لاحِقةٌ بمَعْنى الحامي والحارِسِ [1847] يُنظر: ((جمهرة اللُّغة)) لابن دُرَيْدٍ (3/ 1235)، ((التهذيب)) للأزهري (14/ 53، 64)، ((المحيط في اللُّغة)) للصاحب بن عباد (2/ 338)، ((المعرَّب)) للجواليقي (ص: 293)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 229). .
دَيْسَقٌ:
الدَّيْسَقُ: الخِوانُ المُتَّخَذُ مِن الفِضَّةِ، والخُبْزُ الأبْيَضُ، والصَّحْراءُ الواسِعةُ، والتُّرابُ، وكلُّ شيءٍ يُضيءُ ويُنيرُ يقالُ له: دَيْسَقٌ، ومِنه سَمَّوا السَّرابَ إذا لَمَعَ: دَيْسَقًا، والحَوْضُ المَلآنُ؛ يُقالُ: مَلأتُ الحَوْضَ حتَّى دَسَقَ، أي: فاضَ ماؤُه.
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (طَشْتُخْوان) [1848] يُنظر: ((الصحاح تاج اللُّغة وصحاح العَربيَّة)) للجوهري (4/ 1473، 1474)، ((لسان العَرب)) لابن منظور (10/ 97). .
دينارٌ:
الدِّينارُ والدَّيْنارُ: عُمْلةٌ رومانيَّةٌ قَديمةٌ، وهو يُساوي عنْدَ العَربِ عَشَرةَ دَراهِمَ، يُقالُ: رَجُلٌ مُدَنَّرٌ، أي: كَثيرُ الدَّنانيرِ، ويُجمَعُ على: دَنانيرَ.
ذَكَرَتِ المَعاجِمُ القَديمةُ أنَّ الدِّينارَ مُعرَّبٌ مِن الفارِسيَّةِ [1849] يُنظر: ((الجمهرة)) لابن دُرَيْدٍ (2/ 640)، ((لسان العَرب)) لابن منظور (4/ 292). ، والصَّحيحُ أنَّ أصْلَ اللَّفْظِ مِن اللَّاتينيَّةِ: (denarium)، ومَعْناه: عَشْريٌّ؛ لأنَّ الدِّينارَ يَشْتمِلُ على عَشْرِ وَحَداتٍ [1850] يُنظر: ((تفسير الألْفاظ الدَّخيلة)) لطوبيا العنيسي (ص: 30). ، ودَخَلَ اللَّفْظُ إلى الفَهْلَويَّةِ فكانَ: (denar)، ورَجَّحَ الدُّكْتورُ/ ف. عبْدُ الرَّحيمِ أن يكونَ أصْلُ اللَّفْظِ المُعرَّبِ مِن الفَهْلَويَّةِ [1851] يُنظر: ((المعرَّب)) للجواليقي (ص: 290). .
