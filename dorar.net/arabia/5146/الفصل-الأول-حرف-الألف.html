آبُ:
آبُ: الشَّهْرُ الثَّامِنُ مِن الشُّهورِ الشَّمْسيَّةِ (أَغُسْطُس).
وهو مِن السُّرْيانيَّةِ: ، بمَعْنى: "الغلَّةِ" [1208] يُنظر: ((المحكم)) لابن سيده (10/ 568)، ((شفاء الغليل)) للشهاب الخفاجي (ص: 48)، ((مُعجَم الدَّخيل في اللُّغة العَربيَّة الحديثة)) للدكتور: ف. عبد الرحيم (ص: 19)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 13). .
أَبْرَشِيَّة:
الأبْرَشِيَّةُ: المِنْطقةُ الَّتي يُديرُها الأُسْقُفُ، وقِسْمٌ مِن أقْسامِ المَمْلَكةِ الأرْبَعةِ عنْدَ الرُّومانِ.
واللَّفْظُ مُعرَّبٌ مِن اليونانيَّةِ: (eparchia) [1209] يُنظر: ((المُعجَم الوسيط)) الصادر عن مجمع اللُّغة العَربيَّة بالقاهرة (1/ 2)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 19). .
الإبْريزُ، والإبْريزيُّ:
الإبْرِيزُ والإبْريزيُّ: الذَّهَبُ الخالِصُ، والعَسْجَدُ؛ جاءَ في الحَديثِ: ((إنَّ اللهَ ليُجرِّبُ أحَدَكم بالبَلاءِ، كما يُجرِّبُ أحَدُكم ذَهَبَه بالنَّارِ؛ فمِنه ما يَخرُجُ كالذَّهَبِ الإبْريزِ، فذلك الَّذي نجَّاه اللهُ مِن السَّيِّئاتِ)) [1210] أخرجه الطبراني (8/195) (7698)، والحاكم (7878)، والبيهقي في ((شعب الإيمان)) (9924) باختلافٍ يسيرٍ من حَديثِ أبي أُمامةَ الباهِليِّ رَضِيَ اللهُ عنه. ، وقالَ النَّابِغةُ:
مُزيَّنةٌ بالإبْرِزِيِّ وحَشْوُها
رَضيعُ النَّدى والمُرشِفاتِ الحَواضِنِ
أشارَ ابنُ دُرَيْدٍ [1211] يُنظر: ((الجمهرة)) لابن دُرَيْدٍ (2/ 1193). والفَيُّوميُّ [1212] يُنظر: ((المصباح المنير)) للفيومي (1/ 44). إلى عُجْمتِه، وهو مُعرَّبٌ مِن اليونانيَّةِ: (أبريزون)، ومَعْناه الخالِصُ، وذَكَرَ أَدِّي شِير أنَّه مِن المُمكِنِ أن يكونَ مُعرَّبًا عن الفارِسيَّةِ، وأصْلُه: (آبريز)، مُكوَّنٌ مِن (آب) بمَعْنى الرَّوْنَقِ، و(ريز) بمَعْنى ساكِبٍ، فيكونُ مَعْناه: ساكِبُ الرَّوْنَقِ [1213] يُنظر: ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 6). ، والأرْجَحُ أنَّه مِن اليونانيَّةِ [1214] يُنظر: ((المعرَّب)) للجواليقي (ص: 121، 122)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 19). .
أبْزارٌ:
الأبْزَارُ: التَّوابِلُ، ولَفْظُه مُفرَدٌ ليس جَمْعًا.
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (بوي أبزار)، بمَعْنى أداةِ الرَّائِحةِ الطَّيِّبةِ، و(ديك أبزار)، بمَعْنى: أداةِ القِدْرِ [1215] يُنظر: ((الجمهرة)) لابن دُرَيْدٍ (1/ 256)، ((المعرَّب)) للجواليقي (ص: 114). .
إبْزيمٌ:
الإبْزِيمُ والإبْزِينُ: حَلْقةٌ لها لِسانٌ تكونُ في السَّرْجِ، والجَمْعُ: أبازيمُ.
وهو مِن الفارِسيَّةِ، ولم يَذكُرْ أحَدٌ مِن أصْحابِ المَعاجِمِ عُجْمتَه إلَّا ابنُ دُرَيْدٍ [1216] يُنظر: ((الجمهرة)) لابن دُرَيْدٍ (2/ 1193)، ، وذَكَرَ ابنُ مَنْظورٍ أنَّه مِن الفِعلِ (بَزَمَ) بمَعْنى: عَضَّ بمُقدَّمِ أسْنانِه [1217] يُنظر: ((لسان العَرب)) لابن منظور (12/ 49). ، وعلى قَوْلِه فليس مُعرَّبًا؛ قالَ الرَّاجِزُ:
لولا الأبازيمُ وأنَّ المِنْسَجا
ناهى عن الذِّئبةِ أن تَفَرَّجا [1218] يُنظر: ((المعرَّب)) للجواليقي (ص: 124).
أَبْسِتاق:
الأبِسْتاقُ: الكِتابُ الخاصُّ بدِيانةِ المَجوسِ.
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (أوستا) [1219] يُنظر: ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 20). .
أُبُلَّةٌ:
الأُبُلَّةُ: القِطْعةُ مِن التَّمْرِ، وهو تَمْرٌ يُرَضُّ بيْنَ حَجَرَينِ، ويُحلَبُ عليه اللَّبَنُ، وهو اسمُ مَكانٍ بالبَصْرةِ؛ قالَ الشَّاعِرُ:
فيَأكُلُ ما رُضَّ مِن زادِنا
ويَأبى الأُبُلَّةَ لم تُرْضَضِ
واللَّفْظُ مِن الفارِسيَّةِ، وأخَذَه الفُرْسُ مِن النَّبَطيَّةِ، وذَكَرَ الجَواليقيُّ فيه قِصَّةً؛ أنَّ امْرأةً خمَّارةً كانَ يُقالُ لها: (هوبا) كانَ العُمَّالُ يَعْمَلونَ في الأرْضِ ثُمَّ يَأتونَها، فماتَتْ، فأتَوها فلم يَجِدوها، فقالوا: هوب ليكا، أي: لم تَعُدْ مَوْجودةً، فغَلَطَتِ الفُرْسُ وقالوا: هوب لتْ، فلمَّا عَرَّبَتْها العَربُ، قالوا: الأُبُلَّةُ [1220] يُنظر: ((المعرَّب)) للجواليقي (ص: 109، 110)، ((لسان العَرب)) لابن منظور (11/ 8). .
أَبِيب:
أَبِيب: اسمُ شَهْرٍ مِن شُهورِ الحَرِّ، ويَبدَأُ مِن 8 يوليو إلى 6 أغسطس؛ قالَ النواجيُّ:
فُؤادي مِن ذُنوبي في لَهيبٍ
كوَقْدةِ حَرِّ مِسرى معْ أَبِيبِ
وهو مِن القِبْطيَّةِ [1221] يُنظر: ((شفاء الغليل)) للشهاب الخفاجي (ص: 57). .
أَبيلٌ:
الأبيلُ: الرَّاهِبُ، ورَئيسُ النَّصارى، والَّذي يُمسِكُ عَصا النَّاقوسِ ويَضرِبُه ليَدْعوَ النَّصارى إلى الصَّلاةِ، ويُجمَعُ على: "أبيلون"، و"آبال"، و"أُبُل"، وفيه لُغاتٌ: أيْبَلٌ، وأيْبُلٌ، وأيَبَليٌّ، وأيْبُلِيٌّ، وهَيْبَليٌّ، وأبُلِيٌّ، وأبِيلِيٌّ، واشْتَقُّوا مِنه فِعْلًا، وهو: أبَلَ يأبُلُ أبالةً: إذا تَرهَّبَ وتَنَسَّكَ؛ قالَ عَدِيُّ بنُ زيدٍ:
إنَّني واللهِ فاسْمَعْ حَلِفي
بأَبيلٍ كُلَّما صلَّى جَأَرْ
وقالَ الأعْشى:
فإنِّي ورَبِّ السَّاجِدينَ عَشيَّةً
وما صَكَّ ناقوسَ النَّصارى أبيلُها
وقالَ عَمْرُو بنُ عبْدِ الحَقِّ:
وما قَدَّسَ الرُّهْبانُ في كلِّ هَيكَلٍ
أبيلَ الأبيلينَ المَسيحَ بنَ مَرْيَما
وأبيلُ الأبيلينَ: عيسى بنُ مَرْيَمَ عليه السَّلامُ.
وهو مِن السُّرْيانيَّةِ: (abilo)، ومَعْناه الحَزينُ الباكي؛ لأنَّ الرَّاهِبَ يَبْكي على ذُنوبِه، وليس مِن الفارِسيَّةِ كما ذَكَرَ الجَواليقيُّ [1222] يُنظر: ((المعرَّب)) للجواليقي (ص: 137، 138)، ((تهذيب اللُّغة)) للأزهري (15/ 279)، ((لسان العَرب)) لابن منظور (11/ 7)، ((القاموس المحيط)) للفيروزابادي (ص: 959)، ((الألْفاظ السريانية في المعاجم العَربيَّة)) لأفرام الأوَّل برصوم (ص: 313). .
آبُنوس:
الآبُنوسُ والآبِنوسُ: نَوْعٌ مِن الشَّجَرِ تُتَّخَذُ مِنه السِّهامُ والجِفانُ [1223] يُنظر: ((لسان العَرب)) لابن منظور (12/ 286)، و(5/ 363)، ((القاموس المحيط)) للفيروزابادي (ص: 514، 1120)، ((تاج العروس)) للزبيدي (32/ 363)، و(15/ 182). ، وهو نَوْعٌ مِن الخَشَبِ مَعْروفٌ بالصَّلابةِ والقُوَّةِ، يُجلَبُ مِن الهِنْدِ؛ قالَ عبْدُ اللهِ بنُ العبَّاسِ بنِ الفَضْلِ:
يَتَثنَّى بحُسْنِ جيدِ غَزالٍ
ذي صَليبٍ مُفَضَّضٍ آبِنوسِ [1224] يُنظر: ((مُعجَم البلدان)) لياقوت الحموي (2/ 532).
واللَّفْظُ مُعرَّبٌ مِن اليونانيَّةِ [1225] يُنظر: ((قصد السبيل)) لمحمد الأمين المحبي (1/ 137). .
أتابِك:
الأتابِكُ: لَقَبٌ يُطلَقُ على مُرَبِّي الأُمَراءِ، ثُمَّ أُطلِقَ على مُدبِّرِ الدَّوْلةِ.
مِن الهِنْديَّةِ أو التَّتَريَّةِ، والكَلِمةُ تَعْني: أبَ الأميرِ [1226] يُنظر: ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 22). .
أُتْرُجُّ:
الأتُرُجُّ: ثَمَرٌ مِن جِنْسِ اللَّيْمونِ، والمُفرَدُ: تُرْنُجَّةٌ، وأُتْرُجَّةٌ، عَربيَّتُه: "المُتْكُ" [1227] يُنظر: ((الصحاح)) للجوهري (4/ 1607). ؛ قالَ عَلْقَمةُ بنُ عَبْدةَ:
يَحمِلْنَ أُتْرُجَّةً نَضْحُ العَبيرِ بها
كأنَّ تَطيابَها في الأنْفِ مَشْمومُ
والعامَّةُ تَقولُ: أُتْرُنْجٌ وتُرُنْجٌ [1228] يُنظر: ((لسان العَرب)) لابن منظور (2/ 218). ؛ قالَ مُسلِمُ بنُ الوَليدِ:
جَزى اللهُ مَن أهْدى التُّرُنْجَ تَحيَّةً
ومَنَّ بما يَهْوى عليهِ وعَجَّلا [1229] يُنظر: ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 22، 23).
وهو مِن السُّرْيانيَّةِ: (etrovga) [1230] يُنظر: ((الألْفاظ السريانية في المعاجم العَربيَّة)) لأفرام الأوَّل برصوم (ص: 314)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 22، 23). .
أَتون:
الأَتونُ: مَوقِدُ النَّارِ، والأَتونُ مُخفَّفٌ مِن الأتُّونِ، ويُجمَعُ على أُتُنٍ، وأتاتينَ.
واللَّفْظُ مُتردَّدٌ في عَربيَّتِه، تَردَّدَ فيه الجَوْهَريُّ؛ فقالَ: (ويُقالُ: هو مُولَّدٌ) [1231] ((الصحاح)) للجوهري (5/ 2067). ، وقالَ ابنُ دُرَيْدٍ: (فأمَّا الأَتونُ ... فلا أدْري ما صِحَّتُه في العَربيَّةِ) [1232] ((جمهرة اللُّغة)) لابن دُرَيْدٍ (2/ 1033). ، وجاءَ في اللِّسانِ: (ولا أحْسَبُه عَربيًّا) [1233] ((لسان العَرب)) لابن منظور (13/ 7). ، وهو مِن السُّرْيانيَّةِ: (atouno) [1234] يُنظر: ((شفاء الغليل)) للشهاب الخفاجي (ص: 39)، ((قصد السبيل)) لمحمد الأمين المحبي (1/ 159، 160)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 23). .
أثيرٌ:
الأثيرُ: مِن النَّاحيةِ الفَلْسَفيَّةِ: هو المادَّةُ الأصْليَّةُ للعالَمِ، وعنْدَ القُدَماءِ: هو أنْقى جِهةٍ في الجَوِّ وأعْلاها، وفي الكيمياءِ يُقصَدُ به مَوادُّ آلِيَّةٌ تَطيرُ ولها رائِحةٌ ذَكيَّةٌ، وتُسْتحْضَرُ بعَزْلِ الماءِ عن الكُحولِ، أو باتِّحادِ الحَوامِضِ معَ الكُحولِ.
واللَّفْظُ مِن اليونانيَّةِ: (aither) مِن (aitho)، أي: احْتَرَقَ ولَمَعَ [1235] يُنظر: ((تفسير الألْفاظ الدَّخيلة)) لطوبيا العنيسي (ص: 1). .
إجَّار:
الإجَّارُ: السَّطْحُ الَّذي ليس عليه سُتْرةٌ، ويُجمَعُ على: أجاجِرةٍ، وأجاجيرَ، وقدْ جاءَ في حَديثِ الهِجْرةِ: ((فتَلقَّى النَّاسُ رَسولَ اللهِ صلَّى اللهُ عليه وسلَّمَ في السُّوقِ وعلى الأجاجيرِ)) [1236] أخرجه أحمد (3) من حَديثِ البراءِ بنِ عازبٍ رَضِيَ اللهُ عنه. ، وفي الحَديثِ: ((مَن باتَ على إجَّارٍ ليس حَوْلَه ما يَرُدُّ قَدَمَيه، فقدْ بَرِئَتْ مِنه الذِّمَّةُ)) [1237] أخرجه سعيدُ بنُ منصور (2391)، وأبو نُعَيم في ((معرفة الصحابة)) (3079)، والبيهقيُّ في ((شعب الإيمان)) (4724) باختلافٍ يسيرٍ من حَديثِ زُهيرِ بنِ عبدِ اللهِ. .
جاءَ في اللِّسانِ أنَّه لُغةُ أهْلِ الشَّامِ والحِجازِ، وذَكَرَ أحْمَدُ بنُ فارِسٍ أنَّ الإجَّارَ لُغَةٌ شاميَّةٌ، ورُبَّما تَكلَّمَ بها الحِجازِيُّونَ [1238] يُنظر: ((مقاييس اللُّغة)) لأحمد بن فارس (1/ 63). ، قالَ أفْرامُ الأوَّلُ: (ومِن المَعْلومِ أنَّ المُرادَ باللُّغةِ الشَّاميَّةِ اللُّغةُ الآراميَّةُ السُّرْيانيَّةُ) [1239] ((الألْفاظ السريانية في المعاجم العَربيَّة)) لأفرام الأوَّل برصوم (ص: 314). ، فاللَّفْظُ مُعرَّبٌ مِن السُّرْيانيَّةِ: (egoro) [1240] يُنظر: ((لسان العَرب)) لابن منظور (4/ 11)، ((قصد السبيل)) لمحمد الأمين المحبي (1/ 160)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 23). .
إجَّاصٌ:
الإجَّاصُ: هو نَوْعٌ مِن الفاكِهةِ، ويُطلَقُ على الكُمَّثْرى، وأهْلُ الشَّامِ يُطلِقونَه على المِشْمِشِ [1241] يُنظر: ((تهذيب اللُّغة)) للأزهري (11/ 200)، ((لسان العَرب)) لابن منظور (7/ 3). ، ومُفرَدُه: إجَّاصةٌ، والإنْجاصُ لُغةٌ فيه تَقولُها العامَّةُ [1242] يُنظر: ((الصحاح)) للجوهري (3/ 1029). ؛ قالَ أمَيَّةُ بنُ أبي العائِذِ الهُذَليُّ يَصِفُ بَقَرةً:
يَتَرقَّبُ الخَطْبَ السَّواهِمَ كلَّها
بلَواقِحٍ كحَوالِكِ الإجَّاصِ [1243] يُنظر: ((لسان العَرب)) لابن منظور (7/ 3).
ذَكَرَ ابنُ فارِسٍ [1244] يُنظر: ((مقاييس اللُّغة)) لأحمد بن فارس (1/ 64). والجَوْهَريُّ [1245] ((الصحاح)) للجوهري (3/ 1029). أنَّه ليس عَربيًّا؛ لأنَّ الجيمَ لا تَجْتمِعُ معَ الصَّادِ في العَربيَّةِ، وقالَ ابنُ دُرَيْدٍ: (عَربيٌّ صَحيحٌ) [1246] ((الجمهرة)) لابن دُرَيْدٍ (2/ 1042). ، وقالَ أيضًا: (فأمَّا الإجَّاصُ فقدْ تَكلَّمَتْ به العَربُ، ولا أدْري ما صِحَّتُه) [1247] يُنظر: ((الجمهرة)) لابن دُرَيْدٍ (1/ 457). ، واللَّفْظُ مُعرَّبٌ مِن السُّرْيانيَّةِ: (agoco) [1248] يُنظر: ((الألْفاظ السريانية في المعاجم العَربيَّة)) لأفرام الأوَّل برصوم (ص: 315). .
إجَّانةٌ:
الإجَّانةُ: وِعاءٌ تُغسَلُ فيه الثِّيابُ، وفيه لُغاتٌ: الإنْجانةُ، والأجَّانةُ، وأفْصَحُها: الإجَّانةُ، والجَمْعُ: أجاجِينُ [1249] يُنظر: ((الصحاح)) للجوهري (5/ 2126) .
وفي لِسانِ العَربِ: (وهو بالفارِسيَّةِ إكَّانه) [1250] ((لسان العَرب)) لابن منظور (13/ 8). ، وهو في السُّرْيانيَّةِ: agono، agonto، ولَعلَّه مِن الكَلِماتِ المُشْترَكةِ بيْنَ السَّاميَّاتِ [1251] يُنظر: ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 24). .
آجُرٌّ:
الآجُرُّ: التُّرابُ الَّذي يُحكَمُ عَجْنُه وتَقْريصُه، ثُمَّ يُحرَقُ، وفيه لُغاتٌ: الآجُرُّ، والآجُرُ، والآجورُ، والياجورُ، والآجُرونُ، والآجِرونُ؛ قالَ أبو دوادَ الإياديُّ:
ولقد كانَ ذا كَتائِبَ خُضْرٍ
وبَلاطٍ يُشادُ بالآجُرونِ

إخْشيدٌ:
الإخْشِيدُ: مَلِكُ المُلوكِ، وهو لَقَبُ مُحمَّدِ بنِ طُغُجَ.
واللَّفْظُ بلُغةِ أهْلِ فَرْغانةَ [1253] يُنظر: ((تاج العروس)) للزبيدي (8/ 57)، ((شفاء الغليل)) للشهاب الخفاجي (ص: 43). .
أُخْطُبوطٌ:
الأُخْطُبوطُ: الحَيَوانُ البَحْريُّ ذو الأرْجُلِ الثَّمانيةِ.
واللَّفْظُ مِن اليونانيَّةِ:
آخُور عامِّيٌّ: ماخُور:
الآخُورُ: المَكانُ المُعَدُّ للخَيْلِ (الإصْطَبْلُ)، ودَخَلَ العاميَّةَ فقيلَ: ماخُورٌ.
واللَّفْظُ مُعرَّبٌ مِن الفارِسيَّةِ: آخورٌ [1255] يُنظر: ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 14). .
إدْرَونٌ:
الإدْرَونُ: الأصْلُ، وهو مَعلَفُ الدَّوابِّ، وهو الوَطَنُ.
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (اندر)، بمَعْنى: الدَّاخِلِ أو (في)، أو هو مُعرَّبٌ مِن اليونانيَّةِ [1256] يُنظر: ((لسان العَرب)) لابن منظور (13/ 153، 154)، ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 8). .
آذارُ:
آذارُ: الشَّهْرُ الثَّالِثُ مِن السَّنةِ الشَّمْسيَّةِ (مارس)، وهو واحِدٌ وثَلاثونَ يَوْمًا، يكونُ الرَّبيعُ في الحادي والعِشْرينَ مِنه.
واللَّفْظُ مِن البابِليَّةِ، وقيلَ: مِن الفَهْلَويَّةِ [1257] يُنظر: ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 14). .
آذَرْيَونٌ:
آذَرْيونُ: زَهْرٌ لَوْنُه أصْفَرُ في وَسَطِه خَمْلٌ أسْوَدُ، وتُعظِّمُه الفُرْسُ ويَعتَقِدونَ فيه البَرَكةَ؛ لِذا يَضَعونَه خَلْفَ أُذُنِهم؛ يقولُ ابنُ المُعتَزِّ:
وأردفَ آذَرْيونةً فوقَ أُذُنِهِ
ككَأسِ عَقيقٍ في قَرارتِها تِبْرُ
وقالَ ابنُ الرُّوميِّ:
كأنَّ آذَرْيونَها
والشَّمْسُ فيه كالِيهْ
مَداهِنٌ مِن ذَهَبٍ
فيها بَقايا غالِيهْ
واللَّفْظُ مُعرَّبٌ مِن الفارِسيَّةِ: آذرگون، أي: لَوْنُ النَّارِ [1258] يُنظر: ((القاموس المحيط)) للفيروزابادي (ص: 1175)، ((قصد السبيل)) لمحمد الأمين المحبي (1/ 139)، ((تاج العروس)) للزبيدي (34/ 160)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 14). .
إرانٌ:
الإرانُ: تابوتُ المَوْتى، ويُطلَقُ على تابوتِ العَهْدِ لبَني إسْرائيلَ؛ قالَ طَرَفةُ بنُ العَبْدِ:
أَمونٍ كأَلْواحِ الإِرانِ نَسَأْتُها
على لاحِبٍ كأنَّه ظَهْرُ بُرْجُدِ
واللَّفْظُ مِن العِبْريَّةِ: (إيرون)، ولم تُشِرِ المَعاجِمُ القَديمةُ إلى كَوْنِه مُعرَّبًا [1259] يُنظر: ((لسان العَرب)) لابن منظور (13/ 14، 15)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 25). .
إرْتِوازِيٌّ:
الأَرْتِوازِيُّ: صِفةٌ للبِئْرِ، والبِئْرُ الأَرْتِوازِيَّةُ: بِئْرٌ تُحفَرُ بمِثقَبٍ حتَّى يَصِلَ إلى المِياهِ الجَوْفيَّةِ، فتَتَدفَّقُ بقوَّةٍ، وهي البِئْرُ الَّتي تُؤسَّسُ على نَظَريَّةِ سُطوحِ السَّوائِلِ في الأواني المُسْتطرَقةِ.
واللَّفْظُ مِن الفَرَنْسيَّةِ مَنْسوبٌ لمَدينةِ (artois)، وهو لَفْظٌ مُسْتحدَثٌ [1260] يُنظر: ((مُعجَم الدَّخيل في اللُّغة العَربيَّة الحديثة)) للدكتور/ ف. عبد الرحيم (ص: 22)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 25). .
أَرُزٌّ:
الأَرُزُّ: نَوْعٌ مِن الحُبوبِ يُؤكَلُ، وفيه لُغاتٌ: أُرُزٌّ، وأَرُزٌ، وأُرُزٌ، وأُرْزٌ، ورُزٌّ، ورُنْزٌ، واللُّغةُ الَّتي بالنُّونِ لُغةُ عبْدِ القَيْسِ [1261] يُنظر: ((تاج العروس)) للزبيدي (15/ 165). ، واللُّغةُ المَشْهورةُ عنْدَ الخَواصِّ: أَرُزٌّ، والرُّزُّ هي اللُّغةُ المَشْهورةُ عنْدَ العَوامِّ [1262] يُنظر: ((تاج العروس)) للزبيدي (15/ 11). ؛ قالَ الرَّاجِزُ:
يا خَليلي كُلْ إوَزَّةْ
واجْعَلِ الحَوْذانَ رُنْزَةْ
ذَكَرَ الدُّكْتورُ/ ف. عبْدُ الرَّحيمِ أنَّ أصْلَه يونانيٌّ، وهو في اللَّاتينيَّةِ oryza، وهو دَخيلٌ في اللُّغةِ اليونانيَّةِ مِن اللُّغةِ التَّامِليَّةِ، وهو لُغةُ مِدْراسَ في جَنوبِ الهِنْدِ، وأصْلُه فيها: (أَرِسِ)، ومِن اللَّفْظِ اليونانيِّ riz بالفَرَنْسيَّةِ، وrice بالإنْجليزيَّةِ، وriso بالإيطاليَّةِ، وreis بالألْمانيَّةِ [1263] يُنظر: ((المعرَّب)) للجواليقي (ص: 142، 143). .

إزْمِيلٌ:
الإزْمِيلُ: حَديدةٌ شَبيهةٌ بالهِلالِ تُوضَعُ في طَرَفِ رُمْحٍ لصَيْدِ بَقَرِ الوَحْشِ، وهي المِطْرَقةُ، والرَّجُلُ الإزْميلُ: شَديدُ الأكْلِ؛ قالَ عَبْدةُ بنُ الطَّبيبِ:
عَيْرانةٌ يَنْتحي في الأرْضِ مَنسِمُها
كما انْتَحى في أديمِ الصَّرْفِ إزْميلُ
لم تُشِرِ المَعاجِمُ القَديمةُ إلى أنَّها غيْرُ عَربيَّةٍ، والرَّاجِحُ أنَّها مِن اليونانيَّةِ: (smill-ee) و(smill-on) بمَعْنى: شَفْرةِ الحِذاءِ [1267] يُنظر: ((تاج العروس)) للزبيدي (29/ 139)، ((اصطلاحات يونانية)) لبندلي جوزي (ص: 339)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 29). .
أُسْتاذٌ:
الأُسْتاذُ: المُعلِّمُ ورَئيسُ الصِّناعةِ والماهِرُ بالشَّيءِ، وكانوا يُعظِّمونَ به الخَصِيَّ؛ لأنَّه قد يكونُ تحتَ يَدِه غِلْمانٌ يُرَبِّيهم، وقدْ وَصَفَ بها المُتنبِّي كافورَ، وكانَ مُرَبِّيَ أوْلادِ الإخْشيدِ:
تَرَعْرَعَ المَلِكُ الأسْتاذُ مُكْتهِلًا
قبْلَ اكْتِهالٍ أدَيبًا قبْلَ تَأديبِ
وهي مُعرَّبةٌ مِن الفارِسيَّةِ: (أُسْتاد) بمَعْنى: المُعلِّمِ والماهِرِ بصَنْعتِه، وقدْ ذَكَرَ الجَواليقيُّ دَليلًا على كَوْنِ الكَلِمةِ أعْجَميَّةً أنَّها لو لم تَكُنْ أعْجَميَّةً لكانت مُشتَقَّةً مِن (ستذ)، وهو جَذْرٌ غيْرُ مَعْروفٍ في اللُّغةِ [1268] يُنظر: ((المعرَّب)) للجواليقي (ص: 125). ، وذَكَرَ الفَيُّوميُّ [1269] يُنظر: ((المصباح المنير)) للفيومي (1/ 14). أنَّها أعْجَميَّةٌ؛ لأنَّ السِّينَ والذَّالَ لا تَجْتَمِعانِ في كَلِمةٍ عَربيَّةٍ [1270] يُنظر: ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 10)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 31). .
إسْتارٌ:
الإسْتارُ: مِن العَدَدِ: الأرْبَعةُ، وهو رابِعُ أرْبَعةٍ، ورابِعُ القَوْمِ، وجَمْعُه: أساتيرُ؛ قالَ جَريرٌ:
إِنَّ الفَرَزْدَقَ والبَعيثَ وأُمَّهُ
وأَبا البَعيثِ لشَرُّ ما إِسْتارِ
وقالَ الأخْطَلُ:
لَعَمْرُك إنِّني وابْنَيْ جُعَيْلٍ
وأُمَّهُما لإسْتارٌ لَئيمُ
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (چهار)، بمَعْنى: أرْبعةٍ [1271] يُنظر: ((لسان العَرب)) لابن منظور (4/ 344، 345). .
إسْرِنْجٌ:
الإسْرِنْجُ: صِبْغٌ أحْمَرُ يُصنَعُ مِن الرَّصاصِ المَحْروقِ؛ قالَ الصَّنَوْبريُّ:
إلى لازوَرْدٍ وفَيْروزَجٍ
وماذينَجِ اللَّوْنِ إسْرِنْجِهِ
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (سِرِنْج)، وهو صِبْغٌ أحْمَرُ مَصْنوعٌ مِن حَريقِ الرَّصاصِ والقِليِ، مُركَّبٌ مِن (سُرْخ)، أي: أحْمَرَ، و(آنُك)، أي: الرَّصاصِ [1272] يُنظر: ((تاج العروس)) للزبيدي (6/ 40)، ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 10)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 32، 33). .
إسْطارٌ:
الإسْطارُ: نَقْدٌ مِن ذَهَبٍ أو فِضَّةٍ، الذَّهَبيُّ مِنه يُساوي عِشْرينَ دِرْهَمًا، والفِضِّيُّ يُساوي أرْبَعةَ دَراهِمَ.
وهو مُعرَّبٌ مِن اليونانيَّةِ: (state-er)، بمَعْنى: نَقْدٍ له وَزْنٌ مُحدَّدٌ مِن الذَّهَبِ أو الفِضَّةِ [1273] يُنظر: ((اصطلاحات يونانية)) لبندلي جوزي (ص: 339). .
أَسْطُرْلابٌ:
الأَسْطُرْلابُ: جِهازٌ اسْتَعْملَه القُدَماءُ في تَحْديدِ ارْتِفاعاتِ الأجْرامِ السَّماوِيَّةِ، والجِهاتِ الأصْليَّةِ، والوَقْتِ، ومَعْرفةِ مَواضِعِ الكَواكِبِ وحَرَكتِها [1274] يُنظر: ((المُعجَم الوسيط)) الصادر عن مجمع اللُّغة العَربيَّة بالقاهرة (1/ 17). ، والأَصْطُرْلابُ لُغةٌ فيه [1275] يُنظر: ((تاج العروس)) للزبيدي (4/ 223). .
واللَّفْظُ مُعرَّبٌ مِن اليونانيَّةِ: (Astron)، بمَعْنى: النَّجْمِ، والكَوكَبِ، و(lambanein) [1276] يُنظر: ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 33). ، أي: أخْذٌ؛ فهي: أخْذُ الكَوكَبِ؛ قالَ الزَّبيديُّ في كَوْنِها مُعرَّبةً: (فإنَّ أهْلَ الهَيْئةِ صَرَّحوا بأنَّها روميَّةٌ، مَعْناها: الشَّمْسُ) [1277] ((تاج العروس)) للزبيدي (4/ 224). .
أُسْطولٌ:
الأُسْطولُ: مَجْموعةٌ مِن السُّفُنِ الَّتي تُبحِرُ معًا، سواءٌ كانَت للحَرْبِ أو للتِّجارةِ والنَّقْلِ، ويُطلَقُ على مَجْموعةِ الطَّائِراتِ للأُسْطولِ الجَوِّيِّ، والجَمْعُ: أساطيلُ؛ قالَ الآمِدِيُّ:
أَعْجِبْ بأُسْطولِ الإمامِ مُحمَّدٍ
وبحُسْنِهِ وزَمانِهِ المُسْتغرَبِ
واللَّفْظُ مُعرَّبٌ مِن اليونانيَّةِ: (stol-os) [1278] يُنظر: ((تاج العروس)) للزبيدي (29/ 199)، ((المُعجَم الوسيط)) الصادر عن مجمع اللُّغة العَربيَّة (1/ 17)، ((اصطلاحات يونانية)) لبندلي جوزي (ص: 340). .
إِسْفَنْجٌ:
الإسْفَنْجُ: حَيَوانٌ بَحْريٌّ نَباتيٌّ، رِخْوُ الجِسْمِ، ذو مَسامَّ واسِعةٍ، ويُطلَقُ أيضًا على جِسْمٍ يُتَّخَذُ مِن الحَيَوانِ السَّابِقِ، كَثيرُ المَسامِّ واسِعُها، يُسْتخدَمُ في الاسْتِحمامِ والتَّنْظيفِ لقوَّةِ امْتِصاصِه للماءِ، والمُفرَدُ: إسْفَنْجةٌ [1279] يُنظر: ((المُعجَم الوسيط)) الصادر عن مجمع اللُّغة العَربيَّة بالقاهرة (1/ 18). .
واللَّفْظُ مُعرَّبٌ مِن اليونانيَّةِ: (spoggos)، ومَعْناه: المُنْعَصِرُ [1280] يُنظر: ((تاج العروس)) للزبيدي (6/ 41)، ((تفسير الألْفاظ الدَّخيلة)) لطوبيا العنيسي (ص: 3). .
إسْفَنْطٌ:
الإسْفَنْطُ: اسمٌ مِن أسْماءِ الخَمْرِ، وهو أعْلى الخَمْرِ وأصْفاها، ويُقالُ له أيضًا: الإسْفَنْدُ [1281] يُنظر: ((لسان العَرب)) لابن منظور (7/ 255، 256). ؛ قالَ الأعْشى:
وكأنَّ الخَمْرَ العَتيقَ مِن الإسْ‍
فَنْطِ مَمْزوجةٌ بماءٍ زُلالِ
ذَكَرَ الجَوْهَريُّ أنَّه مُعرَّبٌ مِن الفارِسيَّةِ [1282] يُنظر: ((الصحاح)) للجوهري (3/ 1131). ، وجاءَ عن الأصْمَعيِّ وابنِ السِّكِّيتِ: أنَّه مِن الرُّوميَّةِ، وهو الصَّحيحُ، فاللَّفْظُ مِن اللَّاتينيَّةِ [1283] يُنظر: ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 35). .
إسْفيداجٌ:
الإسْفيداجُ: رَمادُ الرَّصاصِ، وطينٌ يُجلَبُ مِن "أصْفَهانَ" يَكتُبُ به الصِّغارُ.
واللَّفْظُ مُعرَّبٌ مِن الفارِسيَّةِ: (اسْفيداب)، وأصْلُ مَعْناه: الماءُ الأبْيَضُ [1284] يُنظر: ((تاج العروس)) للزبيدي (6/ 40)، ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 10). .
إسْكافٌ:
الإسْكافُ: الصَّانِعُ الحاذِقُ، ويُطلَقُ على صانِعِ الخِفافِ، ثُمَّ أُطلِقَتْ على مُصلِحِ النِّعالِ والأحْذيةِ، وفيه لُغاتٌ: الأُسْكوفُ، والأَسْكَفُ، والأُسْكُفُّ، والسَّكَّافُ، والسَّيْكَفُ، وجَمْعُه: أساكِفةٌ.
واللَّفْظُ مُعرَّبٌ مِن الفارِسيَّةِ: (كَفْشْ كر)، أي: صانِعُ حِذاءٍ [1285] يُنظر: ((لسان العَرب)) لابن منظور (9/ 156، 157)، ((القاموس المحيط)) للفيروزابادي (ص: 820)، ((قصد السبيل)) لمحمد الأمين المحبي (1/ 185)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 35). .
إسْكَنْدَر:
إسْكَنْدرُ وأسْكَنْدَرُ: عَلَمٌ يونانيٌّ، ولا صِحَّةَ لقَصْرِه على الاسْتِعمالِ بالألِفِ واللَّامِ؛ إذ جاءَ في شِعْرِ أبي تَمَّامٍ بحَذْفِها:
مِن عَهْدِ إِسْكَنْدَرٍ أو قبْلَ ذلك قد
شابَتْ نَواصي اللَّيالي وهْي لمْ تَشِبِ
والمُتَعارَفُ أنَّها تكونُ بالألِفِ واللَّامِ؛ مِثلُ: الفَرَزْدَقِ.
ولا خِلافَ في كَوْنِه أعْجَميًّا، وهو مِن اليونانيَّةِ: (ألِكْسَنْدر) [1286] يُنظر: ((المعرَّب)) للجواليقي (ص: 150)، ((شفاء الغليل)) للشهاب الخفاجي (ص: 36)، ((تاج العروس)) للزبيدي (12/ 68). .
أُشْنان:
الأُشْنانُ والإِشْنانُ: نَباتٌ يُغسَلُ به كالصَّابونِ، وعَربيَّتُه: الحُرُضُ، واشْتَقُّوا مِنه فِعْلًا وهو تَأشَّنَ، أي: غَسَلَ يَدَه.
وهو مِن الفارِسيَّةِ: (أُشْنان) [1287] يُنظر: ((جمهرة اللُّغة)) لابن دُرَيْدٍ (1/ 515)، ((المعرَّب)) للجواليقي (ص: 124، 125)، ((القاموس المحيط)) للفيروزابادي (ص: 1176). .
إصْطَبْلٌ:
الإصْطَبْلُ: مَأوى الخَيْلِ، ويُجمَعُ على: إصْطَبْلاتٍ [1288] يُنظر: ((المصباح المنير)) للفيومي (1/ 16). ، وأصاطِبَ [1289] يُنظر: ((تاج العروس)) للزبيدي (27/ 453). ، وأصابِلَ [1290] يُنظر: ((تهذيب اللُّغة)) للأزهري (12/ 191). ؛ قالَ أبو نُخَيْلةَ:
لولا أبو الفَضْلِ ولولا فَضْلُه
لسُدَّ بابٌ لا يُسَنَّى قَفْلُه
ومِنْ صَلاحِ راشِدٍ إِصْطَبْلُه [1291] يُنظر: ((لسان العَرب)) لابن منظور (11/ 378).
ذَكَرَتِ المَعاجِمُ القَديمةُ أنَّ اللَّفْظَ مِن اللُّغةِ الشَّاميَّةِ، وهو مُعرَّبٌ مِن اللَّاتينيَّةِ: (stabulum)، وأُخِذَتْ مِنه الكَلِمةُ الإنْجليزيَّةُ (stable) [1292] يُنظر: ((المعرَّب)) للجواليقي (ص: 112، 113)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 40). .
آغا:
الآغا: الخَصِيُّ، وهُمْ رِجالٌ مَخْصيُّونَ يَقومونَ بالخِدْمةِ في الحَرَمَينِ الشَّريفَينِ. وفي العِراقِ: لَقَبٌ يَدُلُّ على الاحْتِرامِ، ويُجمَعُ على: أغَواتٍ.
وأصْلُ اللَّفْظِ مِن التُّرْكيَّةِ: ag`a، وأصْلُ مَعْناه: السَّيِّدُ [1293] يُنظر: ((مُعجَم الدَّخيل في اللُّغة العَربيَّة الحديثة)) للدكتور/ ف. عبد الرحيم (ص: 30). .
إِفْرِنْد:
الإفْرِنْدُ: جَوهَرُ السَّيْفِ ووَشْيُه، ويُطلَقُ على السَّيْفِ نفْسِه، وهو الفِرِنْدُ والبِرِنْدُ، ويُطلَقُ الفِرِنْدُ على الحَريرِ أيضًا؛ قالَ عَنْتَرةُ:
إذا لم أُرَوِّ صارمي مِن دَمِ العِدا
ويُصبِحُ مِن إفْرِنْدِه الدَّمُ يَقطُرُ
وبمَعْنى السَّيْفِ نفْسِه قالَ جَريرٌ:
وقدْ قَطَعَ الحَديدَ فلا تُماروا
فِرِنْدٌ لا يُفَلُّ ولا يَذوبُ
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (پَرَند) [1294] يُنظر: ((الصحاح)) للجوهري (2/ 519)، ((لسان العَرب)) لابن منظور (3/ 334)، ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 119)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 42). .
إفْريز:
الإفْريزُ: ما يَبرُزُ خارِجًا عن البِناءِ.
وهو مُعرَّبٌ [1295] يُنظر: ((لسان العَرب)) لابن منظور (5/ 391)، ((القاموس المحيط)) للفيروزابادي (ص: 520). مِن الفارِسيَّةِ: (أفرايزيدن) [1296] يُنظر: ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 43). ، أو مِن اليونانيَّةِ: (ophry`-s) [1297] يُنظر: ((اصطلاحات يونانية)) لبندلي جوزي (ص: 340). .
أَفَنْدي عامِّيٌّ:
الأفَنْدي: لَقَبٌ مَعْناه: السَّيِّدُ الكَبيرُ على الإطْلاقِ، وهو المُسْتقِلُّ بذاتِه والمُعتَمِدُ على نفْسِه، وقدْ وُصِفَ به الأمَراءُ الأتْراكُ على سَبيلِ التَّبْجيلِ، وكانَ يُطلَقُ في مِصْرَ على المُثقَّفِ ثَقافةً غَربيَّةً، يقولونَ: أشَيْخٌ هو أم أفَنْدي؟ وفي الشَّامِ يُطلَقُ على كلِّ مَن له وَجاهةٌ واحْتِرامٌ، وقدْ يُقالُ: أفَنْدِم، ويُسْتَعمَلُ كجَوابٍ مُحْترَمٍ عنِ النِّداءِ، أو كطَلَبٍ مِن المُخاطَبِ أن يُعيدَ كَلامَه، وتَطوَّرَ اللَّفْظُ مِن مَعْنى: السَّفَّاحِ والقاتِلِ بيَدِه والجَلَّادِ، ثُمَّ أصْبَحَ: المُستَبِدَّ والسَّيِّدَ الظَّالِمَ، ثُمَّ أُطْلِقَ على السَّيِّدِ مُطلَقًا [1298] يُنظر: ((اصطلاحات يونانية)) لبندلي جوزي (ص: 340). .
واللَّفْظُ أصْلُه مِن اليونانيَّةِ: (afthentes/ afthendes)، ودَخَلَ العَربيَّةَ مِن التُّرْكيَّةِ (efandi)، ومَعْناه في اللُّغتَينِ: السَّيِّدُ [1299] يُنظر: ((تاج العروس)) للزبيدي (8/ 509)، ((مُعجَم الدَّخيل في اللُّغة العَربيَّة الحديثة)) للدكتور/ ف. عبد الرحيم (ص: 31)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 44). .
أَفْيُون:
الأَفْيونُ: عُصارةُ ثَمَرِ الخَشْخاشِ، وهو مُخدِّرٌ للعَقْلِ، تُؤدِّي المُداوَمةُ عليه إلى الإدْمانِ.
واللَّفْظُ مَوْجودٌ في كُلِّ اللُّغاتِ، وهو في الفارِسيَّةِ: أَپيون، وأصْلُه مِن اليونانيَّةِ: (opion)، وفي تاجِ العَروسِ ما يَشِي بأنَّ الكَلِمةَ عَربيَّةٌ؛ إذ ذُكِرَ أنَّ وَزْنَها (أَفْعُول) [1300] يُنظر: ((تاج العروس)) للزبيدي (35/ 524)، ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 11)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 44). .
إقْليمٌ:
الإقْليمُ: القِسْمُ مِن الأرْضِ، الجَمْعُ: أقاليمُ.
واخْتُلِفَ في عَربيَّتِه؛ فابنُ دُرَيْدٍ يَراه مُعرَّبًا [1301] يُنظر: ((جمهرة اللُّغة)) لابن دُرَيْدٍ (2/ 1193). ، وسائِرُ أصْحابِ المَعاجِمِ يَرَوْنَه عَربيًّا مُشتَقًّا مِن القَلْمِ، أي: القَطْعِ؛ كأنَّه مَقْلومٌ (مَقْطوعٌ) مِن الإقليمِ الَّذي يُتاخِمُه؛ قالَ الأزْهَريُّ: (وأحْسَبُه عَربيًّا) [1302] ((تهذيب اللُّغة)) للأزهري (9/ 149). ، والصَّحيحُ ما ارْتَآه ابنُ دُرَيْدٍ؛ فاللَّفْظُ مُعرَّبٌ مِن اليونانيَّةِ: (klim-a)، وأصْلُ مَعْناها: المُنْحدِرُ، ثُمَّ اسْتُعمِلَ للدَّلالةِ على المَعْنى الجُغْرافيِّ [1303] يُنظر: ((المعرَّب)) للجواليقي (ص: 120، 121). .
أَقْلَش:
الأقْلَشُ: المُلاعِبُ الَّذي لا يَملِكُ شيئًا أو الَّذي لا يَثبُتُ على شيءٍ واحِدٍ، وهو القَلَّاشُ [1304] يُنظر: ((تاج العروس)) للزبيدي (17/ 340). .
جاءَ في اللِّسانِ: (وهو دَخيلٌ؛ لأنَّه ليس في كَلامِ العَربِ شينٌ بعْدَ لامٍ في كَلِمةٍ عَربيَّةٍ مَحْضةٍ) [1305] ((لسان العَرب)) لابن منظور (6/ 337). .
أُقْنومٌ:
الأُقْنومُ: الأصْلُ، والجَمْعُ: أقانيمُ، ويُطلَقُ على أرْكانِ الثَّالوثِ في المُعْتَقَدِ المَسيحيِّ، يقولونَ: الأقانيمُ الثَّلاثةُ.
واللَّفْظُ مُعرَّبٌ مِن اليونانيَّةِ [1306] يُنظر: ((الصحاح)) للجوهري (5/ 2016)، ((قصد السبيل)) لمحمد الأمين المحبي (1/ 204)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 45). ، وذَكَرَ طوبيا العُنَيْسيُّ أنَّه مِن السُّرْيانيَّةِ: قنوما، ومَعْناه: الشَّخْصُ [1307] يُنظر: ((تفسير الألْفاظ الدَّخيلة)) لطوبيا العنيسي (ص: 4). .
أكاديميَّةٌ:
الأكاديميَّةُ: مَدْرسةُ التَّعْليمِ العالي، والمَجمَعُ العِلميُّ أو اللُّغويُّ أو الأدَبيُّ.
وهو مِن اليونانيَّةِ: (academeia)، وهو في الأصْلِ اسمُ حَديقةٍ قَريبةٍ مِن أثينا كانَ أفْلاطونُ يُلْقي فيها دُروسَه، وقدْ دَخَلَ في مُعظَمِ اللُّغاتِ الأوْروبيَّةِ؛ ففي الإنْجليزيَّةِ: (academy)، وفي الفَرَنْسيَّةِ: (academie)، وفي الألْمانيَّةِ: (akademie)، وفي الإيطاليَّةِ: (accademia) [1308] يُنظر: ((مُعجَم الدَّخيل في اللُّغة العَربيَّة الحديثة)) للدكتور/ ف. عبد الرحيم (ص: 30، 31)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 45). .
إكْسيرٌ:
الإكْسيرُ: مادَّةٌ مُركَّبةٌ كانَ يُعْتقَدُ قَديمًا أنَّها تُحوِّلُ المَعْدِنَ الرَّخيصَ إلى ذَهَبٍ، أو هو شَرابٌ كانَ يُعْتقَدُ أنَّه يُطيلُ الحَياةَ، ويُبْقي الشَّبابَ، ويَكفُلُ للإنْسانِ الخُلودَ [1309] يُنظر: ((شفاء الغليل)) للشهاب الخفاجي (ص: 40)، ((المُعجَم الوسيط)) الصادر عن مجمع اللُّغة العَربيَّة (1/ 22). .
وهو مِن اليونانيَّةِ: (kseeri-on) [1310] يُنظر: ((اصطلاحات يونانية)) لبندلي جوزي (ص: 340). .
أكَّار:
الأكَّارُ: الفَلَّاحُ والحَرَّاثُ، والجَمْعُ: أَكَرةٌ، وفي حَديثِ قَتْلِ أبي جَهْلٍ: ((فلو غيْرُ أكَّارٍ قَتَلَني )) [1311] أخرجه البخاري (4020)، ومسلم (1800) من حَديثِ أنس بن مالك رَضِيَ اللهُ عنه. .

ألْماسٌ عامِّيٌّ:
الألْماسُ: حَجَرٌ مِن الجَواهِرِ، يَكسِرُ جَميعَ الأجْسامِ الحَجَريَّةِ، ولا تُؤثِّرُ فيه النَّارُ ولا الحَديدُ، بلْ يَكسِرُه الرَّصاصُ، وفيه قَوْلانِ: أحَدُهما: أنَّ الهَمْزةَ واللَّامَ فيه أصْليَّةٌ مِثلُه مِثلُ: إِلْياسَ [1313] يُنظر: ((النهاية في غريب الحديث والأثر)) لابن الأثير (4/ 289). ، والثَّاني: أنَّ قَطْعَ الهَمْزةِ فيه مِن لَحْنِ العامَّةِ [1314] يُنظر: ((القاموس المحيط)) للفيروزابادي (ص: 576)، ((تاج العروس)) للزبيدي (16/ 525، 526). .
وهو مِن اليونانيَّةِ عن طَريقِ اللَّاتينيَّةِ: (adamas) [1315] يُنظر: ((شفاء الغليل)) للشهاب الخفاجي (ص: 37)، ((قصد السبيل)) لمحمد الأمين المحبي (1/ 207، 208)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 47). .

أنْبارٌ:
الأنْبارُ: المُسْتودَعُ أو المَحَلُّ الَّذي يكونُ فيه الطَّعامُ، ومُفرَدُه: نِبْرٌ.
واللَّفْظُ مِن الفارِسيَّةِ: (انبار) [1317] يُنظر: ((الجمهرة)) لابن دُرَيْدٍ (1/ 329). ، بمَعْنى: الامْتِلاءِ، وذَكَرَ الدُّكْتورُ/ ف. عبْدُ الرَّحيمِ أنَّ العَربَ قدِ اعْتَبَروا اللَّفْظَ جَمْعًا؛ لأنَّه على وَزْنِ (أفْعالٍ)؛ ومِن ثَمَّ فقدْ قالوا له مُفرَدًا: نِبْرٌ، وكَلِمةُ العَنْبَرِ في لُغتِنا العامِّيَّةِ بمَعْنى جَناحٍ مِن أجْنِحةِ المُسْتَشْفى مَأخوذةٌ مِن amber بالتُّرْكيَّةِ، المُحرَّفِ عن الفارِسيَّةِ [1318] يُنظر: ((المعرَّب)) للجواليقي (ص: 114، 115)، ((مُعجَم الدَّخيل في اللُّغة العَربيَّة الحديثة)) للدكتور/ ف. عبد الرحيم (ص: 146). .

أنْدَلُسٌ:
الأنْدَلُسُ: شِبْهُ الجَزيرةِ الإيْبيريَّةِ الَّتي افْتَتَحَها المُسلِمونَ سَنَةَ 92 للهِجْرةِ، المَعْروفةُ اليَوْمَ بـ(إسْبانيا).
في أصْلِ الكَلِمةِ أقْوالٌ:
أنَّ أوَّلَ مَن سَكَنَها قَوْمٌ يُعرَفونَ بالأنْدَلُشِ؛ فسُمِّيَ بهم البَلَدُ، فلمَّا عُرِّبَتِ الكَلِمةُ قالوا: الأنْدَلُسُ.
سُمِّيَتْ باسمِ أوَّلِ مَن عَمَرَها، وهو أنْدَلُسُ بنُ يافِثَ بنِ نوحٍ.
أوَّلُ مَن سَكَنَ الأنْدَلُسَ بعْدَ الطُّوفانِ قَوْمٌ عُرِفوا باسمِ "الأنْدَلُسِ"، فعَمَروها وتَداوَلوا المُلْكَ فيها مُدَّةً طَويلةً.
والنَّصارى كانوا يُسمُّونَها (أشْبانيةَ) باسمِ رَجُلٍ صُلِبَ فيها يُدْعى: أشْبانِس، أو باسمِ مَلِكٍ لها في الزَّمانِ الأوَّلِ يُدْعى: (إشْبانَ بنَ طِيطَسَ) [1321] يُنظر: ((الكامل)) لابن الأثير (4/ 35)، ((شفاء الغليل)) للشهاب الخفاجي (ص: 51). .

أَيْبَك:
أيْبَكٌ: الأميرُ الجَميلُ.
وهو مِن التُّرْكيَّةِ: (آي)، وهو القَمَرُ، و(بِك)، وهو الأميرُ [1329] يُنظر: ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 55). .
إيلِياءُ:
إيلياءُ: بَيْتُ المَقدِسِ، جاءَ في العُهْدةِ العُمَريَّةِ: (ولا يَسكُنُ بإيلياءَ أحَدٌ معَهم مِن اليَهودِ)، وفيه سِتُّ لُغاتٍ: إيلِياءُ، وإيلِيا، وإيلِيَّاءُ، وإيلِيَّا، وإلْياءُ، وإلْيا؛ قالَ الفَرَزْدَقُ:
وبَيْتانِ بَيْتُ اللهِ نحن وُلاتُهُ
وبَيْتٌ بأعْلى إيلِياءَ مُشَرَّفُ
وهو مِن اللَّاتينيَّةِ: (aelia)؛ قالَ الأزْهَريُّ: (وكأنَّهما -أي: إيلياءَ وإيليا- روميَّانِ) [1330] ((تهذيب اللُّغة)) للأزهري (15/ 332). ، وهو جُزْءٌ مِن اسمِها اللَّاتينيِّ الطَّويلِ الَّذي سمَّاها به الرُّومُ، وهو colonia aelia capitolina؛ إذ سُمِّيَتْ باسمِ الإمْبراطورِ هادِرْيانَ، وكانَ اسمُه الأوَّلُ aelius، وقدْ مَدَّتْه العَربُ عنْدَ التَّعْريبِ للإلْحاقِ بطِرِمْساءَ [1331] يُنظر: ((المعرَّب)) للجواليقي (ص: 139، 140). .
أيْهُقَان:
الأيْهُقانُ: الجَرجيرُ؛ قالَ لَبيدٌ:
فعَلا فُروعُ الأَيْهُقانِ وأَطْفَلَتْ
بالجَلْهَتَيْنِ ظِباؤُها ونَعامُها
وهو مِن الفارِسيَّةِ: (ايهقان) [1332] يُنظر: ((لسان العَرب)) لابن منظور (10/ 11)، ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 13، 14). .
إِوَان:
الإِوَانُ والإِيوانُ: هو البَهْوُ الواسِعُ العالي السَّقْفِ، ويُجمَعُ الإوانُ على أُوُنٍ، ويُجمَعُ الإيوانُ على إيواناتٍ، وأواوينَ، وفي حَديثِ سَطيحٍ: ((لمَّا وُلِدَ رَسولُ اللهِ صلَّى اللهُ عليه وسلَّمَ ارْتَجَسَ إيوانُ كِسْرى)) [1333] أخرجه أبو نُعيمٍ في ((دلائل النبوة)) (82)، والبيهقيُّ في ((دلائل النبوة)) (1/126)، وابنُ عساكر في ((تاريخ دمشق)) (37/361) باختلافٍ يسيرٍ من حَديثِ هانئٍ المخزوميِّ. قال الذَّهبيُّ في ((تاريخ الإسلام)) (1/35): مُنكَرٌ غَريبٌ. [1334] يُنظر: ((تهذيب اللُّغة)) للأزهري (15/ 392)، ((لسان العَرب)) لابن منظور (13/ 40). .
صَرَّحَ ابنُ دُرَيْدٍ [1335] يُنظر: ((الجمهرة)) لابن دُرَيْدٍ (1/ 249). وابنُ مَنْظورٍ [1336] يُنظر: ((لسان العَرب)) لابن منظور (13/ 40). والجَواليقيُّ [1337] يُنظر: ((المعرَّب)) للجواليقي (ص: 113). بأنَّه مُعرَّبٌ، وذَكَرَ أَدِّي شِير أنَّه في الفارِسيَّةِ: (ايوان)، ثُمَّ رَجَّحَ أن يكونَ اللَّفْظُ في الأصْلِ مِن الآراميَّةِ، وهو فيها مُشتَقٌّ مِن: ، بمَعْنى: ضَمَّ وأوى [1338] يُنظر: ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 13). .
الآيِينُ:
الآيِينُ: العادةُ والسِّياسةُ الَّتي تَتْبَعُها الجَماعةُ، أو القانونُ والشَّريعةُ؛ يقولُ مِهْيارٌ:
يَجمَعُ الخِرِّيتُ حَوْلًا أمْرَه
وهْو لم يَأخُذْ لها آيِينَها
وهو مُعرَّبٌ مِن الفارِسيَّةِ، ذَكَرَ الشِّهابُ الخَفاجيُّ أنَّه مِن تَعْريبِ المُولَّدينَ [1339] يُنظر: ((شفاء الغليل)) للشهاب الخفاجي (ص: 39)، ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 13)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 17). .
