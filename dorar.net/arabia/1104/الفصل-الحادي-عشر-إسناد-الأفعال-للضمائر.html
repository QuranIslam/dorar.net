محتويات الصفحة
انظر أيضا

حُكمُ الصَّحيحِ السَّالمِ:
لا يتغَيَّرُ عند إسنادِه للضَّمائِرِ بأيِّ صُورةٍ مِن صُوَرِ التغْييرِ.
حُكمُ المهموزِ:
حُكمُه كالسَّالمِ، غيرَ أنَّ:
1- الأمرَ مِن (أخَذَ وأكَلَ) تُحذَفُ الهَمْزةُ منه، سواءٌ أكان في ابتداءِ الكلامِ، أو سُبِقَ بشَيءٍ، تقولُ: خُذْ وكُلْ.
2- تُحذَفُ الهَمْزةُ مِن الأمرِ مِن (أمَرَ وسألَ) في ابتداءِ الكلامِ، مِثْلُ: مُرُوا بالمعروفِ، وقَولِه تعالى: سَلْ بَنِي إِسْرَائِيلَ [البقرة: 211] ، فإن سُبِقَا بشَيءٍ جاز الحَذفُ وعَدَمُه، مِثْلُ: "قُلْتُ له: اسأَلْ، أو سَلْ".
3- المُضارِعُ والأمرُ مِن (يرى) تُحذَفُ هَمزَتُه (عينُ الفِعلِ): يَرَى، رَهْ، وأصلُ المُضارِع: يَرْأَى، نُقِلَت حركةُ الهَمْزةِ إلى ما قَبْلَها، ثم حُذِفَت لالتقائِها ساكِنةً مع ما بَعْدَها، والأمرُ مَحموٌل على المُضارِعِ، وتُحذَفُ الهَمْزةُ مِن الفِعلِ (أرى) من جميعِ التصاريفِ، تقولُ: أرَى، ويُرِي، وأرِهْ.
الفِعلُ المضاعَفُ الثُّلاثيُّ ومَزِيدُه:
1- في الماضي منه يجِبُ الإدغامُ: مَدَّ، ومَدُّوا، واسْتَمَدَّ واسْتَمَدُّوا، إلَّا إذا اتَّصل بضميِر رَفعٍ متحَرِّكٍ، فيجِبُ الفَكُّ: مَدَدْتُ، ومَدَدْنَ، واسْتَمْدَدْتُ واسْتَمْدَدْنا.
2- وفي مُضارِعِه يجِبُ الإدغامُ، مِثْلُ: يَرُدُّ ويَسْتَرِدُّ، ويَرُدُّون ويَسْتَرِدُّون، ويجوزُ الفَكُّ والإدغامُ إن كان مجزومًا بالسُّكونِ: لم يرُدَّ، ولم يَرْدُدْ، ولم يَسْتَرِدَّ ولم يَسْتَرْدِدْ، ويجِبُ الفَكُّ إنِ اتَّصَلَت به نُونُ النِّسوةِ: يَرْدُدْنَ، ويَسْتَرْدِدْنَ، فإن كان مجزومًا بغيرِ السُّكونِ وَجَب الإدغامُ: لم يَرُدُّوا، ولم يَسْتَرِدُّوا.
3- الأمرُ كالمُضارِعِ المجزومِ: رُدَّ يا زَيدُ واردُدْ، وارْدُدْنَ يا نِساءُ، ورُدُّوا.
الفِعلُ المِثالُ:
حُكمُه كالصَّحيحِ السَّالمِ إذا كان واويًّا على وَزْنِ "يَفْعِل" فإنَّ فاءَه تُحذَفُ منه في المُضارِعِ والأمرِ، مِثْلُ: وَعَدَ يَعِدُ عِدْ، وفي غيرِ ذلك لا تُحذَفُ الفاءُ.
الفِعلُ الأجوَفُ:
تَثبُتُ العَينُ فيه إن لم يكُنْ فيها إعلالٌ (قُلِبَت ألِفًا)، فإن وقع فيها إعلالٌ، فإنَّ عَينَ الأجوفِ تُحذَفُ في المُضارِعِ المجزومِ، والأمرِ، وعند اتِّصالِه بضَميرِ رَفعٍ متحَرِّكٍ، وتُحرَّكُ الفاءُ بحَرَكةٍ مِن جِنسِ المحذوفِ؛ تقولُ: لم يَقُلْ، قُلْ، قُلْتُ، قُلْنا، قُلْنَ/ لم يَبِعْ، بِعْ، بِعْتُ، بِعْنا، بِعْنَ، وفي غيرِ ذلك فلا تُحذَفُ.
الفِعْلُ النَّاقِصُ:
الماضي منه:
1- تُحذَفُ اللَّامُ مع واوِ الجماعةِ، فإن كانت اللَّامُ ألِفًا، بَقِيَت الفتحةُ دليلًا عليها، وإن كانت اللَّامُ واوًا أو ياءً، ضُمَّ ما قَبْلَ واوِ الجماعةِ، تقولُ: سَعَوا، وسَرُوا ورَضُوا.
2- وإن كان مُسنَدًا لغيرِ واوِ الجماعةِ، لم تُحذَفِ اللَّامُ، فإن كان ثُلاثيًّا: تبقى اللَّامُ على الأصلِ، وتُردُّ إلى أصلِها (الواوِ أو الياءِ) إن كانت ألفًا، تقولُ: سَرُونا، ورَضِينا، وغَزَونا، ورَمَينا، وغَزَوَا ورَمَيَا، فإن كان أزيَدَ مِن ثلاثةِ أحرُفٍ: قُلِبَت اللَّامُ ياءً مُطلَقًا، تقولُ: أعْطَيْتُ، واسْتَعْطَيتُ.
3- وإن اتَّصَلَت به تاءُ التأنيثِ، فإن كانت اللَّامُ ألفًا، فإنَّها تُحذَفُ، تقولُ: رَمَتْ، وغَزَتْ، فإن كان واوًا أو ياءً، فلا تُحذَفُ: سَرُوَتْ، رَضِيَتْ.
المُضارِعُ منه:
1- تُحذَفُ اللَّامُ إذا أُسنِدَ إلى واوِ الجَماعةِ وياءِ المخاطَبةِ المُؤَنَّثةِ، فإن كانت اللَّامُ المحذوفةُ ألفًا، فُتِحَ ما قَبْلَ الضَّميرِ، وإن كانت واوًا أو ياءً، ضُمَّ ما قَبْلَ واوِ الجماعةِ وكُسِرَ ما قَبْلَ ياءِ المخاطَبةِ، مِثْلُ: الرِّجالُ يَسْعَونَ، وأنتِ تَسْعِينَ، والرِّجالُ يَغْزُون ويَرْمُون، وأنتِ تَغْزِين وتَرْمِينَ.
2- لا تُحذَفُ اللَّامُ مع نُونِ النِّسوةِ وألِفِ الاثنَينِ، وتُقلَبُ الألِفُ ياءً، وما قَبْلَ نونِ النِّسْوةِ يكونُ ساكِنًا، وما قَبْلَ ألِفِ الاثنَينِ يكونُ مُتحَرِّكًا بالفَتحِ، تقولُ: يَسْعَيْن، يَغْزُونَ، ويَرْمِينَ، ويَسْعَيَانِ ويَغْزُوَانِ ويَرْمِيَانِ.
الأمرُ منه كالمُضارِعِ المجزومِ؛ تقولُ: اغْزُ وارْمِ واسْعَ، واغْزُوَا وارْمِيَا واسْعَيَا، واغْزُوْا وارْمُوْا واسْعَوْا.
الفِعلُ اللَّفيفُ:
إن كان مقرونًا: فحُكمُه كالنَّاقِصِ ولا فَرْقَ، مِثْلُ: طَوى يطْوِي اطْوِ ...
إن كان مَفروقًا فحُكمُ فائِه كالمِثالِ، وحُكمُ لامِه كالنَّاقِصِ، مِثْلُ: وقى يقِي قِهْ ... [1010] يُنظر: ((شذ العرف في فن الصرف)) للحملاوي (ص: 103 - 106). .
تطبيقاتٌ على إسنادِ الأفعالِ للضَّمائِرِ:
ملحوظةٌ: التاءُ والـ(نا) يختصَّانِ بالدُّخولِ على الماضي، وياءُ المُؤَنَّثةِ المُخاطَبةِ يَشتَرِكُ فيها الأمرُ والمُضارِعُ، ونونُ النِّسْوةِ والضَّمائِرُ السَّاكِنةُ تتَّصِلُ بالأنواعِ الثَّلاثةِ [1011] يُنظر: ((دروس التصريف)) لمحمد محيي الدين عبد الحميد (ص: 139). .
الفِعلُ الصَّحيحُ السَّالمُ:
أولًا: الماضي (نصر):
ففي المُتكَلِّمِ:
مع تاءِ الفاعِلِ: نَصَرْتُ، ومع (نا) الدَّالَّةِ على جماعةِ المُتكَلِّمينَ: نَصَرْنا.
وقد سُكِّنَ آخِرُ الفِعلِ مع هذينِ الضَّميرَينِ؛ حتى لا يتوالى أربَعةُ مُتحَرِّكاتٍ.
المخاطَبُ:
للمُفْرَدِ: نصَرْتَ (للمُذَكَّرِ) ونصَرْتِ (للمُؤَنَّثِ).
للمُثَنَّى: نصَرْتُما (للمُذَكَّرِ والمُؤَنَّثِ).
للجَمعِ: نصَرْتُم (للذُّكورِ) نصرتُنَّ (للإناثِ، والنُّونُ حَرفٌ لجمعِ الإناثِ).
وسُكِّن الفِعلُ في هذه الصُّورِ؛ حتى لا تتوالى أَربَعةُ متحَرِّكاتٍ.
مع الغائِبِ:
مع المُفْرَدِ الغائِبِ: نَصَر (للمُذَكَّر)، ونصَرَتْ (للغائبةِ المُؤَنَّثةِ).
مع المُثَنَّى الغائِبِ: نَصَرَا (للمُذَكَّر)، ونصَرَتا (للمُؤَنَّثِ).
مع الجَمعِ الغائِبِ: نَصَرُوا (المُذَكَّر)، ونصَرْنَ (المُؤَنَّث).
ثانيًا/ الفِعلُ المُضارِعُ:
مع المُتكَلِّمِ:
أنْصُرُ (المُفْرَد)، ونَنْصُرُ (الجمع).
مع المخاطَبِ:
المُفْرَدُ: تَنصُر (المُذَكَّر)، وتَنْصُرِينَ (المُؤَنَّث).
المُثَنَّى: تَنْصُرانِ (للمُذَكَّرِ والمُؤَنَّثِ).
الجَمعُ: تَنصُرونَ (للمُذَكَّر)، وتَنْصُرْنَ (للمُؤَنَّثِ).
مع الغائِبِ:
المُفْرَدُ: يَنْصُرُ (للمُذَكَّر) وتَنصُرُ (للمُؤَنَّثِ).
المُثَنَّى: يَنْصُرانِ (للمُذَكَّر)، وتَنْصُرانِ (للمُؤَنَّثِ).
الجَمعُ: يَنْصُرونَ (للجَمعِ)، ويَنْصُرْنَ (للإناثِ).
ثالثًا: فِعلُ الأمرِ:
مع المخاطَبِ:
المُفْرَد: انصُرْ (للمُذَكَّر)، وانصُرِي (للمُؤَنَّثِ).
المُثَنَّى: انصُرَا (للمُذَكَّرِ والمُؤَنَّثِ).
مع الجَمعِ: انصُروا (للذُّكورِ)، وانصُرْنَ (للإناثِ) [1012] يُنظر: ((المستقصى في علم التصريف)) لعبد اللطيف محمد الخطيب (ص: 222 – 227). .
الفِعْلُ المهموزُ
الماضي (أمر/ سأل/ قرأ):
المُتكَلِّمُ:
المُفْرَدُ: أمَرْتُ، وسأَلْتُ، وقرَأْتُ، والجَمعُ: أَمَرْنا، وسأَلْنا، وقرَأْنا.
المُخاطَبُ:
المُفْرَدُ: أمَرْتَ، وسأَلْتَ، وقرَأْتَ، (المُذَكَّر)، وأمَرْتِ وسأَلْتِ وقَرَأْتِ (المُؤَنَّث).
المُثَنَّى: أمَرْتُما، وسأَلْتُما، وقرَأْتُما (للمُذَكَّرِ والمُؤَنَّثِ).
الغائِبُ:
للمُفْرَدِ: أمَرَ، وسأل، وقرأ (للمُذَكَّر)، وأمَرَتْ وسأَلَتْ وقرَأَتْ (للمُؤَنَّثِ).
المُثَنَّى: أمَرَا، وسأَلَا، وقرَأَا (للمُذَكَّر)، وأمَرَتَا، وسأَلَتَا، وقرَأَتَا (للمُؤَنَّثِ).
الجَمعُ: أَمَرُوا، وسأَلُوا وقرَؤُوا (للذُّكورِ)، وأمَرْنَ وسأَلْنَ، وقرَأْنَ (للإناثِ).
الفِعلُ المُضارِعُ:
المُتكَلِّمُ:
للمُفْرَدِ: أنا آمُرُ (بإدغامِ الهَمزَتَينِ) وهو في الأصلِ: أَأْمُرُ، هَمْزةُ المُضارَعةِ وهَمْزةُ الفِعلِ الأصْليَّةُ، فأبدِلَ منهما مَدٌّ، ووَزْنُه (أَفْعُلُ)، وأنا أسأَلُ، وأنا أقرَأُ، ونحن نأمُرُ، ونسْأَلُ، ونَقْرَأُ.
المُخاطَبُ:
المُفْرَد: تأمُرُ وتسْأَلُ، وتقرَأُ (للمُذَكَّر)، وتأمُرِينَ وتَسْأَلينَ وتَقْرَئِينَ (للمُؤَنَّثِ).
المُثَنَّى: تأمُرَانِ، وتَسألانِ، وتَقْرَأانِ (للمُذَكَّرِ والمُؤَنَّثِ).
الجَمعُ: تَأْمُرونَ، وتَسْأَلونَ وتَقْرَؤون (للذُّكورِ)، وتأمُرْنَ، وتسْأَلْنَ، وتَقْرَأْنَ (للمُؤَنَّثِ).
الغائِبُ:
للمُفْرَدِ: يأمُرُ ويَسأَلُ ويَقرَأُ (للمُذَكَّرِ)، وتأمُرُ وتَسْأَلُ وتَقْرَأُ (للمُؤَنَّثِ).
للمُثَنَّى: يأمُرانِ ويَسألانِ ويَقْرأانِ (للمُذَكَّر)، وتأمُرانِ وتَسألانِ وتَقْرأان (للمُؤَنَّثِ).
الجَمعُ: يأمُرونَ ويَسأَلُونَ ويَقْرَؤونَ (للذُّكورِ)، ويأمُرْنَ ويَسْأَلْنَ ويَقْرَأْنَ (للإناثِ).
ثالثًا: الأمْرُ:
من مَهموزِ الفاءِ:
للمُفْرَدِ: مُرْ (للمُذَكَّر)، ومُرِي (للمُؤَنَّثِ).
المُثَنَّى: مُرَا (للمُذَكَّرِ والمُؤَنَّثِ).
الجَمعُ: مُرُوا (للذُّكورِ)، ومُرْنَ (للإناثِ).
وهنا نلاحِظُ حَذْفَ فاءِ الفِعْلِ (الهَمْزة)، والأصلُ: يأمُرُ، وأَمْرُه: اأْمُرْ، باجتِلابِ هَمْزةِ الوَصْلِ لتعَذُّرِ النُّطقِ بالسَّاكِنِ (الهَمْزة)، ثم قُلِبَت الهَمْزةُ واوًا لانضِمامِ هَمْزةِ الوَصْلِ، فيُصبِحُ: (اؤْمُرْ)، ثمَّ حُذِفَت الهَمْزَتانِ للثِّقَلِ، ومِثْلُه: أخَذَ وأَكَل في الأمرِ منهما، وحَذْفُ الهَمْزَتينِ -كما سبق- في الابتِداءِ مُلتَزَمٌ به، أمَّا إذا تقدَّم الفِعْلَ شَيءٌ، فيُعيدون الهَمْزةَ في الأكثَرِ؛ قال تعالى: وَأْمُرْ أَهْلَكَ بِالصَّلَاةِ [طه: 132] ، فأعادوا الهَمْزةَ؛ لأنَّ الفِعْلَ سُبِقَ بالواوِ.
من مَهموزِ العَينِ:
المُفْرَدُ: اسأَلْ (للمُذَكَّر)، واسأَلِي (للمُؤَنَّثِ).
للمُثَنَّى: اسأَلَا (للمُذَكَّرِ والمُؤَنَّثِ).
للجَمعِ: اسأَلُوا (للذُّكورِ)، اسأَلْنَ (للإناثِ).
ويجوزُ فيه حَذفُ الهَمْزةِ، تقولُ: سَلْ، وسَلِي، وسَلَا، وسَلُوا، وسَلْنَ، بحَذفِ الهَمزَتَينِ التِزامًا في بَدءِ الكلامِ، والأكثَرُ إثباتُ الهَمْزَتينِ إذا سُبِقَ الفِعلُ بشَيءٍ (واوٍ أو فاءٍ)، وكيفيَّةُ الحَذفِ تأتي كالآتي:
أصلُ الفِعْلِ هو (اسْأَلْ)، نُقِلَت حَرَكةُ الهَمْزةِ إلى السِّينِ السَّاكنةِ قَبْلَها ثم حُذِفَت الهَمْزةُ (اِسَلْ)، ولَمَّا تحَّركَت السِّينُ لم يُحتَجْ لهَمْزةِ الوَصْلِ فحُذِفَت أَيضًا (سَلْ، على وَزْنِ فَلْ).
من مَهموزِ اللَّامِ:
المُفْرَدُ: اقرَأْ (للمُذَكَّر)، واقرَئِي (للمُؤَنَّثِ).
المُثَنَّى: اقرَأَا (للمُذَكَّرِ والمُؤَنَّثِ).
الجَمعُ: اقرَؤُوا (للذُّكورِ)، واقرأْنَ (للإناثِ).
ولا صُورةَ فيه غيرُ هذه الصُّورةِ خِلافًا لمهموزِ الفاءِ والعَينِ [1013] يُنظر: ((المستقصى في علم التصريف)) لعبد اللطيف محمد الخطيب (ص: 229 – 238). .
الفِعْلُ المُضاعَفُ
أولًا: الماضي:
المُتكَلِّمُ (شَدَّ):
شَدَدْتُ (للمُفْرَدِ)، وشَدَدْنا (للجَمعِ)، والأصلُ فيه (شَدْدَ) ثُمَّ لَمَّا اتَّصَل بتاءِ الضَّميرِ والـ(نا)، سُكِّنَت الدَّالُ الثَّانيةُ، فكان لا بُدَّ مِن فَتحِ الدَّالِ الأُولى حتى لا يلتَقِيَ ساكنانِ، ما اقتَضَى أن نَفُكَّ الإدغامَ.
المُخاطَبُ:
المُفْرَدُ: شَدَدْتَ (للمُذَكَّر)، وشَدَدْتِ (للمُؤَنَّثِ).
المُثَنَّى: شَدَدْتُما (للمُذَكَّرِ والمُؤَنَّثِ).
الجَمعُ: شَدَدْتُم (للذُّكورِ)، وشَدَدْتُنَّ (للإناثِ).
الغائِبُ:
المُفْرَدُ: شَدَّ (للمُذَكَّر)، وشَدَّتْ (للمُؤَنَّثِ)، وَجَب الإدغامُ هنا لتحَرُّكِ المِثْلَينِ، على مِثالِ: نصَرَ، ونصَرَتْ.
المُثَنَّى: شَدَّا (للمُذَكَّر)، وشَدَّتَا (للمُؤَنَّثِ)، وقد فُتِحَت تاءُ التأنيثِ مع المُؤَنَّثِ؛ بسَبَبِ الألِفِ التي بَعْدَها، وهو فَتحٌ عارِضٌ.
الجَمعُ: شَدُّوا (للذُّكورِ)، وشَدَدْنَ (للإناثِ)، وقد أدغَمْنا مع واوِ الجماعِة وهو ضَميرٌ ساكِنٌ؛ لأنَّ المِثْلَينِ متحَرِّكانِ، فلَمَّا سُكِّنَ المِثْلُ الثَّاني لعِلَّةِ الاتِّصالِ بضَميرِ الرَّفعِ المتحَرِّكِ (نونِ النِّسْوةِ)، فُكَّ الإدغامُ.
ثانيًا/ المُضارِعُ:
المُتكَلِّمُ:
أَشُدُّ (للمُفْرَدِ)، ونَشُدُّ (للجَمعِ)؛ قال تعالى: قَالَ سَنَشُدُّ عَضُدَكَ بِأَخِيكَ [القصص: 35] .
المُخاطَبُ:
المُفْرَدُ: تَشُدُّ (للمُذَكَّر)، وتَشُدِّين (للمُؤَنَّثِ)، بَقِيَ الإدغامُ مع الضَّمائِرِ السَّاكِنةِ والضَّميرِ المُستَتِرِ.
المُثَنَّى: تَشُدَّانِ (للمُذَكَّرِ والمُؤَنَّثِ).
الجَمعُ: تَشُدُّون (للمُذَكَّر)، وتَشْدُدْنَ (للمُؤَنَّثِ) فُكَّ الإدغامُ مع نونِ النِّسْوةِ؛ لسُكونِ الدَّالِ قَبْلَها، فتُفتَحُ الدَّالُ الأُولى، ويُفَكُّ الإدغامُ.
الغائِبُ:
المُفْرَدُ: يَشُدُّ (للمُذَكَّر)، وتَشُدُّ (للمُؤَنَّثِ).
للمُثَنَّى: يَشُدَّانِ (للمُذَكَّر)، وتَشُدَّان (للمُؤَنَّثِ).
للجَمعِ: يَشُدُّونَ (للذُّكورِ)، ويَشْدُدْنَ (للإناثِ).
ثالثًا/ فِعْلُ الأمرِ:
ولا يكونُ إلَّا للمُخاطَبِ:
المُفْرَدُ: اُشدُدْ (للمُذَكَّر)، وشُدِّي (للمُؤَنَّثِ)، ويجوزُ في المُذَكَّرِ الإدغامُ أيضًا؛ لأنَّ السَّاكِنَ الثَّانيَ سُكِّنَ لعِلَّةٍ غيرِ عِلَّةِ الاتِّصالِ بالضَّميرِ المتحَرِّكِ (عِلَّةِ بناءِ الأمرِ على السُّكونِ).
للمُثَنَّى: شُدَّا (للذُّكورِ والإناثِ).
للجَمعِ: شُدُّوا (للذُّكورِ)، اُشْدُدْن (للإناثِ).
المُضَعَّفُ الرُّباعيُّ (زَلْزَل) تمامًا كالسَّالمِ:
الماضي:
المُتكَلِّمُ: وَسْوَسْتُ، وَسْوَسْنا.
المُخاطَبُ: وَسْوَسْتَ، وَسْوَسْتِ، وَسْوَسْتُما، وَسْوَسْتُم، وَسْوَسْتُنَّ.
الغائِبُ: وَسْوَس، وَسْوَسَتْ، وَسْوَسَا، وَسْوَسَتَا، وَسْوَسُوا، وَسْوَسْنَ.
المُضارِعُ:
المُتكَلِّمُ: أُوَسْوِسُ، نُوَسْوِسُ.
المُخاطَبُ: تُوَسْوِسُ، وتُوَسْوِسِينَ، تُوَسْوِسانِ، تُوَسْوِسُونَ، تُوَسْوِسْنَ.
الغائِبُ: يُوسْوِسُ، وتُوَسْوِسُ، ويُوَسْوِسانِ، وتُوَسْوِسانِ، ويُوَسْوِسونَ، يُوَسْوِسْنَ.

المُخاطَبُ:
للمُفْرَدِ: وعَدْتَ (للمُذَكَّر)، وَعَدْتِ (للمُؤَنَّثِ).
للمُثَنَّى: وعَدْتُما (للمُذَكَّرِ والمُؤَنَّثِ).
للجَمعِ: وعَدْتُم (للذُّكورِ)، وعَدْتُنَّ (للإناثِ).
الغائِبُ:
للمُفْرَدِ: وَعَد (للمُذَكَّر)، وعَدَتْ (للمُؤَنَّثِ).

المُثَنَّى: تَعِدَانِ (للمُذَكَّرِ والمُؤَنَّثِ).

الغائِبُ:
المُفْرَدُ: يَعِدُ (للمُذَكَّرِ)، وتَعِدُ (للمُؤَنَّثِ).
المُثَنَّى: يَعِدانِ (للمُذَكَّرِ)، تَعِدانِ (للمُؤَنَّثِ).
الجَمعُ: يَعِدُونَ (الذُّكور)، يَعِدْنَ (الإناث).

الفِعْلُ الأجوَفُ
أوَّلًا: الفِعلُ الماضي (قال، وباع):
المُتكَلِّمُ: قُلْتُ، وبِعْتُ (للمُفْرَدِ)، وقُلْنا وبِعْنا (للجَمعِ)، وهنا حُذِفَ حَرفُ العِلَّةِ؛ تخَلُّصًا من التِقاءِ السَّاكِنَينِ، والأصلُ: قالْتُ، وقالْنا، السَّاكِنُ الأوَّلُ حَرفُ العِلَّةِ، والسَّاكِنُ الثَّاني هو اللَّامُ، سُكِّنَت لأجْلِ الضَّميرِ المتحَرِّكِ، ثم ضُمَّت القافُ وكُسِرَت الباءُ؛ إشارةً إلى أصلِ الألِفِ، وهو الواوُ مِنَ القَولِ، والياءُ مِنَ البَيعِ.
المُخاطَبُ:
المُفْرَدُ: قُلْتَ، بِعْتَ (للمُذَكَّر)، وقُلْتِ، بِعْتِ (للمُؤَنَّثِ).
المُثَنَّى: قُلْتُما، بِعْتُما (للمُذَكَّرِ والمُؤَنَّثِ).
الجَمعُ: قُلْتُم، بِعْتُم (للذُّكورِ)، وقُلْتُنَّ، بِعْتُنَّ (للإناثِ).
الغائِبُ:
المُفْرَدُ: قال، باع (للمُذَكَّر)، قالتْ، باعتْ (للمُؤَنَّثِ).
المُثَنَّى: قالا، باعا (للمُذَكَّر)، قالتَا، باعتَا (للمُؤَنَّثِ)، بَقِيَت الألِفُ للاتِّصالِ بضَميرٍ ساكِنٍ.
الجَمعُ: قالُوا، باعُوا (الذُّكور)، وقُلْنَ، بِعْنَ (الإناث)، بَقِيَت الألِفُ مع واوِ الجَماعةِ لسُكونِها، وحُذِفَت مع نونِ النِّسْوةِ لكَونِه ضَميرًا متحَرِّكًا.
ثانيًا: الفِعلُ المُضارِعُ
المُتكَلِّمُ: أقولُ، أَبِيعُ (للمُفْرَدِ)، نقولُ، نَبِيعُ (للجَمعِ).
المُخاطَبُ:
المُفْرَدُ: تَقولُ، تَبِيعُ (للمُذَكَّرِ)، تَقُولِينَ، تَبِيعينَ (للمُؤَنَّثِ).
المُثَنَّى: تَقُولانِ، تَبِيعانِ (للمُذَكَّرِ والمُؤَنَّثِ).
الجَمعُ: تَقُولونَ، تَبِيعونَ (للذُّكورِ)، تَقُلْنَ، تَبِعْنَ (للإناثِ)، وقد بَقِيَت الواوُ مع ياءِ المُخاطَبةِ المُؤَنَّثةِ وألِفِ الاثنَينِ وواوِ الجَماعةِ لِسُكونِها، وحُذِفَت مع نونِ النِّسْوةِ؛ لأنَّها ضميرٌ مُتحَرِّكٌ.
الغائِبُ:
المُفْرَدُ: يَقُولُ، يَبِيعُ (للمُذَكَّرِ)، تقولُ، تَبيعُ (للمُؤَنَّثِ).
المُثَنَّى: يَقولانِ، يَبِيعانِ (للمُذَكَّرِ)، تَقُولانِ، تَبِيعانِ (للمُؤَنَّثِ).
الجَمعُ: يَقُولونَ، يَبِيعون (للمُذَكَّرِ)، يَقُلْنَ، يَبِعْنَ (للإناثِ).
ثالثًا: فِعْلُ الأمرِ
المُخاطَبُ:
المُفْرَدُ: قُلْ، بِعْ (للمُذَكَّرِ)، قُولي، بِيعِي (للمُؤَنَّثِ)، حُذِفَ حَرفُ العِلَّةِ مع أمْرِ المُخاطَبِ المُفْرَدِ؛ لالتِقاءِ السَّاكِنَينِ؛ سُكونِ حَرفِ العِلَّةِ، وسُكونِ الحَرفِ الأخيرِ العارِضِ لعِلَّةِ بِنائِه على السُّكونِ، وضُمَّت القافُ، وكُسِرَت الياءُ إشارةً إلى الأصلِ.
المُثَنَّى: قُولا، بِيعا (للمُذَكَّرِ والمُؤَنَّثِ).
الجَمعُ: قُولوا، بِيعُوا (للذُّكورِ)، قُلْنَ، بِعْنَ (للإناثِ) [1016] يُنظر: ((المستقصى في علم التصريف)) لعبد اللطيف محمد الخطيب (ص: 252 – 256). .
الفِعْلُ النَّاقِصُ
النَّاقِصُ اليائيُّ:
أوَّلًا: الماضي (قَضَى)
المُتكَلِّمُ: قَضَيْتُ (للمُفْرَدِ)، قَضَينا (للجَمعِ)، اتَّصَل الفِعلُ بضمائِرَ مُتحَرِّكةٍ، فبَقِيَت الياءُ كما هي.
المُخاطَبُ:
المُفْرَدُ: قَضَيْتَ (للمُذَكَّرِ)، قَضَيتِ (للمُؤَنَّثِ).
المُثَنَّى: قَضَيتُما (للمُذَكَّرِ والمُؤَنَّثِ).
الجَمعُ: قَضَيتُم (للذُّكورِ)، قَضَيتُنَّ (للإناثِ).
الغائِبُ:
المُفْرَدُ: قضى (للمُذَكَّرِ)، قَضَتْ (للمُؤَنَّثِ)، حُذِفَ حَرفُ العِلَّةِ عند الاتِّصالِ بتاءِ التأنيثِ السَّاكِنةِ؛ لالتِقاءِ السَّاكِنَينِ؛ سُكونِ حَرفِ العِلَّةِ وسُكونِ تاءِ التأنيثِ.
المُثَنَّى: قَضَيَا (للمُذَكَّرِ)، وقَضَتَا (للمُؤَنَّثِ)، حُذِفَت اللَّامُ لالتِقاءِ السَّاكِنَينِ، وفُتِحَت التاءُ مراعاةً للألِفِ.
الجَمعُ: قَضَوا (للذُّكورِ)، قَضَيْنَ (للإناثِ)، حُذِفَ حَرفُ العِلَّةِ، وهو الألِفُ؛ لالتِقاءِ السَّاكِنَينِ، وفُتِحَ ما قَبْلَ واوِ الجَماعةِ دَليلًا على الألِفِ المحذوفةِ.
ثانيًا: المُضارِعُ
المُتكَلِّمُ: أقضي (المُفْرَد)، نَقْضي (الجَمْع).
المُخاطَبُ:
المُفْرَدُ: تَقْضي (للمُذَكَّرِ)، تَقْضِين (للمُؤَنَّثِ)، حُذِفَت اللَّامُ (الياء) من الفِعلِ مع ياءِ المُخاطَبةِ المُؤَنَّثةِ؛ لالتقاءِ السَّاكِنَينِ، وكُسِرَ ما قَبْلَها، ووَزْنُها: (تَفْعِين).
المُثَنَّى: تَقْضِيان (للمُذَكَّرِ والمُؤَنَّثِ).
الجَمعُ: تَقْضُون (للذُّكورِ)، تَقْضِين (للإناثِ)، وقد حُذِفَت لامُ الفِعْلِ (الياءُ) مع واوِ الجماعةِ لالتِقاءِ السَّاكِنَينِ، ويُفَرَّقُ بين صيغةِ الفِعْلِ مع ياءِ المُخاطَبةِ المُؤَنَّثةِ ونُونِ النِّسْوةِ في الوَزْنِ، فوَزْنُ الفِعْلِ مع ياءِ المُخاطَبةِ المُؤَنَّثةِ: تَفْعِينَ، لحَذفِ اللَّامِ (الياء)، ووَزْنُ الفِعْلِ مع نونِ النِّسْوةِ: تَفْعِلْنَ؛ إذ لا تُحذَفُ منه اللَّامُ.
الغائِبُ:
المُفْرَدُ: يَقْضي (للمُذَكَّرِ)، تقضي (للمُؤَنَّثِ).
المُثَنَّى: يَقْضِيانِ (للمُذَكَّرِ)، تَقْضيانِ (للمُؤَنَّثِ).
الجَمعُ: يَقْضُون (للذُّكورِ)، يَقْضِين (للإناثِ).
ثالثًا: فِعْلُ الأمرِ
المُفْرَدُ: اقْضِ (للمُذَكَّرِ/ افعِ)، اقضِي (للمُؤَنَّثِ/ افعِي)، حُذِفَت اللَّامُ مع المُفْرَدِ المُذَكَّرِ؛ لبناءِ الفِعْلِ على حَذْفِ حَرفِ العِلَّةِ، وحُذِفَت مع ياءِ المُخاطَبةِ المُؤَنَّثة؛ لالتِقاءِ السَّاكِنَينِ.
المُثَنَّى: اقضِيَا (للمُذَكَّرِ والمُؤَنَّثِ).
الجَمعُ: اقضُوا (للذُّكورِ)، اقضِينَ (للإناثِ).
الفِعْلُ النَّاقِصُ الواوِيُّ (دَعَا):
أولًا: الفِعْلُ الماضي
المُتكَلِّمُ: دَعَوْتُ (للمُفْرَدِ)، دَعَوْنا (للجَمعِ).
المُخاطَبُ:
المُفْرَدُ: دَعَوْتَ (للمُذَكَّرِ)، دَعَوْتِ (للمُؤَنَّثِ).
المُثَنَّى: دَعَوتُما (للمُذَكَّرِ والمُؤَنَّثِ).
الجَمعُ: دَعَوْتُم (للذُّكورِ)، دَعَوْتُنَّ (للمُؤَنَّثِ).
الغائِبُ:
المُفْرَدُ: دَعَا (للمُذَكَّرِ)، دعَتْ (للمُؤَنَّثِ)؛ حُذِفَ حَرفُ العِلَّةِ مع تاءِ التأنيثِ؛ لالتِقاءِ السَّاكِنَينِ.
المُثَنَّى: دعَوَا (للمُذَكَّرِ)، دَعَتَا (للمُؤَنَّثِ).
الجَمعُ: دعَوْا (للذُّكورِ/ فَعَوْا)، دَعَونَ (للإناثِ)، حُذِفَ حَرفُ العِلَّةِ مع واوِ الجَماعةِ لالتِقاءِ السَّاكِنَينِ، وبَقِيَت الفَتحةُ دليلًا على المحذوفِ.
ثانيًا: الفِعْلُ المُضارِعُ
المُتكَلِّمُ: أدعو (للمُفْرَدِ)، ندعو (للجَمعِ).
المُخاطَبُ:
المُفْرَدُ: تَدْعو (للمُذَكَّرِ)، تَدْعِينَ (للمُؤَنَّثِ/ تَفْعِينَ)، حُذِفَ حَرفُ العِلَّةِ مع ياءِ المُخاطَبةِ المُؤَنَّثةِ، وكُسِرَ ما قَبْلَ الياءِ للمُناسَبةِ.
المُثَنَّى: تَدْعُوانِ (للمُذَكَّرِ والمُؤَنَّثِ).
الجَمعُ: تَدْعُونَ (للذُّكورِ/ تَفْعُون)، تَدْعُونَ (للإناثِ/ تَفْعُلْنَ)، والفَرْقُ بينهما أنَّه مع واوِ الجَماعةِ حُذِفَت الواوُ لالتِقاءِ السَّاكِنَينِ، واوِ الفِعْلِ وواوِ الجَماعةِ، أمَّا مع نونِ النِّسْوةِ فلم يُحذَفْ شَيءٌ مِنَ الفِعْلِ.
الغائِبُ:
المُفْرَدُ: يدعو (للمُذَكَّرِ)، تَدْعو (للمُؤَنَّثِ).
المُثَنَّى: يَدْعُوانِ (للمُذَكَّرِ)، تَدْعُوانِ (للمُؤَنَّثِ).
الجَمعُ: يَدْعُونَ (للذُّكورِ/ يَفْعُونَ)، يَدْعُونَ (للإناثِ/ يَفْعُلْنَ).
ثالثًا: فِعْلُ الأمرِ
المُخاطَبُ:
المُفْرَدُ: ادْعُ (للمُذَكَّرِ)، ادْعِي (للمُؤَنَّثِ).
المُثَنَّى: ادْعُوَا (للمُذَكَّرِ والمُؤَنَّثِ).
الجَمعُ: ادْعُوا (للذُّكورِ/ افْعُوا)، ادْعُونَ (للإناثِ/ افْعُلْنَ) [1017]يُنظر: ((المستقصى في علم التصريف)) لعبد اللطيف محمد الخطيب (ص: 257 – 262). .
اللَّفِيفُ المَفروقُ
أوَّلًا: الفِعْلُ الماضي (وَقَى)
المُتكَلِّمُ: وَقَيتُ (للمُفْرَدِ)، وَقَيْنا (للجَمعِ).
المُخاطَبُ:
المُفْرَدُ: وَقَيتَ (للمُذَكَّرِ)، وَقَيتِ (للمُؤَنَّثِ).
المُثَنَّى: وَقَيتُما (للمُذَكَّرِ والمُؤَنَّثِ).
الجَمعُ: وَقَيتُم (للذُّكورِ)، وَقَيْتُنَّ (للإناثِ).
الغائِبُ:
المُفْرَدُ: وَقَى (للمُذَكَّرِ)، وَقَتْ (للمُؤَنَّثِ)، حُذِفَت اللَّامُ مع تاءِ التأنيثِ لالتِقاءِ السَّاكِنَينِ، ووَزْنُه "فَعَتْ".
المُثَنَّى: وَقَيَا (للمُذَكَّرِ)، وَقَتَا (للمُؤَنَّثِ).
الجَمعُ: وَقَوا (للذُّكورِ)، وَقَيْنَ (للإناثِ)، حُذِفَت اللَّامُ مع واوِ الجَماعةِ لالتِقاءِ السَّاكِنَينِ، ووَزْنُه (فَعَوْا).
ثانيًا: المُضارِع (يقِي)
المُتكَلِّمُ: أَقِي (للمُفْرَدِ)، نَقِي (للجَمعِ)، حُذِفَت الواوُ من أَوَّلِهما كما في الفِعْلِ المثالِ، ووَزْنُه: (أَعِلُ، نَعِلُ).
المُخاطَبُ:
المُفْرَدُ: تَقِي (للمُذَكَّرِ)، تَقِينَ (للمُؤَنَّثِ)، حُذِفَت اللَّامُ مِنَ الفِعْلِ مع ياءِ المُخاطَبةِ المُؤَنَّثةِ، وكُسِرَ ما قَبْلَ الياءِ للمُناسَبةِ، والوَزْنُ: "تَعِينَ".
المُثَنَّى: تَقِيانِ (للمُذَكَّرِ والمُؤَنَّثِ).
الجَمعُ: تَقُونَ (الذُّكور)، تَقِينَ (الإناث)، حُذِفَت اللَّامُ مع واوِ الجَماعةِ ووَزْنُها "تَعُونَ"، ولم تُحذَفْ مع الضَّميرِ المتحَرِّكِ (نونِ النِّسْوةِ/ تَعِلْنَ).
الغائِبُ:
المُفْرَدُ: يَقِي (للمُذَكَّرِ)، تَقِي (للمُؤَنَّثِ).
المُثَنَّى: يَقِيانِ (للمُذَكَّرِ)، تَقِيانِ (للمُؤَنَّثِ).
الجَمعُ: يَقُونَ (للذُّكورِ/ يَعُونَ)، يَقِينَ (للإناثِ/ يَعِلْنَ).
ثالثًا: فِعْلُ الأمرِ
المُفْرَدُ: قِهْ، أصلُه: قِ، حُذِفَت اللَّامُ للبِناءِ، والفاءُ كالفِعْلِ المثالِ، فوَجَب أن تَلحَقَه هاءُ السَّكْتِ، ووَزْنُه "عِهْ"، قِي (للمُؤَنَّثِ)، حُذِفَ اللَّامُ للاتِّصالِ بياءِ المُخاطَبةِ المُؤَنَّثةِ، وحُذِفَت فاؤُه كالفِعْلِ المِثالِ، ووَزْنُه "عِيْ".
المُثَنَّى: قِيَا (للمُذَكَّرِ والمُؤَنَّثِ/ عِيَا).
الجَمعُ: قُوا (للذُّكورِ/ عُوا)، قِينَ (للإناثِ/ عِلْنَ)، فلا تُحذَفُ لامُه مع نونِ النِّسْوةِ [1018] يُنظر: ((المستقصى في علم التصريف)) لعبد اللطيف محمد الخطيب (ص: 267 – 270). .

المُثَنَّى: يَطْوِيان (للمُذَكَّرِ)، تَطْوِيانِ (للمُؤَنَّثِ).
الجَمعُ: يَطْوُون (للذُّكورِ)، يَطْوِينَ (للإناثِ).
ثالثًا: فِعْلُ الأمرِ
المُخاطَبُ:
المُفْرَدُ: اطْوِ (للمُذَكَّرِ)، واطَوْيِ (للمُؤَنَّثِ).
المُثَنَّى: اطْوِيَا (للمُذَكَّرِ والمُؤَنَّثِ).
الجَمْعُ: اطْوُوا (للذُّكورِ/ افْعُوا)، اطْوِينَ (للإناثِ) [1019] يُنظر: ((المستقصى في علم التصريف)) لعبد اللطيف محمد الخطيب (ص: 263 – 266). .
