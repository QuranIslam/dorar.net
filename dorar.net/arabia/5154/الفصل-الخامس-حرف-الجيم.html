الفَصْلُ الخامِسُ: حَرْفُ الجيمِ

محتويات الصفحة
انظر أيضا

جاثَلِيق:
الجاثَلِيقُ: رَئيسُ النَّصارى في بِلادِ الإسْلامِ بمَدينةِ السَّلامِ، ويَعمَلُ تحتَ يَدِ بِطْرِيقِ أنْطاكِيَةَ [1566] يُنظر: ((القاموس المحيط)) للفيروزابادي (ص: 871). .
لم تُشِرِ المَعاجِمُ القَديمةُ إلى كَوْنِه مُعرَّبًا.
واللَّفْظُ مُعرَّبٌ مِن اليونانيَّةِ: (katholicos)، ومَعْناه: العامُّ، ويُرادُ به في اصْطِلاحِ الكَنيسةِ مُتقدِّمُ الأساقِفةِ [1567] يُنظر: ((قصد السبيل)) لمحمد الأمين المحبي (1/ 362)، ((تفسير الألْفاظ الدَّخيلة)) لطوبيا العنيسي (ص: 19). .
جادِيٌّ:
الجادِيُّ: الزَّعْفَرانُ [1568] يُنظر: ((الصحاح)) للجوهري (2/ 462). ، ويُطلَقُ على الخَمْرِ أيضًا [1569] يُنظر: ((القاموس المحيط)) للفيروزابادي (ص: 1269). ؛ وبمَعْنى الزَّعْفَرانِ جاءَ قَوْلُ حسَّانَ بنِ ثابِتٍ:
وإن جِئْتَهم ألْفَيْتَ حَوْلَ بُيوتِهم
مِن المِسْكِ والجادِي فَتيتًا مُبدَّدا
لم يُشِرْ أيٌّ مِن اللُّغويِّينَ القُدَماءِ إلى كَوْنِه مُعرَّبًا، سِوى الجَواليقيِّ [1570] يُنظر: ((المعرَّب)) للجواليقي (ص: 250). ، وفي التَّهْذيبِ نَسَبَه إلى قَرْيةٍ تُسمَّى (جادية) بالشَّامِ [1571] يُنظر: ((التهذيب)) للأزهري (11/ 109). ، وهو مِن الفارِسيَّةِ: (جادي) [1572] يُنظر: ((قصد السبيل)) لمحمد الأمين المحبي (1/ 362)، ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 39). .
جامٌ:
الجامُ: إناءٌ مِن فِضَّةٍ، والجَمْعُ: أجْؤُمٌ، وأجْوامٌ، وجاماتٌ، وجُومٌ.
وقدْ ذَكَرَ ابنُ مَنْظورٍ أنَّ اللَّفْظَ عَربيٌّ صَحيحٌ [1573] يُنظر: ((لسان العَرب)) لابن منظور (12/ 112). ، وعلى القَوْلِ بأنَّه مُعرَّبٌ، فهو مِن الفارِسيَّةِ بلَفْظِه: (جام)، وهو الكُوبُ [1574] يُنظر: ((قصد السبيل)) لمحمد الأمين المحبي (1/ 365)، ((تفسير الألْفاظ الدَّخيلة)) لطوبيا العنيسي (ص: 21). .
جاموسٌ:
الجاموسُ: حَيَوانٌ أهْليٌّ مِن الفَصيلةِ البَقَريَّةِ، يُربَّى للحَرْثِ ودَرِّ اللَّبَنِ، ويُجمَعُ على: جَواميسَ [1575] يُنظر: ((المُعجَم الوسيط)) الصادر عن مجمع اللُّغة العَربيَّة بالقاهرة (1/ 134). .
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (گاوميش) [1576] يُنظر: ((تاج العروس)) للزبيدي (15/ 513). ، مُكوَّنٌ مِن (گاو)، أي: بَقَرةٍ، و(ميشي)، أي: نَعْجةٍ، ورأيُ ف. عبْدِ الرَّحيمِ أنَّه عُرِّبَ أوَّلًا: (جَواميس)، ثُمَّ لمَّا وافَقَ بِناءً مِن أبْنِيةِ الجَمْعِ قالوا: جاموسٌ، وأيَّدَ رأيَه بأنَّ ابنَ مَنْظورٍ ذَكَرَ أنَّ أصْلَه الفارِسيَّ (كواميش) [1577] يُنظر: ((لسان العَرب)) لابن منظور (6/ 43). [1578] يُنظر: ((المعرَّب)) للجواليقي (ص: 245). ، والفَيُّوميُّ ذَهَبَ إلى أنَّ اللَّفْظَ عَربيٌّ مُشتَقٌّ مِن (جمس)، أي: قَعَدَ وجَمَدَ؛ لأنَّه ليس فيه مِن لينِ البَقَرِ في الحَرْثِ والزَّرْعِ [1579] يُنظر: ((المصباح المنير)) للفيومي (1/ 108). ، وليس بصَحيحٍ.
جِبْسٌ عامِّيٌّ:
الجِبْسُ: هو الجَبانُ العَيِيُّ، أو الضَّعيفُ اللَّئيمُ، وهو الجامِدُ مِن كلِّ شيءٍ، وهو الَّذي يُبْنى به، وتُلاطُ به البُيوتُ، وهو الجِصُّ، ويُقالُ فيه: قَصٌّ، والجَمْعُ: أجْباسٌ وجُبوسٌ [1580] يُنظر: ((لسان العَرب)) لابن منظور (6/ 34)، ((تاج العروس)) للزبيدي (15/ 490). ، وفي تَصحيحِ التَّصْحيفِ أنَّ الصَّوابَ فيه: جِصٌّ [1581] يُنظر: ((تصحيح التصحيف)) لصلاح الدين الصفدي (1/ 7). ، وقدْ دَخَلَ اللَّفْظُ إلى العامِّيَّةِ.
لم تُشِرِ المَعاجِمُ القَديمةُ إلى كَوْنِه مُعرَّبًا، وأوْرَدَه الشِّهابُ الخَفاجيُّ في "شِفاءِ الغَليلِ" [1582] يُنظر: ((شفاء الغليل)) للشهاب الخفاجي (ص: 90). ، وهو مُعرَّبٌ مِن الفارِسيَّةِ: (كچ)، والأقْرَبُ أن يكونَ مِن اليونانيَّةِ: (gypsum) [1583] يُنظر: ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 146). .
جُدَّاد:
الجُدَّادُ: الثِّيابُ البالِيةُ، والمُعقَّدُ مِن الخُيوطِ وأغْصانِ الشَّجَرِ؛ قالَ الأعْشى:
أضاءَ مِظلَّتَه بالسِّرا
جِ واللَّيلُ غامِرُ جُدَّادِها
واللَّفْظُ مُعرَّبٌ مِن الفارِسيَّةِ، كما ذَكَرَ الجَوْهَريُّ [1584] يُنظر: ((الصحاح)) للجوهري (2/ 453). والجَواليقيُّ [1585] يُنظر: ((المعرَّب)) للجواليقي (ص: 233). : (كدادْ)، أو مِن النَّبَطيَّةِ كما ذَكَرَ ابنُ دُرَيْدٍ: (كدادي) [1586] يُنظر: ((الجمهرة)) لابن دُرَيْدٍ (3/ 1426). .
جُدَّة:
جُدَّةٌ: الشَّاطِئُ، وفيه لُغاتٌ: جِدَّةٌ، وجَدٌّ، وجُدٌّ، ومِنه سُمِّيَتْ مَدينةُ جُدَّةَ في المَمْلَكةِ العَربيَّةِ السُّعوديَّةِ؛ لأنَّها على ساحِلِ البَحْرِ بحِذاءِ مَكَّةَ.
وهو مِن النَّبَطيَّةِ: كُدٌّ، بمَعْنى الشَّاطِئِ [1587] يُنظر: ((المعرَّب)) للجواليقي (ص: 251)، ((لسان العَرب)) لابن منظور (3/ 108). .
جُرُبَّان:
الجُرُبَّانُ والجِرِبَّانُ: قِرابُ السَّيْفِ الضَّخْمُ الَّذي تكونُ فيه أداةُ الرَّجُلِ وسَطْوُه وما يَحْتاجُ إليه، وغِمْدُ السيَّفِ، وحَدُّه، وحَمائِلُه، وجَيْبُ القَميصِ، والدِّرْعُ، وفي الحَديثِ: ((أتَيْتُ النَّبيَّ صلَّى اللهُ عليه وسلَّمَ فأدْخَلْتُ يَدي في جُرُبَّانِه)) [1588] أخرجه أحمدُ (15582)، والنَّسائيُّ في ((السنن الكبرى)) (8307)، والطبراني (19/25) (50) باختلافٍ يسيرٍ من حَديثِ قُرَّةَ بنِ إياسٍ رَضِيَ اللهُ عنه. ، وقالَ جَريرٌ:
إذا قيلَ هذا البَيْنُ راجَعْتُ عَبْرةً
لها بجُرُبَّانِ البَنيقةِ واكِفُ
وهو مِن الفارِسيَّةِ: (گربيان) [1589] يُنظر: ((لسان العَرب)) لابن منظور (1/ 261). .
جِرِبَّانةٌ:
الجِرِبَّابةُ: المَرْأةُ الصَّخَّابةُ البَذيئةُ الخُلُقِ، وفيها لُغةٌ: جِلِبَّانةٌ [1590] يُنظر: ((تاج العروس)) للزبيدي (2/ 151). .
لم تُشِرِ المَعاجِمُ القَديمةُ إلى تَعْريبِها، وذَكَرَ أَدِّي شِير أنَّها مُعرَّبةٌ مِن الفارِسيَّةِ: گران بُن، وأصْلُ مَعْناها: الثَّقيلةُ الأصْلِ [1591] يُنظر: ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 39). .
جُرْبُز:
الجُرْبُزُ: المُخادِعُ الخَبيثُ مِن الرِّجالِ، واشْتَقُّوا مِنه فقالوا: جَرْبَزَ الرَّجُلُ جَرْبَزَةً، وفيه لُغةٌ: القُرْبُزُ.
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (گربز)، بمَعْنى: الذَّكَرِ والشُّجاعِ والماكِرِ [1592] يُنظر: ((المعرَّب)) للجواليقي (ص: 235)، ((لسان العَرب)) لابن منظور (5/ 318)، ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 39). .
جِرْجِسٌ:
الجِرْجِسُ: الشَّمْعُ الَّذي يُختَمُ به، والصَّحيفةُ، وفيه لُغةٌ: القِرْقِسُ [1593] يُنظر: ((تاج العروس)) للزبيدي (15/ 493). .
واللَّفْظُ مُعرَّبٌ مِن الفارِسيَّةِ: (جرجشت)، والفارِسيُّ مَأخوذٌ مِن السُّرْيانيَّةِ [1594] يُنظر: ((الجمهرة)) لابن دُرَيْدٍ (2/ 1162)، ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 39). .
جِرْجيرٌ:
الجِرْجيرُ: نَباتٌ ماؤُه يُزِيلُ آثارَ القُروحِ، وهو يُدِرُّ اللَّبَنَ، ويَهضِمُ الغِذاءَ [1595] يُنظر: ((تاج العروس)) للزبيدي (10/ 405). .
الكَلِمةُ مُعرَّبةٌ مِن السُّرْيانيَّةِ: (garghiro) [1596] يُنظر: ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 150). .
جِرْدابٌ:
الجِرْدابُ: وَسَطُ البَحْرِ [1597] يُنظر: ((التهذيب)) للأزهري (11/ 170). .
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (گِرْدَاب)، ومَعْناه: الدَّوَّامةُ، وهو مُركَّبٌ مِن (گرد)، أي: دائِرٍ، و(آب)، أي: ماءٍ [1598] يُنظر: ((المعرَّب)) للجواليقي (ص: 233)، ((تاج العروس)) للزبيدي (2/ 159)، ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 39). .
جَرْدَبان:
الجَرْدَبانُ: الشَّخْصُ الَّذي يُخْفي شيئًا مِن الطَّعامِ بشِمالِه كيْ لا يَتَناوَلَه غيْرُه، فإذا فَرَغَ القَوْمُ مِن الطَّعامِ عَمَدَ إلى أكْلِه، فهو الَّذي يَأكُلُ بيَمينِه ويَمنَعُ بشِمالِه، وفيه لُغةٌ: (الجُرْدُبانُ)، واشْتَقُّوا مِنه فِعْلًا: جَرْدَبَ الطَّعامَ، وجَرْدَمَ [1599] يُنظر: ((التهذيب)) للأزهري (11/ 170). .
واللَّفْظُ مُعرَّبٌ مِن الفارِسيَّةِ: (گِرْده بان)، ومَعْناه: حافِظُ الرَّغيفِ، وهو مُكوَّنٌ مِن جُزْأينِ: (گِرْده)، أي: الرَّغيفِ، و(بان)، أي: حافِظٍ [1600] يُنظر: ((الصحاح)) للجوهري (1/ 99)، ((المعرَّب)) للجواليقي (ص: 233)، ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 39). .
جَرْدَقٌ:
الجَرْدَقُ والجَرْدَقةُ: الرَّغيفُ، والجَرْذَقُ لُغةٌ فيه، والجَرْدَقُ أجْوَدُ.
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (گِرْده) [1601] يُنظر: ((التهذيب)) للأزهري (9/ 282)، ((لسان العَرب)) لابن منظور (10/ 35). ، وذَكَرَ ف. عبْدُ الرَّحيمِ أنَّ اللَّفْظَ مُعرَّبٌ في الأصْلِ مِن الفَهْلَويَّةِ: (girtak)، وفُتِحَتِ الجيمُ فيه لإلْحاقِه بـ(جَعْفَرٍ) [1602] يُنظر: ((المعرَّب)) للجواليقي (ص: 232). .
جَرَّةٌ:
الجَرَّةُ: إناءٌ مِن خَزَفٍ، والجَمْعُ: جَرٌّ، وجِرارٌ.
لم تُشِرِ المَعاجِمُ القَديمةُ إلى كَوْنِه مُعرَّبًا، وهو مُعرَّبٌ مِن الفارِسيَّةِ: (گَره) [1603] يُنظر: ((قصد السبيل)) لمحمد الأمين المحبي (1/ 380)، ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 39). .
جُرْنال عامِّيٌّ:
الجُرْنالُ: الصَّحيفةُ، والجَريدةُ، والعامَّةُ في مِصْرَ تقولُه: جُرْنان بالنُّونِ، وتَجمَعُه على جَرانينَ.
وهو مِن الفَرَنْسيَّةِ: (journal)، وهو مَأخوذٌ مِن اللَّاتينيَّةِ: (diurnalis) بمَعْنى: يَوميٍّ [1604] يُنظر: ((مُعجَم الدَّخيل في اللُّغة العَربيَّة الحديثة)) للدكتور/ ف. عبد الرحيم (ص: 91). .
جَرْمٌ:
الجَرْمُ: الحَرُّ الَّذي هو نَقيضُ البَرْدِ.
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (گَرم) [1605] يُنظر: ((التهذيب)) للأزهري (11/ 45)، ((الصحاح)) للجوهري (5/ 1885)، ((المعرَّب)) للجواليقي (ص: 235)، ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 40). .
جُرْموقٌ:
الجُرْموقُ: ما يُلبَسُ فوقَ الخُفِّ ليَحفَظَه مِن الطِّينِ.
لم تُشِرِ المَعاجِمُ القَديمةُ إلى تَعْريبِه، وذَكَرَه الشِّهابُ الخَفاجيُّ في "شِفاءِ الغَليلِ"، وهو مُعرَّبٌ مِن الفارِسيَّةِ: (سَرْموزه)، وهو مُركَّبٌ مِن (سَرْ)، أي: فوقَ، و(موزه)، أي: خُفٍّ [1606] يُنظر: ((شفاء الغليل)) للشهاب الخفاجي (ص: 93)، ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 40). .
جِرْيال:
الجِرْيالُ: لَوْنُ الحُمْرةِ، وصِبْغٌ أحْمَرُ، وبه سُمِّيَتِ الخَمْرُ؛ للمُشابَهةِ بيْنَهما في اللَّوْنِ الأحْمَرِ، وفيه لُغةٌ: جِرْيانٌ؛ وبمَعْنى لَوْنِ الخَمْرِ جاءَ قَوْلُ الأعْشى:
وسَبيئةٍ ممَّا تُعتِّقُ بابلٌ
كدَمِ الذَّبيحِ سَلَبْتُها جِرْيالَها
أي: سَلَبْتُها لَوْنَها الأحْمَرَ بأن جُعِلَتْ حُمْرتُها فيَّ.
وبمَعْنى الصِّبْغِ الأحْمَرِ جاءَ قَوْلُ الأعْشى أيضًا:
إذا جُرِّدَتْ يَوْمًا حَسِبْتَ خَميصةً
عليها وجِرْيالَ النَّضيرِ الدُّلامِصا
فشَبَّهَ جَسَدَها بالذَّهَبِ ولَوْنَه بالجِرْيالِ [1607] يُنظر: ((التهذيب)) للأزهري (11/ 22)، ((لسان العَرب)) لابن منظور (11/ 108)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 153). .
ذَكَرَ ابنُ دُرَيْدٍ في الجَمْهَرةِ عن الأصْمَعيِّ أنَّه مُعرَّبٌ مِن الرُّوميَّةِ [1608] يُنظر: ((الجمهرة)) لابن دُرَيْدٍ (2/ 1204). ، وفي اللِّسانِ ذَكَرَ أنَّ أصْلَه الرُّوميَّ: (كِرْيالٌ) [1609] يُنظر: ((لسان العَرب)) لابن منظور (11/ 109). ، وذَكَرَ ف. عبْدُ الرَّحيمِ أنَّه مِن اليونانيَّةِ: (كوراليون)، ومَعْناه: المَرْجانُ، ويُطلَقُ خاصَّةً على المَرْجانِ الأحْمَرِ [1610] يُنظر: ((المعرَّب)) للجواليقي (ص: 244). ، ونَقَلَ أَدِّي شِير اشْتِقاقَها مِن اليونانيَّةِ، ثُمَّ ذَكَرَ أنَّها مُعرَّبةٌ مِن الفارِسيَّةِ: (زريون)، وهي كَلِمةٌ مُركَّبةٌ مِن (زر)، أي: ذَهَبٍ، و(يون)، أي: لَوْنٍ [1611] يُنظر: ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 40). .
جَريبٌ:
الجَريبُ: مِن الأرْضِ: مِقْدارٌ مَعْلومُ المَساحةِ [1612] يُنظر: ((التهذيب)) للأزهري (11/ 37). ، والجَمْعُ: أجْرِبةٌ وجُرْبانٌ.
أشارَ ابنُ دُرَيْدٍ إلى تَعْريبِه، فقالَ: (وأحْسَبُه مُعرَّبًا) [1613] ((الجمهرة)) لابن دُرَيْدٍ (1/ 266). ، وهو مِن الفارِسيَّةِ (گَريب)، أو مِن السُّرْيانيَّةِ: (gyribo) [1614] يُنظر: ((المعرَّب)) للجواليقي (ص: 255)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (153). .
جَزَرٌ:
الجَزَرُ: نَباتٌ يُؤكَلُ، وقد تُكسَرُ الجيمُ: (الجِزَرُ).
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (گَزَر) [1615] يُنظر: ((القاموس المحيط)) للفيروزابادي (ص: 364)، ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 41). .
جَزْمةٌ عامِّيٌّ:
الجَزْمةُ: الحِذاءُ بلُغةِ أهْلِ مِصْرَ، والحِذاءُ طَويلُ السَّاقِ بلُغةِ سوريَّةَ ولُبْنانَ.
وهو مِن التُّرْكيَّةِ: (cizma)، وهو الحِذاءُ الطَّويلُ السَّاقِ [1616] يُنظر: ((مُعجَم الدَّخيل في اللُّغة العَربيَّة الحديثة)) للدكتور/ ف. عبد الرحيم (ص: 91). .
جُغْرافيا:
الجُغْرافيا: عِلمٌ بأحْوالِ الأرْضِ مِن حيثُ تَقْسيمُها إلى الأقاليمِ والجِبالِ والأنْهارِ والسُّكَّانِ.
واللَّفْظُ مِن اليونانيَّةِ: (gègrafo)، مُركَّبٌ مِن (gè)، أي: أرْضٍ، و(grafo)، أي: كُتُبٍ [1617] يُنظر: ((قصد السبيل)) لمحمد الأمين المحبي (1/ 387)، ((تفسير الألْفاظ الدَّخيلة)) لطوبيا العنيسي (ص: 20). .
جُلاهِق:
الجُلاهِقُ: طينٌ مُدوَّرٌ يُرْمى به الطَّيْرُ بواسِطةِ القَوْسِ، وهو لُعْبةٌ يَلعَبُ بها الصِّبْيانُ، واشْتَقُّوا مِنه فِعْلًا: جَهْلَقَ، بتَقْديمِ الهاءِ وتَأخيرِ اللَّامِ، والمُفرَدُ: جُلاهِقةٌ؛ قالَ المُتَنبِّي:
كأنَّما الجِلْدُ لعُريِ النَّاهِقِ
مُنْحدِرٌ عن سيتَيْ جُلاهِقِ [1618] يُنظر: ((المعرَّب)) للجواليقي (ص: 235، 236)، ((قصد السبيل)) لمحمد الأمين المحبي (ص: 390).
واللَّفْظُ مِن الفارِسيَّةِ: (جُلاهَه) كما في الجَمْهَرةِ [1619] يُنظر: ((الجمهرة)) لابن دُرَيْدٍ (2/ 1120). ، أو (جُلَه)، بمَعْنى: كُبَّةِ غَزْلٍ [1620] يُنظر: ((لسان العَرب)) لابن منظور (10/ 37)، ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 43)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 157). .
جُلُستان:
الجُلُسْتانُ: الزَّهْرُ.
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (گُلستان)، مُركَّبٌ مِن كَلِمتَينِ: (گُل) بمَعْنى الوَرْدِ، و(ستان) بمَعْنى: مَحَلٍّ، فيكونُ أصْلُ مَعْناه: مَحَلَّ الوَرْدِ [1621] يُنظر: ((شفاء الغليل)) للخفاجي (ص: 93)، ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 43). .
جِلْفاط:
الجِلْفاطُ: الرَّجُلُ الَّذي يَشُدُّ ألْواحَ السَّفينةِ ويُصلِحُها، فهذا الفِعلُ يُسمَّى الجَلْفَطةَ؛ كَتَبَ عُمَرُ لمُعاوِيةَ لمَّا اسْتَأذَنَه في غَزْوِ البَحْرِ: ((إنِّي لا أحْمِلُ المُسْلِمينَ على أعْوادٍ نَجَّرَها النَّجَّارُ وجَلْفَطَها الجِلْفاطُ)) [1622] أخرجه الخطابي في ((غريب الحديث)) (2/68). [1623] يُنظر: ((المعرَّب)) للجواليقي (ص: 256). ، والجِلنفاطُ لُغةٌ فيه [1624] يُنظر: ((القاموس المحيط)) للفيروزابادي (ص: 662). .
ذَكَرَ ابنُ دُرَيْدٍ أنَّه لُغةٌ شاميَّةٌ [1625] يُنظر: ((الجمهرة)) لابن دُرَيْدٍ (2/ 1201). ، وذَكَرَ ف. عبْدُ الرَّحيمِ أنَّه مِن اللَّاتينيَّةِ: (calfatern) (فريتاك) [1626] يُنظر: ((المعرَّب)) للجواليقي (ص: 256، 257). .
جُلٌّ:
الجُلُّ: الوَرْدُ بشَكْلٍ عامٍّ، أو هو الياسَمينُ، والمُفرَدُ: جُلَّةٌ، وبمَعْنى الوَرْدِ ذَكَرَه الأعْشى؛ قالَ:
وشاهِدُنا الجُلُّ والياسَمينُ
والمُسْمِعاتُ بقُصَّابِها [1627] يُنظر: ((لسان العَرب)) لابن منظور (11/ 121).
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (گُلْ)، بمَعْنى: وَرْدٍ [1628] يُنظر: ((المعرَّب)) للجواليقي (ص: 259)، ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 43). .
جُلَّسان:
الجُلَّسانُ: الوَرْدُ الَّذي يُنثَرُ على مَن يَحضُرونَ العُرْسَ، والوَرْدُ الأبْيَضُ، ونَوْعٌ مِن الرَّيْحانِ؛ قالَ الأعْشى:
لنا جُلَّسانٌ عنْدَها وبَنَفْسَجُ [1629] يُنظر: ((لسان العَرب)) لابن منظور (6/ 40).
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (كُلْشان) كما عنْدَ الجَواليقيِّ [1630] يُنظر: ((المعرَّب)) للجواليقي (ص: 247). والجَوْهَريِّ [1631] يُنظر: ((الصحاح)) للجوهري (3/ 915). ، أو (گُلْشَن) كما في القاموسِ [1632] يُنظر: ((القاموس المحيط)) للفيروزابادي (ص: 536). .
جُلَّنار:
الجُلَّنارُ: زَهْرُ الرُّمَّانِ.
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (گُلْنار)، مُركَّبٌ مِن (گُل) بمَعْنى: الوَرْدِ، و(نار) بمَعْنى: الرُّمَّانِ [1633] يُنظر: ((القاموس المحيط)) للفيروزابادي (ص: 367)، ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 43). .
جِلَّوْزٌ:
الجِلَّوْزُ: البُنْدُقُ.
لم يُشِرْ إلى كَوْنِه مُعرَّبًا مِن القُدَماءِ سِوى الجَواليقيِّ [1634] يُنظر: ((المعرَّب)) للجواليقي (ص: 238). ، بلْ ذَكَرَ ابنُ مَنْظورٍ عن سيبَوَيْهِ أنَّه عَربيٌّ [1635] يُنظر: ((لسان العَرب)) لابن منظور (5/ 322). ، وذَكَرَ الدُّكْتورُ/ ف. عبْدُ الرَّحيمِ أنَّ الصَّوابَ أنَّه فارِسيٌّ: (كَلُّوز)، بمَعْنى: البُنْدُقِ [1636] يُنظر: ((المعرَّب)) للجواليقي (ص: 239). .
جُمْباز:
الجُمْبازُ: نَوْعٌ مِن أنْواعِ الرِّياضاتِ البَدَنيَّةِ.
وهو مِن التُّرْكيَّةِ: (cambaz)، وهو مَن يَأتي بحَرَكاتٍ خَطيرةٍ في السِّيركِ، وهو مِن الفارِسيَّةِ: (جانباز)؛ مُركَّبٌ مِن كَلِمتَينِ؛ هما: (جان)، أي: الرُّوحُ، و(باز)، أي: اللَّاعِبُ، فيكونُ المَعْنى: المُغامِرَ أو اللَّاعِبَ بحَياتِه [1637] يُنظر: ((مُعجَم الدَّخيل في اللُّغة العَربيَّة الحديثة)) للدكتور/ ف. عبد الرحيم (ص: 93). .
جُمْجُم:
الجُمْجُمُ: نَوْعٌ مِن الأحْذيةِ.
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (چمچم)، وهو حِذاءٌ يَلبَسُه فُقَراءُ بِلادِ فارِسَ [1638] يُنظر: ((القاموس المحيط)) للفيروزابادي (ص: 1090)، ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 44). .
جُمَّلٌ:
حِسابُ الجُمَّلِ: ما قُطِّعَ على حُروفِ أبْجَدَ.
ذَكَرَ ابنُ دُرَيْدٍ في الجَمْهَرةِ قَوْلَينِ؛ الأوَّلُ قَوْلُه: (وأمَّا الجُمَّلُ مِن الحِسابِ فلا أحْسَبُه عَربيًّا صَحيحًا) [1639] ((الجمهرة)) لابن دُرَيْدٍ (1/ 491). ، والثَّاني قَوْلُه: (والجُمَّلُ مِن قَوْلِهم: حِسابُ الجُمَّلِ، وأحْسَبُها داخِلةً في العَربيَّةِ) [1640] ((الجمهرة)) لابن دُرَيْدٍ (2/ 1166). ، وذَكَرَ الدُّكْتورُ/ ف. عبْدُ الرَّحيمِ أنَّه مِن السُّرْيانيَّةِ: (كومل)، وهو اسمُ حَرْفِ الجيمِ، والعَربُ إنَّما أخَذَتِ الحُروفَ مِن السُّرْيانيَّةِ [1641] يُنظر: ((المعرَّب)) للجواليقي (ص: 240، 241). .
جِنازةٌ:
الجِنازةُ: المَيِّتُ على السَّريرِ، ولا تُسمَّى الجِنازةُ جِنازةً حتَّى يكونَ عليها مَيِّتٌ، وإلَّا هو سَريرٌ أو نَعْشٌ، وقيلَ بعَكْسِ ذلك [1642] يُنظر: ((الصحاح)) للجوهري (3/ 870)، ((القاموس المحيط)) للفيروزابادي (ص: 506). .
وابنُ دُرَيْدٍ ذَكَرَ أنَّ ناسًا يَزْعُمونَ أنَّ اللَّفْظَ عَربيٌّ مُشتَقٌّ مِن جَنَزَ الشَّيءَ: أي: سَتَرَه، ثُمَّ عَلَّقَ وقالَ: (ولا أدْري ما صِحَّتُه) [1643] ((الجمهرة)) لابن دُرَيْدٍ (1/ 472). ، ونَقَلَ ابنُ سِيدَه القَوْلَ بأنَّه نَبَطيٌّ [1644] يُنظر: ((المحكم)) لابن سيده (7/ 299). .
الجُنْبَثْقةُ:
الجُنْبَثْقةُ: المَرْأةُ السِّيِّئةُ.
واللَّفْظُ مُعرَّبٌ مِن الفارِسيَّةِ [1645] يُنظر: ((تهذيب اللُّغة)) للأزهري (9/ 286)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 162). .
جِنِرالٌ:
الجِنِرالُ: ضابِطٌ ذو رُتْبةٍ عَسْكريَّةٍ عاليةٍ، عَربيَّتُه: اللِّواءُ، ويُجمَعُ على "جِنِرالاتٍ".
وهو مِن الفَرَنْسيَّةِ: ((gènèral، وهو في مُعظَمِ اللُّغاتِ الأوْروبِّيَّةِ، وهو مِن اللَّاتينيَّةِ: genus، بمَعْنى: الجِنْسِ، فأصْلُ مَعْنى الجِنرالِ: عامٌّ، أو ما يَنْطبِقُ على جَميعِ أفْرادِ الجِنْسِ [1646] يُنظر: ((مُعجَم الدَّخيل في اللُّغة العَربيَّة الحديثة)) للدكتور/ ف. عبد الرحيم (ص: 94). .
الجَنْكُ:
الجَنْكُ: آلةٌ مِن آلاتِ الطَّرَبِ [1647] يُنظر: ((تاج العروس)) للزبيدي (27/ 100). .
واللَّفْظُ مُعرَّبٌ مِن الفارِسيَّةِ: (چنك) [1648] يُنظر: ((شفاء الغليل)) للشهاب الخفاجي (ص: 101)، ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 46). .
جِهْبِذٌ:
الجِهْبِذُ: النَّاقِدُ الخَبيرُ بغَوامِضِ الأمورِ، والبارِعُ العارِفُ بطُرُقِ التَّمْييزِ بيْنَ الحَسَنِ والرَّديءِ [1649] يُنظر: ((تاج العروس)) للزبيدي (9/ 392). .
واللَّفْظُ مُعرَّبٌ مِن الفارِسيَّةِ: (گُهبُد)، وذَكَرَ أَدِّي شِير أنَّ اللَّفْظَ الفارِسيَّ مُخَفَّفٌ مِن (گُوه بود)، ومَعْناه: المُقيمُ في الجِبالِ، ويُطلَقُ على النسَّاكِ وعلى النَّاقِدِ والعارِفِ [1650] يُنظر: ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 46). .
جُوالِقٌ:
الجُوالِقُ: نَوْعٌ مِن الأوْعيةِ، وفيه لُغاتٌ ثَلاثٌ: جِوالِقٌ، وجُوالَقٌ، وجُوالِقٌ [1651] يُنظر: ((لسان العَرب)) لابن منظور (10/ 36)، ((القاموس المحيط)) للفيروزابادي (ص: 872). .
واللَّفْظُ مُعرَّبٌ مِن الفارِسيَّةِ: (گُواله) [1652] يُنظر: ((المعرَّب)) للجواليقي (ص: 251، 252). ، وذَكَرَ الزَّبيديُّ أنَّه مُعرَّبٌ مِن (چواله)، وذَكَرَ الدُّكْتورُ/ ف. عبْدُ الرَّحيمِ أنَّ اللَّفْظَ الَّذي ذَكَرَه الزَّبيديُّ إنَّما هو بالتُّرْكيَّةِ: (cuval) (چوال)، والعامَّةُ يَقولونَ له: شِوَالٌ [1653] يُنظر: (المعرَّب)) للجواليقي (ص: 252)، ((مُعجَم الدَّخيل في اللُّغة العَربيَّة الحديثة)) للدكتور/ ف. عبد الرحيم (ص: 135). .
جَوْخان:
الجَوْخانُ: بَيْدَرُ القَمْحِ وغيْرِه، وعَربيَّتُه: الجَرينُ والمِسْطَحُ، والجَمْعُ: جَواخينُ.
وهو مِن الفارِسيَّةِ كما أشارَتِ المَعاجِمُ القَديمةُ [1654] يُنظر: ((لسان العَرب)) لابن منظور (3/ 13). .
جُؤذَرٌ:
الجُؤْذَرُ: وَلَدُ البَقَرةِ مُطلَقًا، أو وَلَدُ البَقَرةِ الوَحْشيَّةِ، والجَمْعُ: جآذِرُ، وفيه لُغاتٌ كَثيرةٌ: جُؤْذُرٌ، وجُؤْذَرٌ، وجوذُرٌ، وجوذَرٌ، وجَوْذَرٌ، وجَيْذَرٌ [1655] يُنظر: ((لسان العَرب)) لابن منظور (4/ 124). ، وجَوْذِرٌ [1656] يُنظر: ((القاموس المحيط)) للفيروزابادي (ص: 363). ؛ قالَ عَدِيُّ بنُ زيدٍ:
تَسرِقُ الطَّرْفَ بعَينَي جُؤْذُرٍ
أحْوَرِ المُقْلةِ مَكْحولِ النِّظارِ
واللَّفْظُ مُعرَّبٌ مِن الفارِسيَّةِ: (گَودَر)، مُركَّبٌ مِن (گَاو)، أي: بَقَرةٍ، و(دَر) للتَّذْكيرِ [1657] يُنظر: ((المعرَّب)) للجواليقي (ص: 246)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 167). .
جُودِياءُ:
الجُودِياءُ والجُوذِياءُ: الكِساءُ؛ قالَ الأعْشى:
وبَيْداءَ تَحْسَبُ آرامَها
رِجالَ إيادٍ بأجْيادِها
ويُقصَدُ بقَوْلِه: (بأجْيادِها): الجُودِياءُ.
واللَّفْظُ مُعرَّبٌ مِن الفارِسيَّةِ: (گَواژه)، ويُطلَقُ على الفُوطةِ، وكلِّ ما تُغَطِّي به النِّساءُ رُؤوسَهنَّ.
أو هو مُعرَّبٌ مِن النَّبَطيَّةِ [1658] يُنظر: ((المعرَّب)) للجواليقي (ص: 255، 256)، ((لسان العَرب)) لابن منظور (3/ 138). ، وذَكَرَ أَدِّي شِير أنَّه يُمكِنُ أن يكونَ مِن الآراميَّةِ: (كوذيا) [1659] يُنظر: ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 48)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 168). ، وقيلَ: هو مِن السُّرْيانيَّةِ: (goudio) [1660] يُنظر: ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 168). .
جَوْزٌ:
الجَوْزُ: شَجَرٌ ثَمَرُه يُكسَرُ ويُؤكَلُ لُبُّه، وجاءَ قَديمًا في أمْثالِ العَربِ: (لأشْقَحَنَّك شَقْحَ [1661] الشَّقْحُ: الكسرُ. الجَوْزِ بالجَنْدَلِ).
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (گَوز) [1662] يُنظر: ((الجمهرة)) لابن دُرَيْدٍ (1/ 437)، ((المعرَّب)) للجواليقي (ص: 238)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 168). .
جَوْزِينَق:
الجَوْزِينَقُ والجَوْزينَجُ: نَوْعٌ مِن الحَلْوى يُصنَعُ بلُبِّ الجَوْزِ، واللُّغةُ الفَصيحةُ فيه: (الجَوْزينقُ).
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (گُوزينه) [1663] يُنظر: ((المعرَّب)) للجواليقي (ص: 239)، ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 48). .
جَوْسَقٌ:
الجَوْسَقُ: القَصْرُ، أو الحِصْنُ.
ذَكَرَ ابنُ دُرَيْدٍ وغيْرُه أنَّه مُعرَّبٌ مِن الفارِسيَّةِ: (كُوشَك)، بمَعْنى: الصَّغيرِ [1664] يُنظر: ((الجمهرة)) لابن دُرَيْدٍ (2/ 1174). ، وذَكَرَ أَدِّي شِير أنَّ أصْلَه الفارِسيَّ: (جوسه)، بمَعْنى: القَصْرِ [1665] يُنظر: ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 48). ، وذَكَرَ الدُّكْتورُ/ ف. عبْدُ الرَّحيمِ أنَّ أصْلَه الفارِسيَّ: (كُوشْك)، بمَعْنى البِناءِ العالي [1666] يُنظر: ((المعرَّب)) للجواليقي (ص: 237). .
جُوفيٌّ:
الجُوفيُّ والجوفِياءُ: نَوْعٌ مِن السَّمَكِ.
لم يُشِرْ إلى تَعْريبِه سِوى الجَواليقيِّ؛ قالَ: (وأحْسَبُهما مُعرَّبَينِ) [1667] ((المعرَّب)) للجواليقي (ص: 257). ، بلْ قالَ ابنُ دُرَيْدٍ: (والجُوفيُّ: ضَرْبٌ مِن حيتانِ البَحْرِ، عَربيٌّ مَعْروفٌ) [1668] ((الجمهرة)) لابن دُرَيْدٍ (1/ 489). .
جَوْقٌ:
الجَوْقُ: الجَماعةُ مِن النَّاسِ، واشْتَقُّوا مِنه فِعْلًا فقالوا: جَوَّقَهم تَجْويقًا، أي: جَمَعَهم، وتَجوَّقوا، أي: اجْتَمَعوا [1669] يُنظر: ((القاموس المحيط)) للفيروزابادي (ص: 872). .
ذَكَرَ ابنُ دُرَيْدٍ أنَّ اللَّفْظَ مُعرَّبٌ [1670] يُنظر: ((الجمهرة)) لابن دُرَيْدٍ (1/ 490). ، وهو مِن الفارِسيَّةِ: (جَوخ)، ومَعْناه: الجَماعةُ مِن النَّاسِ أو الحَيَوانِ [1671] يُنظر: ((المعرَّب)) للجواليقي (ص: 231)، ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 49). .
جَوَّا عامِّيٌّ:
جَوَّا: ما في الدَّاخِلِ، وجَوُّ البَيْتِ: داخِلُه، وجَوُّ كلِّ شيءٍ وجَوَّتُه: داخِلُه، والنِّسْبةُ إليه: الجَوَّانيُّ: الدَّاخِليُّ، وفي حَديثِ سَلْمانَ: ((إنَّ لكلِّ امْرِئٍ جَوَّانِيًّا وبَرَّانيًّا، فمَن أصْلَحَ جَوَّانِيَّه أصْلَحَ اللهُ بَرَّانِيَّه)) [1672] أخرجه أبو داود في ((الزهد)) (259) باختلافٍ يسيرٍ. ، أي: لكلِّ امْرِئٍ ظاهِرٌ وباطِنٌ، وقالَ أبو ذُؤَيْبٍ الهُذَلِيُّ:
يَجْري بجَوَّتِهِ مَوْجُ الفُراتِ كأنْضاحِ
الخزاعى حازَتْ رَنْقَه الرِّيحُ [1673] يُنظر: ((لسان العَرب)) لابن منظور (14/ 157).
لم تُشِرِ المَعاجِمُ القَديمةُ إلى كَوْنِه مُعرَّبًا، وهو مِن السُّرْيانيَّةِ: (gauo)، وتَعْني: داخِلَ البَطْنِ، و(gawoyo)، وتَعْني: (داخِليٌّ) [1674] يُنظر: ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 171). .
جِيرٌ:
الجِيرُ: الجِصُّ، واشْتَقُّوا مِنه الفِعلَ: جَيَّرَ، يَقولونَ: جَيَّرَ الحَوْضَ [1675] يُنظر: ((لسان العَرب)) لابن منظور (4/ 156، 157). .
واللَّفْظُ مِن السُّرْيانيَّةِ: (gairo)؛ بنفْسِ المَعْنى [1676] يُنظر: ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 172). .
