الفَصْلُ التَّاسِعَ عَشَرَ: حَرْفُ القافِ

محتويات الصفحة
انظر أيضا

قابوسُ:
قابوسُ: عَلَمٌ أعْجَميٌّ، وأبو قابوسَ: كُنْيةُ المَلِكِ النُّعْمانِ بنِ المُنذِرِ؛ قالَ النَّابِغةُ:
نُبِّئْتُ أنَّ أبا قابوسَ أوْعَدَني
ولا قَرارَ على زَأْرٍ مِن الأَسَدِ
وقالَ أيضًا:
فإن يَهلِكْ أبو قابوسَ يَهلِكْ
رَبيعُ النَّاسِ والبَلَدُ الحَرامُ
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (كاووس/ كاوس) [2199] يُنظر: ((الجمهرة)) لابن دُرَيْدٍ (3/ 1326). ، وهو اسمٌ لأحَدِ المُلوكِ الكيانيَّةِ، وغالِبًا ما يُذكَرُ قبْلَها: (كَيْ)، ومَعْناه: المَلِكُ أو الإمْبراطورُ، فتكونُ (كَيكاوُسْ) [2200] يُنظر: ((المعرَّب)) للجواليقي (ص: 499). .
قارٌ:
القارُ والقِيرُ: هو مادَّةٌ سَوْداءُ تُطْلى بها الإبِلُ إذا أُصيْبَتْ بالجَرَبِ، وتُطْلى به السُّفُنُ ليَمنَعَ الماءَ مِن التَّسَرُّبِ إليها، أو هو الزِّفْتُ، واشْتَقُّوا مِنه الفِعلَ: (قَيَّرْتُ السَّفينةَ)، أي: طَلَيْتُها بالقارِ [2201] يُنظر: ((لسان العَرب)) لابن منظور (5/ 124). ؛ قالَ النَّابِغةُ:
فلا تَتْرُكَّني بالوَعيدِ كأنَّني
إلى النَّاسِ مَطْليٌّ به القارُ أجْرَبُ
واللَّفْظُ مُعرَّبٌ مِن السُّرْيانيَّةِ: (قيرا)، وهو فيها مَأخوذٌ مِن اليونانيَّةِ: (كيروس)، بمَعْنى: الشَّمْعِ [2202] يُنظر: ((المعرَّب)) للجواليقي (ص: 510)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 362). .
قازوزةٌ:
القاقُزَّةُ والقاقوزةُ والقازوزةُ: إناءٌ مِن آنيةِ الشَّرابِ، قالَ النَّابِغةُ الجَعْديُّ:
كأنِّي إنَّما نادَمْتُ كِسْرى
فلي قاقُزَّةٌ ولهُ اثْنَتانِ
واللَّفْظُ مُعرَّبٌ [2203] يُنظر: ((التهذيب)) للأزهري (8/ 215). مِن السُّرْيانيَّةِ: (قزقزا)، بمَعْنى: الكُوبِ [2204] يُنظر: ((المعرَّب)) للجواليقي (ص: 523، 524). .
قُباذُ:
قُباذُ: اسمُ مَلِكٍ مِن مُلوكِ الفُرْسِ، وهو والِدُ كِسْرى أَنوشِرْوانَ، ويُقالُ: حِنْطةٌ قُباذيَّةٌ، بمَعْنى: عَتيقةٍ رَديئةٍ، كأنَّها مِن عَهْدِ المَلِكِ قُباذَ [2205] يُنظر: ((تاج العروس)) للزبيدي (9/ 455). .
وهو بالفارِسيَّةِ الحَديثةِ: (قُبادُ) [2206] يُنظر: ((المعرَّب)) للجواليقي (ص: 508). .
قَبَّانٌ:
القَبَّانُ: الميزانُ، ويُقالُ: فُلانٌ قَبَّانٌ على فُلانٍ، أي: أمينٌ عليه، ورَئيسُه الَّذي يَتْبَعُ أمْرَه، وإطْلاقُ القَبَّانِ على الميزانِ مَأخوذٌ مِن المَعْنى الثَّاني [2207] يُنظر: ((التهذيب)) للأزهري (9/ 154، 155)، ((الصحاح)) للجوهري (6/ 2179). ، وقَبَّنَ الشَّيءَ، أي: وَزَنَه بالقَبَّانِ.
واللَّفْظُ مُعرَّبٌ مِن الفارِسيَّةِ: (كپان) [2208] يُنظر: ((المعرَّب)) للجواليقي (ص: 526)، ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 124). .
قَبْجٌ:
القَبْجُ: الحَجَلُ، وهو طائِرٌ في حَجْمِ الحَمامِ، طَيِّبٌ لَحْمُه، أحْمَرُ المِنْقارِ والرِّجْلَينِ [2209] يُنظر: ((المُعجَم الوسيط)) الصادر عن مجمع اللُّغة العَربيَّة بالقاهرة (1/ 158). ، وهو الكَرَوانُ، والقَبْجةُ تُطلَقُ على الذَّكَرِ والأُنْثى، وللذَّكَرِ لَفْظٌ يَختَصُّ به وهو: يَعْقوبُ.
وهو مُعرَّبٌ مِن الفارِسيَّةِ، وذَكَرَ ابنُ مَنْظورٍ [2210] يُنظر: ((لسان العَرب)) لابن منظور (2/ 351). والزَّبيديُّ [2211] يُنظر: ((تاج العروس)) للزبيدي (6/ 168). أنَّ أصْلَه: (كبج)، وذَكَرَ أَدِّي شِير [2212] يُنظر: ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 123). وطوبيا العُنَيْسيُّ [2213] يُنظر: ((تفسير الألْفاظ الدَّخيلة)) لطوبيا العنيسي (ص: 55). والدُّكْتورُ/ ف. عبْدُ الرَّحيمِ [2214] يُنظر: ((المعرَّب)) للجواليقي (ص: 502). أنَّ أصْلَه: گبك.
القِبْطُ:
القِبْطُ: اسمٌ يُطلَقُ على أهْلِ مِصْرَ [2215] يُنظر: ((لسان العَرب)) لابن منظور (7/ 373). .
واللَّفْظُ مُعرَّبٌ مِن اليونانيَّةِ: (aigyptos) [2216] يُنظر: ((تفسير الألْفاظ الدَّخيلة)) لطوبيا العنيسي (ص: 55). .
قُبْطانٌ:
القُبْطانُ: رُبَّانُ السَّفينةِ، وقائِدُ الجَيْشِ، وعَميدُ القَوْمِ، وعُمْدةٌ.
واللَّفْظُ مُعرَّبٌ مِن الإيطاليَّةِ: (capitano) [2217] يُنظر: ((تفسير الألْفاظ الدَّخيلة)) لطوبيا العنيسي (ص: 55). .
قُرْطَقٌ:
القُرْطَقُ والقَرْطَقُ: نَوْعٌ مِن الثِّيابِ الفارِسيَّةِ، تُشبِهُ القَبَاءَ، ومِنه الفِعلُ: قَرْطَقْتُه فتَقَرْطَقَ، أي: ألْبَسْتُه القُرْطَقَ.
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (كُرْتَهْ)، وهو بالفَهْلَويَّةِ: (كرتك)، وهو أصْلُ اللَّفْظِ المُعرَّبِ [2218] يُنظر: ((المعرَّب)) للجواليقي (ص: 507، 508)، ((القاموس المحيط)) للفيروزابادي (ص: 920)، ((تاج العروس)) للزبيدي (26/ 336، 337). .
قِرْقِسٌ:
القِرْقِسُ: طينٌ أو شَمْعٌ يُسْتعمَلُ في الخَتْمِ، وهو أيضًا البَعوضُ أو البَقُّ، وفيه لُغةٌ: الجِرْجِسُ [2219] يُنظر: ((الجمهرة)) لابن دُرَيْدٍ (2/ 1162). .
اخْتُلِفَ في أيِّ اللُّغاتِ عُرِّبَ مِنها؛ فقيلَ: مِن الفارِسيَّةِ: (جرجشت)، وهو قَوْلُ ابنِ مَنْظورٍ [2220] يُنظر: ((لسان العَرب)) لابن منظور (6/ 173). ، وأَدِّي شِير [2221] يُنظر: ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 39). ، وقيلَ: هو مِن اليونانيَّةِ: (كركس)، وقيلَ: هو مِن السُّرْيانيَّةِ: (gargsa)، وهو مَوْجودٌ في البَراهينِ الحِسِّيَّةِ [2222] يُنظر: ((البراهين الحسية)) لاغناطيوس يعقوب الثالث (ص: 66). ، وأيَّدَ ذلك الرَّأيَ الدُّكْتورُ/ ف. عبْدُ الرَّحيمِ [2223] يُنظر: ((المعرَّب)) للجواليقي (ص: 517). .
قُرْقورٌ:
القُرْقورُ: نَوْعٌ مِن السُّفُنِ الَّتي تَكونُ طَويلةً؛ قالَ الرَّاجِزُ:
قُرْقورُ ساجٍ ساجُه مَطْليُّ
بالقيرِ والضَّبَّاتِ زَنْبَريُّ
ويُجمَعُ على قَراقيرَ؛ قالَ النَّابِغةُ:
قَراقيرَ النَّبيطِ على التِّلالِ [2224] يُنظر: ((التهذيب)) للأزهري (8/ 229).
ذَكَرَ ابنُ دُرَيْدٍ أنَّه عَربيٌّ [2225] يُنظر: ((الجمهرة)) لابن دُرَيْدٍ (1/ 199). ، والصَّحيحُ أنَّه مُعرَّبٌ مِن السُّرْيانيَّةِ: (قرقورا)، ومَعْناه: سَفينةٌ خَفيفةٌ [2226] يُنظر: ((المعرَّب)) للجواليقي (ص: 517، 518). .
قِرِلَّى:
القِرِلَّى: طائِرٌ يَصْطادُ السَّمَكَ، جِسْمُه صَغيرٌ، يَخْتطِفُ السَّمَكَ اخْتِطافًا سَريعًا، لا يكونُ إلَّا مُرَفْرِفًا على جانِبِ الماءِ، يَضَعُ عَيْنًا في الماءِ ليَلْتقِطَ فَريستَه، وعَيْنًا أُخرى في الهَواءِ حَذَرًا؛ لِذا يُضرَبُ به المَثَلُ: يَقولونَ: أحْزَمُ مِن قِرِلَّى، وأخْطَفُ مِن قِرِلَّى، وأحْذَرُ مِن قِرِلَّى، ويقولونَ: كنْ حَذِرًا كالقِرِلَّى؛ إن رأى خَيْرًا تَدَلَّى، وإن رأى شَرًّا تَوَلَّى [2227] يُنظر: ((التهذيب)) للأزهري (9/ 83)، ((مجمع الأمثال)) للميداني (1/ 228). .
وهو مُعرَّبٌ مِن السُّرْيانيَّةِ: (قورلى) [2228] يُنظر: ((المعرَّب)) للجواليقي (ص: 511). .
قِرْمِزٌ:
القِرْمِزُ: صِبْغٌ أحْمَرُ [2229] يُنظر: ((التهذيب)) للأزهري (9/ 297). .
وهو مِن الفارِسيَّةِ [2230] يُنظر: ((لسان العَرب)) لابن منظور (5/ 394). : (قِرْمِز) [2231] يُنظر: ((المعرَّب)) للجواليقي (ص: 519، 520)، ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 125). .
قِرْمِيدٌ:
القِرْمِيدُ والقَرْمَدُ: هو الآجُرُّ، وهو حِجارةٌ تُعرَّضُ للنِّيرانِ ثُمَّ تُسْتخدَمُ في طِلاءِ الأحْواضِ، وذَكَرَ الأصْمَعيُّ أنَّها تُسْتخدَمُ في طِلاءِ الحَمَّاماتِ خاصَّةً، والمُقَرْمَدُ: المَبنيُّ بالقِرْميدِ، ثُمَّ إنَّهم تَوَسَّعوا في مَدْلولِه ليَشمَلَ كلَّ ما يُطْلى للزِّينةِ، يقولونَ: ثَوْبٌ مُقَرْمَدٌ بالزَّعْفَرانِ والطِّيبِ، أي: مَطْليٌّ بهما، ومِنه قَوْلُ النَّابِغةِ يَصِفُ رَكَبَ امْرأةٍ:
رابي المَجَسَّةِ بالعَبيرِ مُقَرْمَدِ
أي: مَطْلِيٌّ بالزَّعْفَرانِ.
وهو مُعرَّبٌ مِن الرُّوميَّةِ: (كراميدا)، بمَعْنى: الآجُرِّ، ولمَّا أعْرَبوه قالوا: قَراميدُ، فوافَقَ بِناءً مِن أبْنِيةِ الجَمْعِ، فأخَذوا مِنه المُفرَدَ: قِرْميدٌ، وقَرْمَدٌ، ولِذا اشْتَهرَ الأصْلُ (قَراميد) على ألْسِنةِ العَوامِّ [2232] يُنظر: ((الجمهرة)) لابن دُرَيْدٍ (2/ 1109)، ((التهذيب)) للأزهري (9/ 305)، ((المعرَّب)) للجواليقي (ص: 493، 494). ؛ قالَ الزَّبيديُّ: (والمَشْهورُ على ألْسِنتِهم قَراميدُ) [2233] ((تاج العروس)) للزبيدي (9/ 32). .
قَرَنْفُلٌ:
القَرَنْفُلُ: نَباتٌ زَهْرُه طَيِّبُ الرَّائِحةِ، أصْلُ شَجَرتِه في الهِنْدِ؛ قالَ امْرُؤُ القَيْسِ:
إِذا الْتَفَتَتْ نَحْوي تَضَوَّعَ ريحُها
نَسيمَ الصَّبا جاءَتْ بِرَيَّا القَرَنْفُلِ
لم تُشِرِ المَعاجِمُ القَديمةُ إلى تَعْريبِه، وهو مُعرَّبٌ مِن الهِنْديَّةِ أو مِن اليونانيَّةِ: (karyofyllon) [2234] يُنظر: ((التهذيب)) للأزهري (9/ 310)، ((تفسير الألْفاظ الدَّخيلة)) لطوبيا العنيسي (ص: 56)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 368، 369). .
قَزٌّ:
القَزُّ: الحَريرُ الخامُ، والجَمْعُ: قُزوزٌ.
ذَكَرَ الجَوْهَريُّ [2235] يُنظر: ((الصحاح)) للجوهري (3/ 891). والأزْهَريُّ [2236] يُنظر: ((التهذيب)) للجوهري (8/ 214). وابنُ سِيدَهْ [2237] يُنظر: ((المحكم)) لابن سيده (6/ 107). أنَّه مُعرَّبٌ، وخالَفَ في ذلك ابنُ دُرَيْدٍ فذَكَرَ أنَّه عَربيٌّ [2238] يُنظر: ((الجمهرة)) لابن دُرَيْدٍ (1/ 130). ، والصَّحيحُ أنَّه مُعرَّبٌ مِن الفارِسيَّةِ: (كژ/ كج/ قز) [2239] يُنظر: ((المعرَّب)) للجواليقي (ص: 522، 523). .
قِسْطارٌ:
القِسْطارُ والقَسْطارُ والقَسْطريُّ والقَسْطَرُ: هو الصَّيْرَفيُّ الَّذي يَنْتقِدُ الدَّراهِمَ، واشْتَقُّوا مِنه الفِعلَ: قَسْطَرَ، بمَعْنى: انْتَقَدَ الدَّراهِمَ، وهو أيضًا الجِهْبِذُ، ومَن يَقومُ على شُؤونِ القَرْيةِ، وهو الجَسيمُ مِن النَّاسِ، ويُجمَعُ على: القَساطِرةِ؛ قالَ الشَّاعِرُ:
دَنانيرُنا مِن قَرْنِ ثَوْرٍ ولم تَكُنْ
مِن الذَّهَبِ المَضْروبِ عِنْدَ القَساطِرةْ [2240] يُنظر: ((التهذيب)) للأزهري (9/ 290)، ((لسان العَرب)) لابن منظور (5/ 93).
ذَكَرَتِ المَعاجِمُ أنَّه لُغةٌ لأهْلِ الشَّامِ، وهو مُعرَّبٌ مِن اللَّاتينيَّةِ: (quaestor)، وهو مُوظَّفٌ عنْدَ الرُّومانِ يَعمَلُ نائِبًا عامًّا وخازِنًا لأمْوالِ الدَّوْلةِ، ودَخَلَ إلى السُّرْيانيَّةِ: (قسطاور) بمَعْنى: الخازِنِ، أمَّا مَعْنى الصَّيْرَفيِّ، فإنَّ الصِّرافةَ في السُّرْيانيَّةِ: (قاسطيريا)، وبِهذا المَعْنى تَناوَلَتْه العَرَبُ، أمَّا مَعْنى "الرَّجُلِ الجَسيمِ" فلَعلَّ القَساطِرةَ مِن الرُّومِ كانَ لهم بَسْطةٌ في أجْسامِهم؛ فأُخِذَ مِنه ذلك المَعْنى [2241] يُنظر: ((المعرَّب)) للجواليقي (ص: 505). .
قَفَدانٌ:
القَفَدانُ: غِلافُ المُكْحُلةِ، والمُكْحُلةُ هي الوِعاءُ الَّذي يكونُ فيه الكُحْلُ [2242] يُنظر: ((تاج العروس)) للزبيدي (9/ 63). ، والقَفَدانةُ هي خَريطةُ العَطَّارِ الَّتي يَحمِلُ فيها أدَواتِه [2243] يُنظر: ((الجمهرة)) لابن دُرَيْدٍ (2/ 672). .
واللَّفْظُ مُعرَّبٌ مِن الفارِسيَّةِ: (كَفْدان)، وهي كَلِمةٌ مُركَّبةٌ مِن كَلِمتَينِ: (كفْ)، ومَعْناها: نَوْعٌ مُعيَّنٌ مِن الكُحْلِ، و(دان)، وهي لاحِقةٌ تُفيدُ في الفارِسيَّةِ مَعْنى الظَّرْفيَّةِ [2244] يُنظر: ((المعرَّب)) للجواليقي (ص: 504)، ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 126). .
قَفْش:
القَفْشُ: الخُفُّ مُطلَقًا، أو هو الخُفُّ المَقْطوعُ الَّذي لم تُحكَمْ صَنْعتُه، أو الخُفُّ القَصيرُ.
وهو مِن الفارِسيَّةِ، وأصْلُه -كما ذَكَرَ الأزْهَريُّ [2245] يُنظر: ((التهذيب)) للأزهري (8/ 262). ونَقَلَه عنه ابنُ مَنْظورٍ [2246] يُنظر: ((لسان العَرب)) لابن منظور (6/ 337). -: (كفج)، أو أصْلُه -كما ذَكَرَ الفَيْروزاباديُّ [2247] يُنظر: ((القاموس المحيط)) للفيروزابادي (ص: 603). -: (كفش)، وهو ما ذَكَرَه أَدِّي شِير [2248] يُنظر: ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 56). .
هذا، وقدْ ذَكَرَ الخَفاجيُّ مَعنًى آخَرَ لكَلِمةِ (القَفْش) لا صِلةَ له بالمَعْنى الأوَّلِ، وهو الكَلامُ الَّذي لا أصْلَ له [2249] يُنظر: ((شفاء الغليل)) للشهاب الخفاجي (ص: 211). ، والعامَّةُ في مِصْرَ يَنطِقونَ الكَلِمةَ بِهذا المَعْنى يقولونَ: (أَفْش) [2250] يُنظر: ((المعرَّب)) للجواليقي (ص: 513). .
قَفْشَليل:
القَفْشَليلُ والقَفْشَليلةُ: المِغْرَفةُ.
واللَّفْظُ مُعرَّبٌ مِن الفارِسيَّةِ، وذَكَرَ الأزْهَريُّ أنَّ أصْلَه: (كَفْجَلِين) [2251] يُنظر: ((التهذيب)) للأزهري (9/ 285). ، وعنْدَ ابنِ مَنْظورٍ أصْلُه: (كِبْجَلار) [2252] يُنظر: ((لسان العَرب)) لابن منظور (11/ 563). ، والصَّحيحُ أنَّ أصْلَه (كَفْچلِيز)، كما ذَكَرَ صاحِبُ القاموسِ لكنَّه كَتَبَه هكذا: (كَفْچهْ لِيز) [2253] يُنظر: ((القاموس المحيط)) للفيروزابادي (ص: 1049). ، وكذلك ذَكَرَ أَدِّي شِير [2254] يُنظر: ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 127). ، وهو لَفْظٌ مُركَّبٌ مِن كَلِمتَينِ: (كَفْچهْ)، بمَعْنى: المِلْعَقةِ، أو المِغْرَفةِ، و(ليز)، بمَعْنى: المِقبَضِ.
وقدْ ذَكَرَ ابنُ مَنْظورٍ (القَفْشَليلةَ)، وهي أيضًا مُعرَّبةٌ مِن (كَفْچلِيزَه)، وهي لُغةٌ في: (كَفْچلِيز) [2255] يُنظر: ((المعرَّب)) للجواليقي (ص: 489). .
قَفيزٌ:
القَفيزُ: مِكْيالٌ، ومِساحةٌ مَعْلومةٌ مِن الأرْضِ، والجَمْعُ: أقْفِزةٌ وقُفْزانٌ [2256] يُنظر: ((التهذيب)) للأزهري (8/ 330)، ((الصحاح)) للجوهري (3/ 892). .
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (كفيز) [2257] يُنظر: ((المعرَّب)) للجواليقي (ص: 526، 527). .
قَلْسٌ:
القَلْسُ: حَبْلٌ ضَخْمٌ يُتَّخَذُ مِن لِيفٍ أو خُوصٍ، وهو الحَبْلُ مِن حِبالِ السَّفينةِ [2258] يُنظر: ((الصحاح)) للجوهري (3/ 965). .
وهو مُعرَّبٌ مِن السُّرْيانيَّةِ: (قلسا)، وهو دَخيلٌ في السُّرْيانيَّةِ مِن اليونانيَّةِ: (kalos)، بمَعْنى: الحَبْلِ الضَّخْمِ [2259] يُنظر: ((المعرَّب)) للجواليقي (ص: 509، 510)، ((المُعجَم المفصل في المعرب والدَّخيل)) لسعدي ضناوي (ص: 374). .
قَلَّاشٌ عامِّيٌّ:
قَلَّاشٌ: الدَّاهيةُ المُحْتالُ، والَّذي يُعَمِّي على الأُمورِ.
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (قَلاش)، بمَعْنى: المُفلِسِ [2260] يُنظر: ((القاموس المحيط)) للفيروزابادي (ص: 603)، ((قصد السبيل)) لمحمد الأمين المحبي (2/ 360)، ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 127). .
قِمَطْرٌ:
القِمَطْرُ والقِمَطْرةُ: ما تُصانُ فيه الكُتُبُ، وهو بالتَّشْديدِ (القِمَّطرةُ) شاذٌّ [2261] يُنظر: ((الصحاح)) للجوهري (2/ 797). .
وهو مِن اليونانيَّةِ: (كمبترا)، بمَعْنى: الوِعاءِ والصُّنْدوقِ [2262] يُنظر: ((المعرَّب)) للجواليقي (ص: 509). .
قَمَنْجَرٌ:
القَمَنْجَرُ: القَوَّاسُ الَّذي يَبري القِسِيَّ، واشْتَقُّوا مِنه الفِعلَ: (قَمْجَرَ)، والمَصدَرُ: القَمْجَرةُ، وهي إصْلاحُ الشَّيءِ على إطْلاقِه، وهي في الأصْلِ تَغْطيةُ القَوْسِ بالغِراءِ مِن أجْلِ تَقْويتِها، يقولونَ: قَمْجَروا عليها؛ قالَ الرَّاجِزُ:
مِثلَ القِسِيِّ عاجَها القَمَنْجَرُ
ويُسمَّى هذا الغِطاءُ مِن الغِراءِ: (القِمْجارَ والغِمْجارَ) [2263] يُنظر: ((لسان العَرب)) لابن منظور (5/ 115، 116). ، وذَكَرَ الجَواليقيُّ أيضًا أنَّه مُعرَّبٌ مِن الفارِسيَّةِ [2264] يُنظر: ((المعرَّب)) للجواليقي (ص: 491). ، وذَكَرَ الأزْهَريُّ له مَعنًى آخَرَ، وهو غِلافُ السِّكِّينِ [2265] يُنظر: ((التهذيب)) للأزهري (9/ 282). ، فكأنَّه مُشتَقٌّ مِن المَعْنى الأوَّلِ.
واللَّفْظُ مُعرَّبٌ مِن الفارِسيَّةِ: (كَمانْ گر) [2266] يُنظر: ((التهذيب)) للأزهري (9/ 282). ، وهو مُركَّبٌ مِن كَلِمتَينِ: (كمان)، بمَعْنى: القَوْسِ، و(گر)، وهي لاحِقةٌ تُفيدُ مَعْنى الصَّانِعِ [2267] يُنظر: ((المعرَّب)) للجواليقي (ص: 491، 492). .
قِنَّارٌ:
القِنَّارُ والقِنَّارةُ: الخَشَبةُ الَّتي يُعلِّقُ عليها الجَزَّارُ اللَّحْمَ.
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (كَناره) [2268] يُنظر: ((المعرَّب)) للجواليقي (ص: 515)، ((لسان العَرب)) لابن منظور (5/ 120)، ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 130). .
قُنَّبِيطٌ:
القُنَّبِيطُ: أغْلَظُ أنْواعِ الكُرْنُبِ [2269] يُنظر: ((القاموس المحيط)) للفيروزابادي (ص: 684). ، وهو القَرْنَبيطُ أيضًا بلُغةِ أهْلِ مِصْرَ [2270] يُنظر: ((تاج العروس)) للزبيدي (20/ 56). ، ولا تَزالُ العامَّةُ تقولُها في مِصْرَ؛ قالَ الشَّاعِرُ:
لكنْ يَرَوْنَ البَصَلَ الحِرِّيفا
والقُنَّبِيطَ مُعْجِبًا طَريفا
ذَكَرَ الجَواليقيُّ أنَّ أصْلَه مِن النَّبَطيَّةِ، ورَجَّحَ الدُّكْتورُ/ ف. عبْدُ الرَّحيمِ أنَّ أصْلَه مِن اليونانيَّةِ: () [2271] يُنظر: ((المعرَّب)) للجواليقي (ص: 511، 512). .
قَنْدٌ:
القَنْدُ: عُصارةُ قَصَبِ السُّكَّرِ [2272] يُنظر: ((الصحاح)) للجوهري (2/ 528). ، وما يُعمَلُ مِنه السُّكَّرُ، والسُّكَّرُ مِن القَنْدِ كما السَّمْنُ مِن الزُّبْدِ [2273] يُنظر: ((المصباح المنير)) للفيومي (2/ 517). ، يقولونَ: سِويقٌ مَقْنُودٌ ومُقَنَّدٌ؛ قالَ الشَّاعِرُ:
أهاجَك أظْعانٌ رَحَلْنَ ونِسْوةٌ
بكَرْمانَ يُغْبَقْنَ السَّوِيقَ المُقَنَّدا [2274] يُنظر: ((الجمهرة)) لابن دُرَيْدٍ (2/ 677).
واللَّفْظُ مُعرَّبٌ مِن الفارِسيَّةِ: (كَنْدْ) [2275] يُنظر: ((المعرَّب)) للجواليقي (ص: 501). .
قَنْدَفيرٌ:
القَنْدَفيرُ: العَجوزُ.
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (گَنده پِير) [2276] يُنظر: ((الجمهرة)) لابن دُرَيْدٍ (2/ 1219)، ((التهذيب)) للأزهري (9/ 314)، ((المعرَّب)) للجواليقي (ص: 520، 521). .
قَنْدَفيلٌ:
القَنْدَفيلُ: الضَّخْمُ، والنَّاقةُ العَظيمةُ الرَّأسِ؛ قالَ المَخْروعُ السَّعْديُّ:
وتحتَ رَحْلي حُرَّةٌ ذَمولُ
مائِرةُ الضَّبْعَينِ قَنْدَفيلُ
للمَرْوِ في أخْفافِها صَليلُ
وفيه لُغةٌ: القَنْدَويل، وهو الرَّجُلُ العَظيمُ الرَّأسِ، والرَّجُلُ الطَّويلُ القَفا [2277] يُنظر: ((لسان العَرب)) لابن منظور (11/ 570). .
واللَّفْظُ مُعرَّبٌ مِن الفارِسيَّةِ: (گنده پيل) المُركَّبةِ مِن كَلِمتَينِ: (گنده)، بمَعْنى: الضَّخْمِ، و(پيل)، بمَعْنى: الفيلِ [2278] يُنظر: ((الصحاح)) للجوهري (5/ 1800)، ((المعرَّب)) للجواليقي (ص: 521). .
قُنْصُلٌ:
القُنْصُلُ: مُمَثِّلُ الدَّوْلةِ لدى دَوْلةٍ أُخرى يَحْمي حُقوقَها وتِجارتَها ويُدافِعُ عن رَعِيَّتِها [2279] يُنظر: ((المُعجَم الوسيط)) الصادر عن مجمع اللُّغة العَربيَّة بالقاهرة (2/ 762). .
وهو مُعرَّبٌ [2280] يُنظر: ((قصد السبيل)) لمحمد الأمين المحبي (2/ 366). مِن اللَّاتينيَّةِ: (consul)، بمَعْنى: المُسْتَشارِ [2281] يُنظر: ((تفسير الألْفاظ الدَّخيلة)) لطوبيا العنيسي (ص: 59). .
قِنْقِنٌ:
القِنْقِنُ والقُناقِنُ: المُهَنْدِسُ الَّذي يُقدِّرُ حَجْمَ الماءِ في باطِنِ الأرْضِ ويُقدِّرُ -مِن ثَمَّ- مَجاريَ القَنَواتِ، ويُجمَعُ على: (قَناقِنَ).
وهو مِن الفارِسيَّةِ: (كَنْ) وهو أمْرٌ مِن (كَندن)، بمَعْنى: الحَفْرِ [2282] يُنظر: ((الجمهرة)) لابن دُرَيْدٍ (1/ 220)، ((لسان العَرب)) لابن منظور (13/ 350). ، وذَكَرَ الدُّكْتورُ/ ف. عبْدُ الرَّحيمِ أنَّ كَلِمةَ (كن) لا تَأتي في الفارِسيَّةِ بمَعْنى المُهَنْدِسِ، وأنَّ أقْرَبَ لَفْظٍ لها بالفارِسيَّةِ (كم كم)، حِكايةً لصَوْتِ البِئْرِ وغيْرِها [2283] يُنظر: ((المعرَّب)) للجواليقي (ص: 500، 501). .
قَهْرَمان:
القَهْرَمانُ والقُهْرمانُ والقَرْهَمانُ: أمينُ المَلِكِ ومِن خاصَّةِ رِجالِه، وهو الوَكيلُ والقائِمُ بأمورِ مَن تحتَ يَدِه.
وهو مُعرَّبٌ مِن الفارِسيَّةِ [2284] يُنظر: ((لسان العَرب)) لابن منظور (12/ 496). وأصْلُه: (قَرَمان) [2285] يُنظر: ((قصد السبيل)) لمحمد الأمين المحبي (2/ 374). أو (كَهْرَمان) [2286] يُنظر: ((شفاء الغليل)) للشهاب الخفاجي (ص: 206). ، أو (قَهْرَمان)، بمَعْنى الآمِرِ صاحِبِ الحُكْمِ [2287] يُنظر: ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 130). .
قُوسٌ:
القُوسُ: صَوْمعةُ الرَّاهِبِ، أو رأسُ الصَّوْمعةِ، أو بَيْتُ الصَّيَّادِ؛ قالَ جَريرٌ:
لا وَصْلَ إذ صَرَفَتْ هِنْدٌ ولو وَقَفَتْ
لاستفْتَنَتْني وذا المِسْحَينِ في القُوسِ [2288] يُنظر: ((التهذيب)) للأزهري (9/ 178)، ((لسان العَرب)) لابن منظور (6/ 186).
ذَكَرَ الجَواليقيُّ أنَّه مُعرَّبٌ مِن الفارِسيَّةِ [2289] يُنظر: ((المعرَّب)) للجواليقي (ص: 533). : (كوشه) بمَعْنى: الزَّاويةِ، أو (كوچه)، بمَعْنى: الزُّقاقِ، وقدْ يكونُ مِن السُّرْيانيَّةِ، بمَعْنى: الرِّياضةِ والعُزْلةِ [2290] يُنظر: ((الألْفاظ الفارِسيَّة المعربة)) لأَدِّي شِير (ص: 130). .
قُوشٌ:
القُوشُ: الرَّجُلُ الضَّئيلُ الجِسْمِ؛ قالَ رُؤْبةُ:
في جِسْمِ شَخْتِ المَنْكِبَينِ قُوشِ
وهو مُعرَّبٌ مِن الفارِسيَّةِ: (كُوچك) [2291] يُنظر: ((الصحاح)) للجوهري (3/ 1017)، ((المعرَّب)) للجواليقي (ص: 496). .
قَوْمَسٌ:
القَوْمَسُ والقُمَّسُ: الأميرُ [2292] يُنظر: ((الجمهرة)) لابن دُرَيْدٍ (3/ 1324). ، وهو القائِدُ مِن قُوَّادِ الرُّومِ الَّذي يكونُ تحتَ رايتِه مِئتا رَجُلٍ [2293] يُنظر: ((القاموس المحيط)) للفيروزابادي (ص: 868). ، والرَّجُلُ الشَّريفُ، والجَمْعُ القَمامِسةُ [2294] يُنظر: ((العباب الزاخر)) للصغاني (1/ 172). .
قالَ الشَّاعِرُ:
وعَلِمْتُ أنِّي قد مُنِيتُ بنِئْطِلٍ
إِذْ قيلَ كانَ مِنَ الِ دَوْفَنَ قُمَّسُ
ذَكَرَ الصَّغانيُّ أنَّه مُعرَّبٌ مِن النَّبَطيَّةِ [2295] يُنظر: ((العباب الزاخر)) للصغاني (1/ 172). ، وذَكَرَ ابنُ دُرَيْدٍ أنَّه مُعرَّبٌ مِن الرُّوميَّةِ [2296] يُنظر: ((الجمهرة)) لابن دُرَيْدٍ (3/ 1324). ، وقدْ صَحَّحَ الدُّكْتورُ/ ف. عبْدُ الرَّحيمِ كِلا القَوْلَينِ، فذَكَرَ أنَّ أصْلَ اللَّفْظِ مِن اللَّاتينيَّةِ: (comes)، بمَعْنى: مُلازِمِ المَلِكِ والأميرِ، ودَخَلَ هذا اللَّفْظُ إلى اليونانيَّةِ، ومِنها تَسرَّبَ إلى السُّرْيانيَّةِ: (قوميس)، بمَعْنى: الرَّئيسِ والحاكِمِ [2297] يُنظر: ((المعرَّب)) للجواليقي (ص: 497، 498). .
قَيْصَرُ:
قَيْصَرُ: لَقَبُ مَلِكِ الرُّومِ؛ قالَ امْرُؤُ القَيْسِ:
بَكى صاحبي لمَّا رأى الدَّرْبَ دونَه
وأيْقَنَ أنَّا لاحِقانِ بقَيْصَرا
واللَّفْظُ مُعرَّبٌ مِن اللَّاتينيَّةِ: (caeser)، وهو في الأصْلِ لَقَبٌ للمَلِكِ الرُّومانيِّ يُولْيُوس، ثُمَّ أُطلِقَ بعْدَه على كُلِّ مَلِكٍ مِن مُلوكِ الرُّومانِ [2298] يُنظر: ((المعرَّب)) للجواليقي (ص: 518). .
قَيْطون:
القَيْطونُ: بَيْتٌ في بَيْتٍ، أو هو البَيْتُ الشَّتْويُّ، وهو غُرْفةُ النَّوْمِ، عَربيَّتُه: المُخْدَعُ [2299] يُنظر: ((الصحاح)) للجوهري (6/ 2183). ؛ قالَ عبْدُ الرَّحْمنِ بنُ حسَّانَ:
قُبَّةٌ مِن مَراجِلٍ ضَرَبَتْها
عِنْدَ بَرْدِ الشِّتاءِ في قَيْطونِ [2300] يُنظر: ((لسان العَرب)) لابن منظور (13/ 345).
ذَكَرَ الخَفاجيُّ [2301] يُنظر: ((شفاء الغليل)) للشهاب الخفاجي (ص: 209). والزَّبيديُّ [2302] يُنظر: ((تاج العروس)) للزبيدي (36/ 8). أنَّه مُعرَّبٌ مِن الرُّوميَّةِ، وهو قَوْلٌ صَحيحٌ؛ فاللَّفْظُ مُعرَّبٌ مِن اليونانيَّةِ: (كُويتون) بمَعْنى: غُرْفةِ النَّوْمِ، والسُّرْيانيُّ مِنه: (قَيْطونا)، بمَعْنى: غُرْفةِ النَّوْمِ أيضًا، وغُرْفةٍ داخليَّةٍ، وقِسْمِ النِّساءِ في البَيْتِ [2303] يُنظر: ((المعرَّب)) للجواليقي (ص: 520). .

  • الفَصْلُ الرَّابِعُ: حَرْفُ الثَّاءِ.
