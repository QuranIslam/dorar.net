محتويات الصفحة
انظر أيضا

الفَرْعُ الأوَّلُ: اشتراطُ أن يكونَ المَبِيعُ مَعلومًا للمُتعاقِدَينِ
يُشْتَرَطُ لصِحَّةِ البَيْعِ كَونُ المَبِيعِ مَعلومًا لكُلٍّ مِنَ العاقِدَينِ (البائِعِ والمُشتَري).
الأدِلَّةُ:
أوَّلًا: مِنَ الكِتابِ
قَولُه تعالى: لَا تَأْكُلُوا أَمْوَالَكُمْ بَيْنَكُمْ بِالْبَاطِلِ إِلَّا أَنْ تَكُونَ تِجَارَةً عَنْ تَرَاضٍ مِنْكُمْ [النساء: 29]
وَجْهُ الدَّلالةِ:
أنَّ الآيةَ فيها نَهيٌ عن أكلِ أموالِ النَّاسِ بالباطِلِ، ومِن ذلك إذا كانَ المَبيعُ مَجهولًا، وأيضًا فإنَّ قَولَه تعالى: عَنْ تَرَاضٍ يَدُلُّ على اشتِراطِ الرِّضا، والجَهالةُ قد تَتَنافى مع الرِّضا؛ لأنَّ أحَدَهُما قد لا يَرضى المَبيعَ إذا عَلِمَه ينظر: ((أحكام القرآن)) لابن العربي (1/521).
ثانيًا: مِنَ السُّنَّةِ
عن أبي هُرَيرةَ رَضِيَ اللهُ عنه قال: ((نَهى رَسولُ اللهِ صلَّى اللهُ عليه وسلَّم عن بَيعِ الحَصاةِ، وعن بَيْعِ الغَرَرِ )) أخرجه مسلم (1513)
وَجْهُ الدَّلالةِ:
أنَّ النَّبيَّ صلَّى اللهُ عليه وسلَّم نَهى عن بَيعِ الغَرَرِ، وبَيْعِ المَجهولِ غَرَرٌ؛ لأنَّه لا يُعرَفُ جِنسُه ولا نَوعُه ((المحلى)) لابن حزم (7/490) ((المهذب)) للشيرازي (2/14).
ثالِثًا: مِنَ الإجماعِ
نَقَل الإجْماعَ على ذلك: الشَّافِعيُّ قال الشَّافعيُّ: (البَيعُ في السُّنَّةِ والإجماعِ لا يجوزُ أن يكونَ مَجهولًا عند واحدٍ منهما، فإنْ لم يجُزْ بأن يجهَلَه أحدُ المتبايِعَينِ لم يجُزْ بأن يجهَلاه معًا) ((الأم)) (3/102). ، وابنُ العَرَبيِّ قال ابنُ العربي: (قد اتَّفَقَت الأمَّةُ على أنَّه لا يجوزُ إلَّا بَيعُ مَعلومٍ بمعلومٍ، بأيِّ طريقٍ مِن طرُقِ العِلْمِ وقع، وإنما اختلف العُلَماءُ في تفاصيلِ طُرُقِ العِلْم) ((المسالك في شرح موطأ مالك)) (6/30). ، وشَمسُ الدِّينِ ابنُ قُدامةَ قال شمسُ الدِّينِ ابنُ قُدامةَ: (بَيعُ المجهولِ لا يَصِحُّ إجماعًا) ((الشرح الكبير)) (6/5). ، والعَينيُّ قال العينيُّ: (لا خِلافَ لأحَدٍ أنَّ جَهالةَ المَبيعِ يمنَعُ الجوازَ) ((البناية)) (8/151).
رابعًا: لأنَّ جَهالةَ المَبيعِ تَمنَعُ مِن تَسليمِه وتَسَلُّمِه، وتُؤَدِّي إلى تَنازُعِ المُتَبايِعَينِ، ويَصيرُ العَقْدُ بها غَيرَ مُفيدٍ، وكُلُّ عَقْدٍ يُؤَدِّي إلى النِّزاعِ فاسِدٌ ((درر الحكام)) لعلي حيدر (1/178).
الفَرْعُ الثَّاني: رُؤيةُ المَبِيعِ
المَسألةُ الأُولى: الحاضِرُ المَرْئيُّ
يَجوزُ بَيْعُ الحاضِرِ المَرئيِّ.
الأدِلَّةُ:
أوَّلًا: مِنَ السُّنَّةِ
1 - عن أبي هُرَيرةَ رَضِيَ اللهُ عنه قال: ((نَهى رَسولُ اللهِ صلَّى اللهُ عليه وسلَّم عن بَيعِ الحَصاةِ، وعن بَيعِ الغَرَرِ)) أخرجه مسلم (1513).
وَجْهُ الدَّلالةِ:
أنَّ العَينَ إذا كانَت حاضِرةً مَرئيَّةً في مَجْلِسِ العَقدِ، فإنَّها تَكونُ قد سَلِمَتْ مِنَ الغَرَرِ الَّذي نَهى عنه النَّبيُّ صلَّى اللهُ عليه وسلَّم في الحَديثِ ((التمهيد)) لابن عبد البر (12/136).
2 - عن حَكيمِ بنِ حِزامٍ رَضِيَ اللهُ عنه، قال: ((قُلتُ: يا رَسولَ اللهِ، يَأتيني الرَّجُلُ يَسألُني البَيعَ ليسَ عِندي ما أبيعُه، ثُمَّ أبيعُه مِنَ السُّوقِ؟ فقال: لا تَبِعْ ما ليس عِندَك)) أخرجه أبو داود (3503)، والترمذي (1232)، والنسائي (4613)، وابن ماجه (2187)، وأحمد (15311) واللَّفظُ له. صَحَّحه ابنُ العربي في ((عارضة الأحوذي)) (3/193)، والنووي في ((المجموع)) (9/259)، وابن دقيق في ((الاقتراح)) (99)، وابن الملقن في ((البدر المنير)) (6/448)، وابن باز في ((مجموع الفتاوى)) (19/119)، والألباني في ((صحيح سنن ابن ماجه)) (2187)، وقال ابنُ القيم في ((زاد المعاد)) (5/716)، وابن حجر في ((تهذيب التهذيب)) (11/424): محفوظٌ.
وَجْهُ الدَّلالةِ:
أنَّ ما ليس عِندَ البائِعِ غَيرُ مَرئيٍّ للمُشتَري والبائِعِ، فيَدخُلُ غَيرُ المَرئيِّ في النَّهيِ عن بَيعِ ما ليس عِندَه ((الحاوي الكبير)) للماوردي (5/ 325).
ثانيًا: مِنَ الإجماعِ
نَقَل الإجْماعَ على ذلك: ابنُ حَزمٍ قال ابنُ حَزمٍ: (بَيعُ الحاضِرِ المرئيِّ المقلبِ بمِثْلِه أو بدنانيرَ أو دراهِمَ حاضِرةٍ مَقبوضةٍ، أو إلى أجَلٍ مُسَمًّى، أو حالةٍ في الذِّمَّة؛ فمُتَّفَقٌ على جوازِه) ((المحلى)) (7/214). ، وابنُ عَبدِ البَرِّ قال ابنُ عبدِ البَرِّ: (قال مالِكٌ: وأمَّا كُلُّ شَيءٍ كان حاضِرًا يُشترى على وَجْهِه؛ مِثلُ اللَّبَنِ إذا حُلِبَ، والرَّطبِ يُستجنى، فيأخُذُ المبتاعُ يومًا بيومٍ، فلا بأسَ به. قال أبو عُمَرَ: هذا لا خِلافَ فيه إذا اشترى على وَجْهِه بعد النَّظَرِ إليه وقد حُلِبَ اللَّبَنُ، وجُني التَّمرُ) ((الاستذكار)) (6/337). ، وابنُ رُشدٍ قال ابنُ رشد: (المبيعاتُ على نوعينِ: مَبِيعٌ حاضِرٌ مَرئيٌّ، فهذا لا خِلافَ في بَيْعِه) ((بداية المجتهد)) (3/174).
المَسألةُ الثَّانيةُ: الرُّؤيةُ المُتقَدِّمةُ على العَقدِ
تَجوزُ الرُّؤيةُ المُتَقدِّمةُ على العَقدِ إذا لم يَتَغَيَّرِ المَبيعُ في الغالِبِ كأرضٍ وحديدٍ ونُحاسٍ وآنيةٍ ونَحوِ ذلك. ، وذلك باتِّفاقِ المَذاهِبِ الفِقْهيَّةِ الأربعةِ: الحَنَفيَّةِ ((الفتاوى الهندية)) (6/281). ، والمالِكيَّةِ ((مختصر خليل)) (ص 145)، ((منح الجليل)) لعُلَيش (4/486). ، والأظهَرُ عند الشَّافِعيَّةِ ((تحفة المحتاج)) لابن حجر الهيتمي (4/264)، ((نهاية المحتاج)) للرَّملي (3/417). ، والحَنابِلةِ ((الإنصاف)) للمرداوي (4/213)، ((كشاف القناع)) للبهوتي (3/164).
وذلك للآتي:
أوَّلًا: أنَّ شَرْطَ الصِّحَّةِ العِلمُ، وقد حصل بطَريقِه، وهي الرُّؤيةُ المتقَدِّمةُ ((كشاف القناع)) للبهوتي (3/164).
ثانيًا: أنَّه جَرى العَمَلُ على تَبايُعِ العُمْيانِ مِن غَيرِ نَكيرٍ ما دامَ الأعمى يَعرِفُ المَبيعَ بما يَكفي عن رُؤيَتِه، وهَذا مِثلُه في المَعنى ينظر: ((العناية)) للبابرتي (6/348).
الفَرْعُ الثَّالثُ: وسائِلُ تَعْيينِ المَبِيعِ المرادُ بالتعيينِ هو تحديدُ السِّلعةِ، كأن يقولَ للمُشتري: بِعْتُك هذه السَّيارةَ، أو يُعَيِّنَ السِّلعةَ بالوَصْفِ وبيانِ المِقدارِ إذا كانت السِّلعةُ غائبةً.
المَسألةُ الأُولى: الإشارةُ
من وَسائِلِ تَعْيينِ المَبِيعِ الإشارةُ، وذلك باتِّفاقِ المَذاهِبِ الفِقْهيَّةِ الأربعةِ: الحَنَفيَّةِ ((الهداية)) للمَرْغِيناني (3/24)، ((العناية)) للبابرتي (6/259، 260). ، والمالِكيَّةِ ((حاشية العدوي على كفاية الطالب الرباني)) (2/194). ، والشَّافِعيَّةِ ((فتح العزيز)) للرافعي (8/432)، ((المجموع)) للنووي (9/310). ، والحَنابِلةِ ((الإقناع)) للحجاوي (2/69)، ((كشاف القناع)) للبهوتي (3/171). ، وذلك لأنَّ بالإشارةِ كِفايةً في التعريفِ، وترتَفِعُ به الجَهالةُ المُفْضِيةُ إلى المنازَعةِ ينظر: ((الهداية)) للمرغيناني (3/24) ((العناية)) للبابرتي (6/259).
المَسألةُ الثَّانية: الوَصْفُ
من وَسائِلِ تَعْيينِ المَبِيعِ الوَصْفُ إذا كان المَبِيعُ غائِبًا ينظر: مسألة: بَيْعُ الغائِبِ عن طريقِ الوَصْفِ المُزيلِ للجَهالةِ.
المَسألةُ الثَّالِثةُ: بَيانُ المِقدارِ
من وَسائِلِ تَعْيينِ المَبِيعِ بيانُ المِقْدارِ المرادُ به: بيانُ مِقدارِه بالوَزنِ أو الكَيلِ أو العَدَدِ. ، وذلك باتِّفاقِ المَذاهِبِ الفِقْهيَّةِ الأربعةِ: الحَنَفيَّةِ ((تبيين الحقائق)) للزَّيلعي (4/4، 5)، ((العناية)) للبابرتي (6/303). ، والمالِكيَّةِ ((التاج والإكليل)) للمواق (4/278)، ((منح الجليل)) لعُلَيش (4/465، 466). ، والشَّافِعيَّةِ ((روضة الطالبين)) للنووي (ص 103)، ((مغني المحتاج)) للشربيني (2/73). ، والحَنابِلةِ ((الفروع)) لابن مفلح (6/158)، ((كشاف القناع)) للبهوتي (3/176). ، وذلك لأنَّ جَهالةَ المِقْدارِ تُفْضي إلى النِّزاعِ المانِعِ مِن التسليمِ والتسَلُّمِ ((تبيين الحقائق)) للزيلعي (4/4، 5) ((المجموع)) للنووي (9/314).
الفَرْعُ الرَّابعُ: بَيْعُ الأعمى وشِراؤُه
يَصِحُّ بَيْعُ الأعمى وشِراؤُه في الجُملةِ، وهو مَذْهَبُ الجُمهورِ: الحَنَفيَّةِ يَصِحُّ عندهم مُطلقًا بالوَصْفِ وبدُونِه ((العناية)) للبابرتي (6/348)، ((الفتاوى الهندية)) (3/65). ، والمالِكيَّةِ يَصِحُّ بالوَصْفِ أو ما يقومُ مَقامَه. ((الكافي)) لابن عبد البر(2/731)، ((التاج والإكليل)) للمواق (4/294)، ((منح الجليل)) لعُلَيش (4/486). ، والحَنابِلةِ يَصِحُّ عند الحنابلةِ بَيعُ الأعمى بما يمكِنُه مَعرِفَتُه بغيرِ حاسَّةِ البَصَرِ؛ كشَمٍّ ولَمسٍ وذَوقٍ، ويَصِحُّ بَيْعُه وشراؤُه بالصِّفةِ ممَّا يَصِحُّ السَّلمُ فيه. ((الفروع)) لابن مفلح (6/144)، ((كشاف القناع)) للبهوتي (3/165). ، وقَولٌ عند الشَّافِعيَّةِ ((روضة الطالبين)) للنَّوَوي (3/370، 371)، ((المجموع)) للنووي (9/302، 303).
وذلك للآتي:
أوَّلًا: أنَّ العُمْيانَ في كُلِّ زَمانٍ مِن لَدُنْ رَسولِ اللهِ صلَّى اللهُ عليه وسلَّم لم يُمنَعوا مِن بياعاتِهم وأشرِيَتِهم، بَل بايَعوا في سائِرِ الأعصارِ مِن غَيرِ إنكارٍ ((بدائع الصنائع)) للكاساني (5/164).
ثانيًا: لأنَّ الأعمى يُمكِنُه الاطِّلاعُ على المقصودِ ومَعرِفَتُه، فأشبَهَ بَيْعَ البَصيرِ ((المغني)) لابن قدامة (4/159).
ثالِثًا: لأنَّ إشارةَ الأخرَسِ تقومُ مَقامَ نُطْقِه، فكذلك شَمُّ الأعمى وذَوقُه ((المغني)) لابن قدامة (4/159).
الفَرْعُ الخامِسُ: بَيْعُ الغائِبِ
المَسألةُ الأُولى: بَيْعُ الغائِبِ بدُونِ رُؤيةٍ أو وَصْفِ يُزيلُ الجَهالةَ
يَجوزُ بَيعُ الغائِبِ بدونِ رُؤيةٍ أو وَصْفٍ، ولِلمُشتَري الخيارُ إذا رَآه، وهو مَذْهَبُ الحَنَفيَّةِ   ((تبيين الحقائق)) للزَّيلعي (4/24، 25)، ((البحر الرائق)) لابن نجيم (6/28). ، والمالِكيَّةِ اشتَرَط المالِكيَّةُ لصِحَّةِ البَيعِ أن يُجعَلَ للمُشتري الخيارُ إذا رآه، فإنْ سَكَتَا عن شرطِ الخِيارِ فالبَيعُ فاسِدٌ. ((مواهب الجليل)) للحطَّاب (6/118). ، وقَولٌ عند الشَّافِعيَّةِ ((مغني المحتاج)) للشربيني (2/18)، ((حاشيتا قليوبي وعميرة)) (2/205). ، وروايةٌ عن أحمَدَ ((الإنصاف)) للمرداوي (4/213). وينظر: ((المغني)) لابن قدامة (3/494، 495). ، ونُقِلَ عن جُمهورِ العُلَماءِ قال النَّوويُّ: (في مذاهِبِ العُلَماءِ في بَيعِ العَينِ الغائبةِ... وقال مالِكٌ، وأبو حنيفةَ، وأحمدُ، وابنُ المُنذِرِ، وجمهورُ العُلَماءِ مِنَ الصَّحابةِ والتابعينَ ومَن بَعْدَهم: يَصِحُّ، نقله البَغَويُّ وغَيرُه عن أكثَرِ العُلَماءِ) ((المجموع)) (9/301). ، واختاره ابنُ تَيمِيَّةَ في أظهَرِ قَولَيه قال ابنُ تيميَّةَ: (فمن اشتَرَى ما لم يَرَه على أنَّه بالخيارِ إذا رآه فلا محذورَ في هذا البَيعِ أصلًا، بل الأظهَرُ أنَّه يَصِحُّ، كما هو إحدى الرِّوايتينِ عن أحمَدَ، ومَذهَبُ أبي حنيفةَ وغَيرِه؛ فإنَّ الصَّحابةَ كانوا يتبايَعونَ الأعيانَ الغائبةَ كما ثَبَت ذلك عنهم في عِدَّة قضايا، ولم يُعْرَفْ عن أحَدٍ مِنَ الصَّحابةِ أنَّه أنكر ذلك) ((نظرية العقد)) (ص: 225). وقال: (يجوزُ بَيْعُ الأعيانِ الغائبةِ بلا صفةٍ مع الخيارِ) ((الفتاوى الكبرى)) (4/22). وقال: (مِثْلُ بَيعِ الأعيانِ الغائبةِ: مِنَ الفُقَهاءِ مَن جَوَّز بَيْعَها مُطلَقًا وإن لم تُوصَفْ، ومنهم من منع بَيْعَها مع الوَصْفِ، ومالِكٌ جَوَّز بَيْعَها مع الصِّفةِ دونَ غَيرِها، وهذا أعدَلُ) ((مجموع الفتاوى)) (20/345)، و(29/306)، وينظر: ((المستدرك على مجموع الفتاوى)) (4/7)، ((الإنصاف)) للمرداوي (4/213).
الأدِلَّةُ:
أوَّلًا: مِنَ الآثارِ
عن سالِمٍ قال: قال ابنُ عُمَرَ رَضِيَ اللهُ عنهما: (كُنَّا إذا تَبايَعْنا كانَ كُلُّ واحِدٍ مِنَّا بالخِيارِ ما لم يَتَفَرَّقِ المُتَبايِعانِ. قال: فتَبايَعْتُ أنا وعُثمانُ، فبِعْتُه مالًا لي بالوادي بما لهُ بخَيبَرَ. قال: فلَمَّا بايَعْتُه طَفِقْتُ أنكِصَ على عَقِبي نَكْصَ القَهْقَرى؛ خَشيةَ أن يَتَرادَّني البَيعَ عُثْمانُ قَبلَ أن أُفارِقَه) أخرجه البخاريُّ معلَّقًا بصيغة الجزم بنحوه (2116)، وأخرجه موصولًا الطحاوي في ((شرح معاني الآثار)) (7327) واللَّفظُ له، والدارقطني (3/6). صَحَّحه ابنُ عبد البر في ((التمهيد)) (14/26)، وصَحَّح إسناده الألباني في ((إرواء الغليل)) (5/155).
وَجْهُ الدَّلالةِ:
أنَّ عُثمانَ بنَ عَفَّانَ، وعَبدَ اللهِ بنَ عُمَرَ رَضِيَ الله عنهم قد تَبايَعا ما هو غائِبٌ عنهما، ورَأيا ذلك جائِزًا، وذلك بحَضرةِ أصحابِ رَسولِ اللهِ صلَّى اللهُ عليه وسلَّم، فلم يُنكِرْه عليهما مُنكِرٌ   ((شرح معاني الآثار)) للطحاوي (4/363).
ثانيًا: لأنَّ الجَهالةَ فيه لا تُفْضي إلى المُنازَعةِ؛ لأنَّه إذ لم يوافِقْه رَدَّه، فصار كجَهالةِ الوَصفِ في المُشاهَدِ المُعايَنِ ((تبيين الحقائق)) للزيلعي (4/24).
ثالِثًا: لأنَّه عَقْدُ مُعاوَضةٍ، فلم تفتَقِرْ صِحَّتُه إلى رؤيةِ المعقودِ عليه، كالنِّكاحِ ((المغني)) لابن قدامة (3/ 495)، ((المجموع)) للنووي (9/ 288).
المَسألةُ الثَّانيةُ: بَيْعُ الغائِبِ عن طَريقِ الوَصْفِ المُزِيلِ للجَهالةِ كالوَصْفِ بما يُسَمَّى (الكتالوج).
يَجوزُ بَيْعُ العَينِ الغائِبةِ عن طَريقِ الوَصْفِ المُزيلِ للجَهالةِ، وهو مَذْهَبُ الجُمهورِ: الحَنَفيَّةِ   يَصِحُّ عند الحَنَفيَّة بيعُ الغائبِ بدونِ وَصفٍ، فمِن بابِ أَولى أن يَصِحَّ عِندَهم مع الوَصْفِ، ويكونُ الخِيارُ للمُشتري عند الرُّؤيةِ ولو بدونِ شَرطِ الخيارِ. ((تبيين الحقائق)) للزَّيلعي (4/24، 25)، ((البحر الرائق)) لابن نجيم (6/28). ، والمالِكيَّةِ ((مواهب الجليل)) للحطَّاب (6/118)، ((حاشية الدسوقي على الشرح الكبير)) (3/20). ، والحَنابِلةِ عند الحنابلةِ: يَصِحُّ البَيعُ إن كان ممَّا يَصِحُّ فيه السَّلَمُ. ((الإنصاف)) للمرداوي (4/213)، ((شرح منتهى الإرادات)) للبهوتي (2/12). ، وقَولٌ للشَّافِعيَّةِ ((مغني المحتاج)) للشربيني (2/18)، ((حاشيتا قليوبي وعميرة)) (2/205).
وذلك للآتي:
أوَّلًا: لأنَّ الصِّفةَ تَقومُ مَقامَ رُؤيةِ المَوصوفِ ((المقدمات الممهدات)) لابن رشد الجد (2/ 77)، ((نهاية المطلب)) للجويني (5/10).
ثانيًا: قياسًا على صِحَّةِ بَيْعِ الأعمى؛ فإنَّه يَبيعُ ويشتري بالوَصْفِ ((بدائع الصنائع)) للكاساني (5/164).
ثالثًا: لأنَّه نوعُ بَيْعٍ بالصِّفةِ، فيَصِحُّ كالسَّلَمِ   وهذا دليل الحنابلة. ((المغني)) لابن قدامة (3/495).
رابعًا: قياسًا على جَوازِ بَيعِ الحَبِّ في أكمامِه، فإذا جازَ بَيعُ الحَبِّ في أكمامِه، وهو غَيرُ مَرئيٍّ، جازَ أن يَشتَريَ منه إذا كانَ غائِبًا، على صِفةٍ؛ إذ لا فَرْقَ إذا غابَ المَبيعُ بَينَ أن يَبيعَه على الصِّفةِ أو على مِثالٍ يُريه إيَّاه ((المقدمات الممهدات)) لابن رشد الجد (2/ 78).
المَسألةُ الثَّالِثةُ: ضَمانُ المَبيعِ الغائِبِ
ضمانُ المَبِيعِ الغائِبِ كأن تتلَفَ السِّلعةُ الغائبةُ قبل أن يَقبِضَها المُشتري، فيكونَ ضَمانُها على البائِعِ. يكونُ على البائِعِ إجمالًا، وهو مَذْهَبُ الجُمهورِ: الحَنَفيَّةِ ((العناية)) للبابرتي (9/28)، ((البناية)) للعيني (8/46). ، والمالِكيَّةِ -في المشهورِ عِندَهم- استثنى المالِكيَّةُ إذا كان المَبيعُ عَقارًا وبِيعَ بالوَصْفِ يَضمَنُه المشتري بالعَقْدِ. ((التاج والإكليل)) للمواق (4/299)، ((منح الجليل)) لعُلَيش (5/233، 234) ((شرح الزرقاني على مختصر خليل)) (5/271) ((الشرح الكبير للدردير وحاشية الدسوقي)) (3/28). ، والحَنابِلةِ ((الإقناع)) للحجاوي (2/110)، ((كشاف القناع)) للبهوتي (3/242). ، وذلك لأنَّ بقاءَ سُلطةِ البائِعِ عليه تُفيدُ بَقاءَ ضَمانِه ((مغني المحتاج)) للشربيني (2/65).

انظر أيضا:

  • المَطلَبُ الأوَّلُ: كَونُ المَبيعِ مُباحَ الِانتِفاعِ به شَرعًا فلا يجوزُ بَيْعُ المحَرَّمِ: كالخَمرِ، والدُّخَانِ، وأشرِطةِ الفيديو المحرَّمةِ، وآلاتِ اللَّهْوِ المحرَّمِ، وغيرِها من المحَرَّماتِ .
  • المَطْلَبُ الثَّاني: كونُ المَبِيعِ مَوجودًا حالَ العَقْدِ .
  • المَطْلَبُ الثَّالثُ: كونُ المَبِيعِ طاهِرًا .
  • المَطْلَبُ الرَّابعُ: كَونُ المَبِيعِ مَقدورًا على تَسلِيمِه وغيرُ المقدورِ على تسليمِه؛ كبَيعِ الطَّيرِ في الهواءِ، والسَّمَكِ في الماءِ، والجَمَلِ الشَّاردِ، والسَّيَّارةِ الضَّائِعةِ، والمغصوبِ .
